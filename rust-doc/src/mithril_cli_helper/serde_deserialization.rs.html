6use serde::{Deserialize, Deserializer, de};

8/// Deserialize a value that can be either a string or a struct if it implements `FromStr`.
9///
10/// Useful for deserializing environment variables that can be passed as a JSON string or as a map.
11/// Ie: for the following enum:
12/// ```rust
13/// # use std::str::FromStr;
14///
15/// #[derive(Debug, serde::Deserialize)]
16/// #[serde(tag = "type", rename_all = "lowercase")]
17/// enum EnumParam {
18///    Foo { foo: String },
19/// }
20///
21/// impl FromStr for EnumParam {
22///   type Err = serde_json::Error;
23///
24///   fn from_str(s: &str) -> Result<Self, Self::Err> {
25///       serde_json::from_str(s)
26///   }
27/// }
28/// ```
29/// this allows `config` to deserialize the following environment variables:
30/// ```shell
31/// export ENUM_PARAM='{"type":"foo", "foo": "foo value"}'
32/// ```
33/// or
34/// ```shell
35/// export ENUM_PARAM__TYPE="foo"
36/// export ENUM_PARAM__FOO="foo value"
37/// ```
38///
39/// Code based on: <https://serde.rs/string-or-struct.html>
40pub fn string_or_struct<'de, T, D>(deserializer: D) -> Result<T, D::Error>
41where
42    T: Deserialize<'de> + FromStr,
43    D: Deserializer<'de>,
44    T::Err: fmt::Display,
45{
46    struct StringOrStruct<T>(PhantomData<fn() -> T>);
47
48    impl<'de, T> Visitor<'de> for StringOrStruct<T>
49    where
50        T: Deserialize<'de> + FromStr,
51        T::Err: fmt::Display,
52    {
53        type Value = T;
54
55        fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
56            formatter.write_str("string or map")
57        }
58
59        fn visit_str<E>(self, value: &str) -> Result<T, E>
60        where
61            E: Error,
62        {
63            FromStr::from_str(value).map_err(Error::custom)
64        }
65
66        fn visit_map<M>(self, map: M) -> Result<T, M::Error>
67        where
68            M: MapAccess<'de>,
69        {
70            Deserialize::deserialize(de::value::MapAccessDeserializer::new(map))
71        }
72    }
73
74    deserializer.deserialize_any(StringOrStruct(PhantomData))
75}

77#[cfg(test)]
78mod tests {
79    use config::Source;
80    use std::collections::HashMap;
81
82    use super::*;
83
84    #[derive(Debug, PartialEq, Deserialize)]
85    struct TestSettings {
86        #[serde(deserialize_with = "string_or_struct")]
87        enum_param: InternallyTaggedEnumParam,
88    }
89
90    #[derive(Debug, PartialEq, Deserialize)]
91    #[serde(tag = "type", rename_all = "lowercase")]
92    enum InternallyTaggedEnumParam {
93        Foo { foo: String },
94        Bar { bar: u16 },
95    }
96
97    impl FromStr for InternallyTaggedEnumParam {
98        type Err = serde_json::Error;
99
100        fn from_str(s: &str) -> Result<Self, Self::Err> {
101            serde_json::from_str(s)
102        }
103    }
104
105    fn read_config_from_source<T>(source: T) -> TestSettings
106    where
107        T: Source + Send + Sync + 'static,
108    {
109        config::Config::builder()
110            .add_source(source)
111            .build()
112            .unwrap()
113            .try_deserialize()
114            .unwrap()
115    }
116
117    #[test]
118    fn deserialize_config_from_single_string_in_environment_vars() {
119        let single_string_environment = HashMap::from([(
120            "ENUM_PARAM".to_string(),
121            r#"{"type":"foo", "foo": "foo value"}"#.to_string(),
122        )]);
123
124        let config: TestSettings = read_config_from_source(
125            config::Environment::default()
126                .source(Some(single_string_environment))
127                .separator("__"),
128        );
129
130        assert_eq!(
131            config.enum_param,
132            InternallyTaggedEnumParam::Foo {
133                foo: "foo value".to_string()
134            }
135        );
136    }
137
138    #[test]
139    fn deserialize_config_from_map_of_strings_in_environment_vars() {
140        let map_environment = HashMap::from([
141            ("ENUM_PARAM__TYPE".to_string(), "foo".to_string()),
142            ("ENUM_PARAM__FOO".to_string(), "foo value".to_string()),
143        ]);
144
145        let config: TestSettings = read_config_from_source(
146            config::Environment::default()
147                .source(Some(map_environment))
148                .separator("__"),
149        );
150
151        assert_eq!(
152            config.enum_param,
153            InternallyTaggedEnumParam::Foo {
154                foo: "foo value".to_string()
155            }
156        );
157    }
158
159    #[test]
160    fn deserialize_config_from_object_in_json_file() {
161        let json = r#"{"enum_param": {"type":"bar", "bar": 23452} }"#.to_string();
162
163        let config: TestSettings =
164            read_config_from_source(config::File::from_str(&json, config::FileFormat::Json));
165
166        assert_eq!(
167            config.enum_param,
168            InternallyTaggedEnumParam::Bar { bar: 23452 }
169        );
170    }
171
172    #[test]
173    fn deserialize_config_from_escaped_json_string_in_json_file() {
174        let json = r#"{"enum_param": "{\"type\":\"bar\", \"bar\": 23452}" }"#;
175
176        let config: TestSettings =
177            read_config_from_source(config::File::from_str(json, config::FileFormat::Json));

179        assert_eq!(
