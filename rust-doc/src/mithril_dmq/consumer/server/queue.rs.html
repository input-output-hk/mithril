11/// A queue for storing DMQ messages.
12pub(crate) struct MessageQueue {
13    messages: Mutex<VecDeque<DmqMessage>>,
14    new_message_notify: Notify,
15    timestamp_provider: Arc<dyn UnixTimestampProvider>,
16    max_size: usize,
17}
18
19impl MessageQueue {
20    /// The default maximum size of the message queue.
21    const MAX_SIZE_DEFAULT: usize = 10000;
22
23    /// Creates a new instance of [BlockingNonBlockingQueue].
24    pub fn new(max_size: usize, timestamp_provider: Arc<dyn UnixTimestampProvider>) -> Self {
25        Self {
26            messages: Mutex::new(VecDeque::new()),
27            new_message_notify: Notify::new(),
28            timestamp_provider,
29            max_size,
30        }
31    }
32
33    /// Cleans the queue
34    ///
35    /// Removes expired messages and ensures the queue does not exceed the maximum size.
36    async fn clean_queue(&self) {
37        let mut message_queue_guard = self.messages.lock().await;
38        // Remove expired messages from the front of the queue
39        // There may be other expired messages in the queue, but they will be removed on dequeue
40        // This avoids full scan of the queue.
41        while let Some(message) = message_queue_guard.front()
42            && self.has_message_expired(message).unwrap_or(false)
43        {
44            message_queue_guard.pop_front();
45        }
46
47        while message_queue_guard.len() > self.max_size {
48            message_queue_guard.pop_front();
49        }
50    }
51
52    /// Checks if a message has expired.
53    fn has_message_expired(&self, message: &DmqMessage) -> StdResult<bool> {
54        let current_timestamp: u32 = self.timestamp_provider.current_timestamp()?.try_into()?;
55        Ok(message.msg_payload.expires_at < current_timestamp)
56    }
57
58    /// Enqueues a new message into the queue.
59    pub async fn enqueue(&self, message: DmqMessage) {
60        {
61            // Run in a block to avoid Mutex deadlock in clean_queue
62            let mut message_queue_guard = self.messages.lock().await;
63            (*message_queue_guard).push_back(message);
64        }
65        self.clean_queue().await;
66
67        self.new_message_notify.notify_waiters();
68    }
69
70    /// Returns the messages from the queue in a non blocking way, if available.
71    pub async fn dequeue_non_blocking(&self, limit: Option<usize>) -> Vec<DmqMessage> {
72        self.clean_queue().await;
73        let mut message_queue_guard = self.messages.lock().await;
74        let limit = limit.unwrap_or((*message_queue_guard).len());
75        let mut messages = Vec::new();
76        for _ in 0..limit {
77            if let Some(message) = (*message_queue_guard).pop_front()
78                && !self.has_message_expired(&message).unwrap_or(false)
79            {
80                messages.push(message);
81            }
82        }
83
84        messages
85    }
86
87    /// Returns the messages from the queue in a blocking way, waiting for new messages if necessary.
88    pub async fn dequeue_blocking(&self, limit: Option<usize>) -> Vec<DmqMessage> {
89        loop {
90            let messages = self.dequeue_non_blocking(limit).await;
91            if !messages.is_empty() {
92                return messages;
93            }
94
95            self.new_message_notify.notified().await;
96        }
97    }
98
99    /// Checks if the message queue is empty.
100    pub async fn is_empty(&self) -> bool {
101        self.len().await == 0
102    }
103
104    /// Get the length of the message queue.
105    pub async fn len(&self) -> usize {
106        let message_queue_guard = self.messages.lock().await;
107        (*message_queue_guard).len()
108    }
109}
110
111impl Default for MessageQueue {
112    fn default() -> Self {
113        Self::new(
114            Self::MAX_SIZE_DEFAULT,
115            Arc::new(SystemUnixTimestampProvider),
116        )
117    }
118}
