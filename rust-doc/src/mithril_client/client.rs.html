39/// Options that can be used to configure the client.
40#[derive(Debug, Clone, Default, Serialize, Deserialize)]
41pub struct ClientOptions {
42    /// HTTP headers to include in the client requests.
43    pub http_headers: Option<HashMap<String, String>>,
44
45    /// Tag to retrieve the origin of the client requests.
46    #[cfg(target_family = "wasm")]
47    #[cfg_attr(target_family = "wasm", serde(default))]
48    pub origin_tag: Option<String>,
49
50    /// Whether to enable unstable features in the WASM client.
51    #[cfg(target_family = "wasm")]
52    #[cfg_attr(target_family = "wasm", serde(default))]
53    pub unstable: bool,
54
55    /// Whether to enable certificate chain verification caching in the WASM client.
56    ///
57    /// `unstable` must be set to `true` for this option to have any effect.
58    ///
59    /// DANGER: This feature is highly experimental and insecure, and it must not be used in production
60    #[cfg(target_family = "wasm")]
61    #[cfg_attr(target_family = "wasm", serde(default))]
62    pub enable_certificate_chain_verification_cache: bool,
63
64    /// Duration in seconds of certificate chain verification cache in the WASM client.
65    ///
66    /// Default to one week (604800 seconds).
67    ///
68    /// `enable_certificate_chain_verification_cache` and `unstable` must both be set to `true`
69    /// for this option to have any effect.
70    #[cfg(target_family = "wasm")]
71    #[cfg_attr(target_family = "wasm", serde(default = "one_week_in_seconds"))]
72    pub certificate_chain_verification_cache_duration_in_seconds: u32,
73}
74
75impl ClientOptions {
76    /// Instantiate a new [ClientOptions].
77    pub fn new(http_headers: Option<HashMap<String, String>>) -> Self {
78        Self {
79            http_headers,
80            #[cfg(target_family = "wasm")]
81            origin_tag: None,
82            #[cfg(target_family = "wasm")]
83            unstable: false,
84            #[cfg(target_family = "wasm")]
85            enable_certificate_chain_verification_cache: false,
86            #[cfg(target_family = "wasm")]
87            certificate_chain_verification_cache_duration_in_seconds: one_week_in_seconds(),
88        }
89    }
90
91    /// Enable unstable features in the WASM client.
92    #[cfg(target_family = "wasm")]
93    pub fn with_unstable_features(self, unstable: bool) -> Self {
94        Self { unstable, ..self }
95    }

217    /// Returns a `Client` that uses the dependencies provided to this `ClientBuilder`.
218    ///
219    /// The builder will try to create the missing dependencies using default implementations
220    /// if possible.
221    pub fn build(self) -> MithrilResult<Client> {
222        let logger = self
223            .logger
224            .clone()
225            .unwrap_or_else(|| Logger::root(slog::Discard, o!()));
226
227        let feedback_sender = FeedbackSender::new(&self.feedback_receivers);
228
229        let aggregator_client = match self.aggregator_client {
230            None => Arc::new(self.build_aggregator_client(logger.clone())?),
231            Some(client) => client,
232        };
233
234        let certificate_verifier = match self.certificate_verifier {
235            None => Arc::new(
236                MithrilCertificateVerifier::new(
237                    aggregator_client.clone(),
238                    &self.genesis_verification_key,
239                    feedback_sender.clone(),
240                    #[cfg(feature = "unstable")]
241                    self.certificate_verifier_cache,
242                    logger.clone(),
243                )
244                .with_context(|| "Building certificate verifier failed")?,
245            ),
246            Some(verifier) => verifier,
247        };
248        let certificate_client = Arc::new(CertificateClient::new(
249            aggregator_client.clone(),
250            certificate_verifier,
251            logger.clone(),
252        ));
253
254        let mithril_stake_distribution_client = Arc::new(MithrilStakeDistributionClient::new(
255            aggregator_client.clone(),
256        ));
257
258        #[cfg(feature = "fs")]
259        let http_file_downloader = match self.http_file_downloader {
260            None => Arc::new(RetryDownloader::new(
261                Arc::new(
262                    HttpFileDownloader::new(feedback_sender.clone(), logger.clone())
263                        .with_context(|| "Building http file downloader failed")?,
264                ),
265                FileDownloadRetryPolicy::default(),
266            )),
267            Some(http_file_downloader) => http_file_downloader,
268        };
269
270        #[cfg(feature = "fs")]
271        let ancillary_verifier = match self.ancillary_verification_key {
272            None => None,
273            Some(verification_key) => Some(Arc::new(AncillaryVerifier::new(
274                verification_key
275                    .try_into()
276                    .with_context(|| "Building ancillary verifier failed")?,
277            ))),
278        };
279
280        let snapshot_client = Arc::new(SnapshotClient::new(
281            aggregator_client.clone(),
282            #[cfg(feature = "fs")]
283            http_file_downloader.clone(),
284            #[cfg(feature = "fs")]
285            ancillary_verifier.clone(),
286            #[cfg(feature = "fs")]
287            feedback_sender.clone(),
288            #[cfg(feature = "fs")]
289            logger.clone(),
290        ));
291
292        #[cfg(feature = "unstable")]
293        let cardano_database_client = Arc::new(CardanoDatabaseClient::new(
294            aggregator_client.clone(),
295            #[cfg(feature = "fs")]
296            http_file_downloader,
297            #[cfg(feature = "fs")]
298            ancillary_verifier,
299            #[cfg(feature = "fs")]
300            feedback_sender,
301            #[cfg(feature = "fs")]
302            logger,
303        ));
304
305        let cardano_transaction_client =
306            Arc::new(CardanoTransactionClient::new(aggregator_client.clone()));
307
308        let cardano_stake_distribution_client =
309            Arc::new(CardanoStakeDistributionClient::new(aggregator_client));
310
311        Ok(Client {
312            certificate_client,
313            mithril_stake_distribution_client,
314            snapshot_client,
315            #[cfg(feature = "unstable")]
316            cardano_database_client,
317            cardano_transaction_client,
318            cardano_stake_distribution_client,
319        })
320    }

322    fn build_aggregator_client(
323        &self,
324        logger: Logger,
325    ) -> Result<AggregatorHTTPClient, anyhow::Error> {
326        let endpoint = self
327            .aggregator_endpoint.as_ref()
328            .ok_or(anyhow!("No aggregator endpoint set: \
329                    You must either provide an aggregator endpoint or your own AggregatorClient implementation"))?;
330        let endpoint_url = Url::parse(endpoint).with_context(|| {
331            format!("Invalid aggregator endpoint, it must be a correctly formed url: '{endpoint}'")
332        })?;
333
334        let headers = self.compute_http_headers();
335
336        AggregatorHTTPClient::new(
337            endpoint_url,
338            APIVersionProvider::compute_all_versions_sorted()
339                .with_context(|| "Could not compute aggregator api versions")?,
340            logger,
341            Some(headers),
342        )
343        .with_context(|| "Building aggregator client failed")
344    }

346    fn compute_http_headers(&self) -> HashMap<String, String> {
347        let mut headers = self.options.http_headers.clone().unwrap_or_default();
348        if let Some(origin_tag) = self.origin_tag.clone() {
349            headers.insert(MITHRIL_ORIGIN_TAG_HEADER.to_string(), origin_tag);
350        }
351        if let Some(client_type) = self.client_type.clone() {
352            headers.insert(MITHRIL_CLIENT_TYPE_HEADER.to_string(), client_type);
353        } else if !headers.contains_key(MITHRIL_CLIENT_TYPE_HEADER) {
354            headers.insert(
355                MITHRIL_CLIENT_TYPE_HEADER.to_string(),
356                DEFAULT_CLIENT_TYPE.to_string(),
357            );
358        }
359
360        headers
361    }
