35/// Options that can be used to configure the client.
36#[derive(Debug, Clone, Default, Serialize, Deserialize)]
37pub struct ClientOptions {
38    /// HTTP headers to include in the client requests.
39    pub http_headers: Option<HashMap<String, String>>,
40
41    /// Tag to retrieve the origin of the client requests.
42    #[cfg(target_family = "wasm")]
43    #[cfg_attr(target_family = "wasm", serde(default))]
44    pub origin_tag: Option<String>,
45
46    /// Whether to enable unstable features in the WASM client.
47    #[cfg(target_family = "wasm")]
48    #[cfg_attr(target_family = "wasm", serde(default))]
49    pub unstable: bool,
50
51    /// Whether to enable certificate chain verification caching in the WASM client.
52    ///
53    /// `unstable` must be set to `true` for this option to have any effect.
54    ///
55    /// DANGER: This feature is highly experimental and insecure, and it must not be used in production
56    #[cfg(target_family = "wasm")]
57    #[cfg_attr(target_family = "wasm", serde(default))]
58    pub enable_certificate_chain_verification_cache: bool,
59
60    /// Duration in seconds of certificate chain verification cache in the WASM client.
61    ///
62    /// Default to one week (604800 seconds).
63    ///
64    /// `enable_certificate_chain_verification_cache` and `unstable` must both be set to `true`
65    /// for this option to have any effect.
66    #[cfg(target_family = "wasm")]
67    #[cfg_attr(target_family = "wasm", serde(default = "one_week_in_seconds"))]
68    pub certificate_chain_verification_cache_duration_in_seconds: u32,
69}
70
71impl ClientOptions {
72    /// Instantiate a new [ClientOptions].
73    pub fn new(http_headers: Option<HashMap<String, String>>) -> Self {
74        Self {
75            http_headers,
76            #[cfg(target_family = "wasm")]
77            origin_tag: None,
78            #[cfg(target_family = "wasm")]
79            unstable: false,
80            #[cfg(target_family = "wasm")]
81            enable_certificate_chain_verification_cache: false,
82            #[cfg(target_family = "wasm")]
83            certificate_chain_verification_cache_duration_in_seconds: one_week_in_seconds(),
84        }
85    }

87    /// Enable unstable features in the WASM client.
88    #[cfg(target_family = "wasm")]
89    pub fn with_unstable_features(self, unstable: bool) -> Self {
90        Self { unstable, ..self }
91    }

185    /// Use [ClientBuilder::aggregator] if you don't need to set a custom [AggregatorClient]
186    /// to request data from the aggregator.
187    pub fn new(genesis_verification_key: &str) -> ClientBuilder {
188        Self {

191            origin_tag: None,
192            aggregator_client: None,
193            certificate_verifier: None,
194            #[cfg(feature = "fs")]
195            http_file_downloader: None,
196            #[cfg(feature = "unstable")]

204    /// Returns a `Client` that uses the dependencies provided to this `ClientBuilder`.
205    ///
206    /// The builder will try to create the missing dependencies using default implementations
207    /// if possible.
208    pub fn build(self) -> MithrilResult<Client> {
209        let logger = self
210            .logger
211            .clone()
212            .unwrap_or_else(|| Logger::root(slog::Discard, o!()));
213
214        let feedback_sender = FeedbackSender::new(&self.feedback_receivers);
215
216        let aggregator_client = match self.aggregator_client {
217            None => Arc::new(self.build_aggregator_client(logger.clone())?),
218            Some(client) => client,
219        };
220
221        let certificate_verifier = match self.certificate_verifier {
222            None => Arc::new(
223                MithrilCertificateVerifier::new(
224                    aggregator_client.clone(),
225                    &self.genesis_verification_key,
226                    feedback_sender.clone(),
227                    #[cfg(feature = "unstable")]
228                    self.certificate_verifier_cache,
229                    logger.clone(),
230                )
231                .with_context(|| "Building certificate verifier failed")?,
232            ),
233            Some(verifier) => verifier,
234        };
235        let certificate_client = Arc::new(CertificateClient::new(
236            aggregator_client.clone(),
237            certificate_verifier,
238            logger.clone(),
239        ));
240
241        let mithril_stake_distribution_client = Arc::new(MithrilStakeDistributionClient::new(
242            aggregator_client.clone(),
243        ));
244
245        #[cfg(feature = "fs")]
246        let http_file_downloader = match self.http_file_downloader {
247            None => Arc::new(RetryDownloader::new(
248                Arc::new(
249                    HttpFileDownloader::new(feedback_sender.clone(), logger.clone())
250                        .with_context(|| "Building http file downloader failed")?,
251                ),
252                FileDownloadRetryPolicy::default(),
253            )),
254            Some(http_file_downloader) => http_file_downloader,
255        };
256
257        let snapshot_client = Arc::new(SnapshotClient::new(
258            aggregator_client.clone(),
259            #[cfg(feature = "fs")]
260            http_file_downloader.clone(),
261            #[cfg(feature = "fs")]
262            feedback_sender.clone(),
263            #[cfg(feature = "fs")]
264            logger.clone(),
265        ));
266
267        #[cfg(feature = "unstable")]
268        let cardano_database_client = Arc::new(CardanoDatabaseClient::new(
269            aggregator_client.clone(),
270            #[cfg(feature = "fs")]
271            http_file_downloader,
272            #[cfg(feature = "fs")]
273            feedback_sender,
274            #[cfg(feature = "fs")]
275            logger,
276        ));
277
278        let cardano_transaction_client =
279            Arc::new(CardanoTransactionClient::new(aggregator_client.clone()));
280
281        let cardano_stake_distribution_client =
282            Arc::new(CardanoStakeDistributionClient::new(aggregator_client));
283
284        Ok(Client {
285            certificate_client,
286            mithril_stake_distribution_client,
287            snapshot_client,
288            #[cfg(feature = "unstable")]
289            cardano_database_client,
290            cardano_transaction_client,
291            cardano_stake_distribution_client,
292        })
293    }

295    fn build_aggregator_client(
296        &self,
297        logger: Logger,
298    ) -> Result<AggregatorHTTPClient, anyhow::Error> {
299        let endpoint = self
300            .aggregator_endpoint.as_ref()
301            .ok_or(anyhow!("No aggregator endpoint set: \
302                    You must either provide an aggregator endpoint or your own AggregatorClient implementation"))?;
303        let endpoint_url = Url::parse(endpoint).with_context(|| {
304            format!("Invalid aggregator endpoint, it must be a correctly formed url: '{endpoint}'")
305        })?;
306
307        let headers = self.compute_http_headers();
308
309        AggregatorHTTPClient::new(
310            endpoint_url,
311            APIVersionProvider::compute_all_versions_sorted()
312                .with_context(|| "Could not compute aggregator api versions")?,
313            logger,
314            Some(headers),
315        )
316        .with_context(|| "Building aggregator client failed")
317    }
318
319    fn compute_http_headers(&self) -> HashMap<String, String> {
320        let mut headers = self.options.http_headers.clone().unwrap_or_default();
321        if let Some(origin_tag) = self.origin_tag.clone() {
322            headers.insert(MITHRIL_ORIGIN_TAG_HEADER.to_string(), origin_tag);
323        }
324
325        headers
326    }

437    async fn test_with_origin_tag_not_overwrite_other_client_options_attributes() {
438        let builder = ClientBuilder::new("")
439            .with_options(ClientOptions { http_headers: None })
440            .with_origin_tag(Some("TEST".to_string()));
441        assert_eq!(None, builder.options.http_headers);
442        assert_eq!(Some("TEST".to_string()), builder.origin_tag);
443
444        let http_headers = HashMap::from([("Key".to_string(), "Value".to_string())]);
445        let builder = ClientBuilder::new("")

450        assert_eq!(Some(http_headers), builder.options.http_headers);
451        assert_eq!(Some("TEST".to_string()), builder.origin_tag);
