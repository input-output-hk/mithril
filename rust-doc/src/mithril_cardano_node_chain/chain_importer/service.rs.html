18/// Import and store Cardano chain data ([CardanoTransaction]).
19#[derive(Clone)]
20pub struct CardanoChainDataImporter {
21    block_scanner: Arc<dyn BlockScanner>,
22    transaction_store: Arc<dyn ChainDataStore>,
23    last_polled_point: Arc<Mutex<Option<RawCardanoPoint>>>,
24    logger: Logger,
25}
26
27impl CardanoChainDataImporter {
28    /// Constructor
29    pub fn new(
30        block_scanner: Arc<dyn BlockScanner>,
31        transaction_store: Arc<dyn ChainDataStore>,
32        logger: Logger,
33    ) -> Self {
34        Self {
35            block_scanner,
36            transaction_store,
37            last_polled_point: Arc::new(Mutex::new(None)),
38            logger: logger.new_with_component_name::<Self>(),
39        }
40    }
41
42    async fn start_point(
43        &self,
44        highest_stored_chain_point: &Option<ChainPoint>,
45    ) -> StdResult<Option<RawCardanoPoint>> {
46        let last_polled_point = self.last_polled_point.lock().await.clone();
47        if last_polled_point.is_none() {
48            debug!(
49                self.logger,
50                "No last polled point available, falling back to the highest stored chain point"
51            );
52        }
53
54        Ok(last_polled_point.or(highest_stored_chain_point.as_ref().map(RawCardanoPoint::from)))
55    }
56
57    async fn import_transactions(&self, up_to_beacon: BlockNumber) -> StdResult<()> {
58        let highest_stored_beacon = self.transaction_store.get_highest_beacon().await?;
59        let from = self.start_point(&highest_stored_beacon).await?;
60
61        if highest_stored_beacon
62            .as_ref()
63            .is_some_and(|f| f.block_number >= up_to_beacon)
64        {
65            debug!(
66                self.logger,
67                "No need to retrieve Cardano transactions, the database is up to date for block_number '{up_to_beacon}'",
68            );
69
70            Ok(())
71        } else {
72            debug!(
73                self.logger, "Retrieving Cardano transactions until block numbered '{up_to_beacon}'";
74                "starting_slot_number" => ?from.as_ref().map(|c| c.slot_number),
75                "highest_stored_block_number" => ?highest_stored_beacon.as_ref().map(|c| c.block_number),
76            );
77
78            self.parse_and_store_transactions_not_imported_yet(from, up_to_beacon)
79                .await
80        }
81    }
82
83    async fn parse_and_store_transactions_not_imported_yet(
84        &self,
85        from: Option<RawCardanoPoint>,
86        until: BlockNumber,
87    ) -> StdResult<()> {
88        let mut streamer = self.block_scanner.scan(from, until).await?;
89
90        while let Some(blocks) = streamer.poll_next().await? {
91            match blocks {
92                ChainScannedBlocks::RollForwards(forward_blocks) => {
93                    let parsed_transactions: Vec<CardanoTransaction> = forward_blocks
94                        .into_iter()
95                        .flat_map(|b| b.into_transactions())
96                        .collect();
97
98                    self.transaction_store.store_transactions(parsed_transactions).await?;
99                }
100                ChainScannedBlocks::RollBackward(slot_number) => {
101                    self.transaction_store
102                        .remove_rolled_back_transactions_and_block_range(slot_number)
103                        .await?;
104                }
105            }
106        }
107
108        if let Some(point) = streamer.last_polled_point() {
109            *self.last_polled_point.lock().await = Some(point);
110        }
111
112        Ok(())
113    }
114
115    async fn import_block_ranges(&self, until: BlockNumber) -> StdResult<()> {
116        let block_ranges = match self.transaction_store.get_highest_block_range().await?.map(
117            |highest_stored_block_range| {
118                BlockRange::all_block_ranges_in(
119                    BlockRange::start(highest_stored_block_range.end)..=(until),
120                )
121            },
122        ) {
123            // No block range root stored yet, start from the beginning
124            None => BlockRange::all_block_ranges_in(BlockNumber(0)..=(until)),
125            // Not enough block to form at least one block range
126            Some(ranges) if ranges.is_empty() => return Ok(()),
127            Some(ranges) => ranges,
128        };
129
130        debug!(
131            self.logger, "Computing Block Range Roots";
132            "start_block" => *block_ranges.start(), "end_block" => *block_ranges.end(),
133        );
134
135        let mut block_ranges_with_merkle_root: Vec<(BlockRange, MKTreeNode)> = vec![];
136        for block_range in block_ranges {
137            let transactions = self
138                .transaction_store
139                .get_transactions_in_range(block_range.start..block_range.end)
140                .await?;
141
142            if transactions.is_empty() {
143                continue;
144            }
145
146            let merkle_root = MKTree::<MKTreeStoreInMemory>::new(&transactions)?.compute_root()?;
147            block_ranges_with_merkle_root.push((block_range, merkle_root));
148
149            if block_ranges_with_merkle_root.len() >= 100 {
150                let block_ranges_with_merkle_root_save =
151                    mem::take(&mut block_ranges_with_merkle_root);
152                self.transaction_store
153                    .store_block_range_roots(block_ranges_with_merkle_root_save)
154                    .await?;
155            }
156        }
157
158        self.transaction_store
159            .store_block_range_roots(block_ranges_with_merkle_root)
160            .await
161    }
162}

164#[async_trait]
165impl ChainDataImporter for CardanoChainDataImporter {
166    async fn import(&self, up_to_beacon: BlockNumber) -> StdResult<()> {
167        let importer = self.clone();
168        task::spawn_blocking(move || {
169            Handle::current().block_on(async move {
170                importer.import_transactions(up_to_beacon).await?;
171                importer.import_block_ranges(up_to_beacon).await?;
172                Ok(())
173            })
174        })
175        .await
176        .with_context(|| "ChainDataImporter - worker thread crashed")?
177    }
178}

221    fn build_blocks(
222        start_block_number: BlockNumber,
223        number_of_consecutive_block: BlockNumber,
224    ) -> Vec<ScannedBlock> {
225        (*start_block_number..*(start_block_number + number_of_consecutive_block))
226            .map(|block_number| {
227                ScannedBlock::new(
228                    format!("block_hash-{block_number}"),
229                    BlockNumber(block_number),
230                    SlotNumber(block_number * 100),
231                    vec![format!("tx_hash-{}", block_number)],
232                )
233            })
234            .collect()
235    }

237    fn into_transactions(blocks: &[ScannedBlock]) -> Vec<CardanoTransaction> {
238        blocks.iter().flat_map(|b| b.clone().into_transactions()).collect()
239    }

241    fn merkle_root_for_blocks(block_ranges: &[ScannedBlock]) -> MKTreeNode {
242        let tx: Vec<_> = block_ranges
243            .iter()
244            .flat_map(|br| br.clone().into_transactions())
245            .collect();
246        MKTree::<MKTreeStoreInMemory>::new(&tx)
247            .unwrap()
248            .compute_root()
249            .unwrap()
250    }

919    #[tokio::test]
920    async fn test_import_is_non_blocking() {
921        static COUNTER: AtomicUsize = AtomicUsize::new(0);
922        static MAX_COUNTER: usize = 25;
923        static WAIT_TIME: u64 = 50;
924
925        // Use a local set to ensure the counter-task is not dispatched on a different thread
926        let local = task::LocalSet::new();
927        local
928            .run_until(async {
929                let importer = CardanoChainDataImporter::new_for_test(
930                    Arc::new(DumbBlockScanner::new()),
931                    Arc::new(BlockingRepository {
932                        wait_time: Duration::from_millis(WAIT_TIME),
933                    }),
934                );
935
936                let importer_future = importer.import(BlockNumber(100));
937                let counter_task = task::spawn_local(async {
938                    while COUNTER.load(std::sync::atomic::Ordering::SeqCst) < MAX_COUNTER {
939                        tokio::time::sleep(Duration::from_millis(1)).await;
940                        COUNTER.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
941                    }
942                });
943                importer_future.await.unwrap();
944
945                counter_task.abort();
946            })
947            .await;
948
949        assert_eq!(
950            MAX_COUNTER,
951            COUNTER.load(std::sync::atomic::Ordering::SeqCst)
952        );
953
954        struct BlockingRepository {
955            wait_time: Duration,
956        }
957
958        impl BlockingRepository {
959            fn block_thread(&self) {
960                std::thread::sleep(self.wait_time);
961            }
962        }
963
