10pub struct ReqwestGitHubApiClient {
11    client: Client,
12    github_token: Option<String>,
13}

15impl ReqwestGitHubApiClient {
16    pub fn new(github_token: Option<String>) -> MithrilResult<Self> {
17        let client = Client::builder()
18            .user_agent("mithril-client")
19            .build()
20            .context("Failed to build Reqwest GitHub API client")?;
21
22        Ok(Self {
23            client,
24            github_token,
25        })
26    }
27
28    async fn download<U: IntoUrl, T: DeserializeOwned>(&self, source_url: U) -> MithrilResult<T> {
29        let url = source_url
30            .into_url()
31            .with_context(|| "Given `source_url` is not a valid Url")?;
32        let mut request = self.client.get(url.clone());
33        if let Some(token) = &self.github_token {
34            request = request.bearer_auth(token);
35        }
36        let response = request
37            .send()
38            .await
39            .with_context(|| format!("Failed to send request to GitHub API: {url}"))?;
40        match response.status() {
41            reqwest::StatusCode::OK => {}
42            status => {
43                let body = response
44                    .text()
45                    .await
46                    .with_context(|| "Failed to read response body")?;
47                return Err(anyhow!(
48                    "GitHub API request failed with status code '{status}': {body}",
49                ));
50            }
51        }
52        let body = response.text().await?;
53        let parsed_body = serde_json::from_str::<T>(&body)
54            .with_context(|| format!("Failed to parse response from GitHub API: {body:?}"))?;
55
56        Ok(parsed_body)
57    }
58}

60#[async_trait]
61impl GitHubReleaseRetriever for ReqwestGitHubApiClient {
62    async fn get_release_by_tag(
63        &self,
64        organization: &str,
65        repository: &str,
66        tag: &str,
67    ) -> MithrilResult<GitHubRelease> {
68        let url =
69            format!("https://api.github.com/repos/{organization}/{repository}/releases/tags/{tag}");
70        let release = self.download(url).await?;
71
72        Ok(release)
73    }
74
75    async fn get_latest_release(
76        &self,
77        organization: &str,
78        repository: &str,
79    ) -> MithrilResult<GitHubRelease> {
80        let url =
81            format!("https://api.github.com/repos/{organization}/{repository}/releases/latest");
82        let release = self.download(url).await?;
83
84        Ok(release)
85    }
86
87    async fn get_prerelease(
88        &self,
89        organization: &str,
90        repository: &str,
91    ) -> MithrilResult<GitHubRelease> {
92        let releases = self.get_all_releases(organization, repository).await?;
93        let prerelease = releases
94            .into_iter()
95            .find(|release| release.prerelease)
96            .with_context(|| "No pre-release found")?;
97
98        Ok(prerelease)
99    }
100
101    async fn get_all_releases(
102        &self,
103        organization: &str,
104        repository: &str,
105    ) -> MithrilResult<Vec<GitHubRelease>> {
106        let url = format!("https://api.github.com/repos/{organization}/{repository}/releases");
107        let releases = self.download(url).await?;
108
109        Ok(releases)
110    }
111}
