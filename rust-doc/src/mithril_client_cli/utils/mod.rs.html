34/// The key used to store the caution message when printing a JSON directly to stderr
35pub(crate) const JSON_CAUTION_KEY: &str = "caution";
36
37/// Prints a simple warning message to stderr
38///
39/// The message should be no more than one or two lines, else the json output would be too large,
40/// and it would be best to break it down in a structured object. (See `warn_fast_bootstrap_not_available`
41/// in cardano db download command for an example.)
42pub(crate) fn print_simple_warning(message: &str, is_json_output_enabled: bool) {
43    if is_json_output_enabled {
44        eprintln!(r#"{{"{JSON_CAUTION_KEY}":"{message}"}}"#);
45    } else {
46        eprintln!("Warning: {message}");
47    }
48}

50/// Converts a [Path] to a [String], returning an error if the path is not valid UTF-8.
51pub(crate) fn path_to_string(path: &Path) -> MithrilResult<String> {
52    let path = path
53        .to_str()
54        .with_context(|| {
55            format!(
56                "Path '{}' contains invalid UTF-8 characters.",
57                path.display()
58            )
59        })?
60        .to_string();
61
62    Ok(path)
63}
