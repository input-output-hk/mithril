20/// Clap command to certify that given Cardano block hashes are included in a Mithril-certified Cardano block set.
21#[derive(Parser, Debug, Clone)]
22pub struct CardanoBlocksCertifyCommand {
23    /// Genesis verification key to check the certificate chain.
24    #[clap(long, env = "GENESIS_VERIFICATION_KEY")]
25    genesis_verification_key: Option<String>,
26
27    /// Hashes of the blocks to certify.
28    #[clap(value_delimiter = ',', required = true)]
29    blocks_hashes: Vec<String>,
30}
31
32impl CardanoBlocksCertifyCommand {
33    /// Cardano block certify command
34    pub async fn execute(&self, mut context: CommandContext) -> MithrilResult<()> {
35        context.require_unstable("cardano-block certify <block1>,<block2>,...,<blockn>", None)?;
36        context.config_parameters_mut().add_source(self)?;
37        let logger = context.logger();
38        let progress_output_type = if context.is_json_output_enabled() {
39            ProgressOutputType::JsonReporter
40        } else {
41            ProgressOutputType::Tty
42        };
43        let progress_printer = ProgressPrinter::new(progress_output_type, 4);
44        let client = context
45            .setup_mithril_client_builder_with_fallback_genesis_key()?
46            .add_feedback_receiver(Arc::new(IndicatifFeedbackReceiver::new(
47                progress_output_type,
48                logger.clone(),
49            )))
50            .with_capabilities(SignedEntityTypeDiscriminants::CardanoBlocksTransactions.into())
51            .build()?;
52
53        Self::certify(
54            &self.blocks_hashes,
55            client,
56            progress_printer,
57            context.is_json_output_enabled(),
58            logger.clone(),
59        )
60        .await?;
61
62        Ok(())
63    }
64
65    /// Certify that a given list of block hashes are included in the Cardano block set
66    pub async fn certify(
67        blocks_hashes: &[String],
68        client: mithril_client::Client,
69        progress_printer: ProgressPrinter,
70        is_json_output_enabled: bool,
71        logger: slog::Logger,
72    ) -> MithrilResult<()> {
73        progress_printer.report_step(1, "Fetching a proof for the given blocks…")?;
74        let cardano_block_proof = client
75            .cardano_block()
76            .get_proof(blocks_hashes)
77            .await
78            .with_context(|| {
79                format!(
80                    "Can not get proof from aggregator, blocks hashes: '{:?}'",
81                    blocks_hashes
82                )
83            })?;
84        debug!(logger, "Got Proof from aggregator"; "proof" => ?cardano_block_proof);
85        let verified_blocks =
86            Self::verify_proof_validity(2, &progress_printer, &cardano_block_proof)?;
87
88        progress_printer.report_step(
89            3,
90            "Fetching the associated certificate and verifying the certificate chain…",
91        )?;
92        let certificate = client
93            .certificate()
94            .verify_chain(&cardano_block_proof.certificate_hash)
95            .await
96            .with_context(|| {
97                format!(
98                    "Can not verify the certificate chain from certificate_hash: '{}'",
99                    verified_blocks.certificate_hash()
100                )
101            })?;
102
103        Self::verify_proof_match_certificate(4, &progress_printer, &certificate, &verified_blocks)?;
104        Self::log_certify_information(
105            &verified_blocks,
106            &cardano_block_proof.non_certified_blocks,
107            is_json_output_enabled,
108        )
109    }
110
111    fn verify_proof_validity(
112        step_number: u16,
113        progress_printer: &ProgressPrinter,
114        cardano_block_proof: &CardanoBlocksProofs,
115    ) -> MithrilResult<VerifiedCardanoBlocks> {
116        progress_printer.report_step(step_number, "Verifying the proof…")?;
117        match cardano_block_proof.verify() {
118            Ok(verified_blocks) => Ok(verified_blocks),
119            Err(VerifyProofsV2Error::NoCertifiedItem(..)) => Err(anyhow!(
120                "Mithril could not certify any of the given blocks.
121
122Mithril may not have signed those blocks yet, please try again later."
123            )),
124            err => err.with_context(|| "Proof verification failed"),
125        }
126    }
127
128    fn verify_proof_match_certificate(
129        step_number: u16,
130        progress_printer: &ProgressPrinter,
131        certificate: &MithrilCertificate,
132        verified_blocks: &VerifiedCardanoBlocks,
133    ) -> MithrilResult<()> {
134        progress_printer.report_step(
135            step_number,
136            "Verify that the proof is signed in the associated certificate",
137        )?;
138        let message = MessageBuilder::new()
139            .compute_cardano_blocks_proofs_message(certificate, verified_blocks);
140        if !certificate.match_message(&message) {
141            return Err(anyhow!(
142                "Proof and certificate don't match (certificate hash = '{}').",
143                certificate.hash
144            ));
145        }
146
147        Ok(())
148    }
149
150    fn log_certify_information(
151        verified_blocks: &VerifiedCardanoBlocks,
152        non_certified_blocks: &[BlockHash],
153        json_output: bool,
154    ) -> MithrilResult<()> {
155        let latest_certified_block_number = verified_blocks.latest_certified_block_number();
156        let security_parameter = verified_blocks.security_parameter();
157
158        if json_output {
159            println!(
160                r#"{{"certified_blocks": {}, "non_certified_blocks": {}}}"#,
161                serde_json::to_string(
162                    &verified_blocks
163                        .certified_blocks()
164                        .iter()
165                        .map(|block| {
166                            serde_json::json!({
167                                "block_hash": block.block_hash,
168                                "block_number": block.block_number,
169                                "slot_number": block.slot_number,
170                                "block_depth": compute_depth(
171                                    latest_certified_block_number,
172                                    security_parameter,
173                                    block.block_number,
174                                ),
175                            })
176                        })
177                        .collect::<Vec<_>>()
178                )?,
179                serde_json::to_string(non_certified_blocks)?,
180            );
181        } else {
182            println!(
183                r###"Cardano blocks proof has been successfully signed in the associated Mithril certificate."###,
184            );
185
186            if !non_certified_blocks.is_empty() {
187                println!(
188                    r###"
189No proof could be computed for some Cardano blocks. Mithril may not have signed those blocks yet, please try again later."###,
190                );
191            }
192
193            let result_table = verified_blocks
194                .certified_blocks()
195                .iter()
196                .map(|block| {
197                    vec![
198                        block.block_hash.clone().cell(),
199                        format!("{}", block.block_number).cell(),
200                        format!("{}", block.slot_number).cell(),
201                        format!(
202                            "{}",
203                            compute_depth(
204                                latest_certified_block_number,
205                                security_parameter,
206                                block.block_number,
207                            )
208                        )
209                        .cell(),
210                        "✅".cell().justify(cli_table::format::Justify::Center),
211                    ]
212                })
213                .chain(non_certified_blocks.iter().map(|block| {
214                    vec![
215                        block.clone().cell(),
216                        String::new().cell(),
217                        String::new().cell(),
218                        String::new().cell(),
219                        "❌".cell().justify(cli_table::format::Justify::Center),
220                    ]
221                }))
222                .table()
223                .title(vec![
224                    "Block Hash".cell(),
225                    "Block Number".cell(),
226                    "Slot Number".cell(),
227                    "Block depth".cell(),
228                    "Certified".cell(),
229                ]);
230
231            print_stdout(result_table)?
232        }
233
234        Ok(())
235    }

238impl ConfigSource for CardanoBlocksCertifyCommand {
239    fn collect(&self) -> Result<HashMap<String, String>, ConfigError> {
240        let mut map = HashMap::new();
241
242        if let Some(genesis_verification_key) = self.genesis_verification_key.clone() {
243            map.insert(
244                "genesis_verification_key".to_string(),
245                genesis_verification_key,
246            );
247        }
248
249        Ok(map)
250    }
