22pub async fn fetch_certificate_and_verifying_chain(
23    step_number: u16,
24    progress_printer: &ProgressPrinter,
25    client: &Client,
26    certificate_hash: &str,
27) -> MithrilResult<MithrilCertificate> {
28    progress_printer.report_step(
29        step_number,
30        "Fetching the certificate and verifying the certificate chain…",
31    )?;
32    let certificate = client
33        .certificate()
34        .verify_chain(certificate_hash)
35        .await
36        .with_context(|| {
37            format!(
38                "Can not verify the certificate chain from certificate_hash: '{certificate_hash}'"
39            )
40        })?;
41
42    Ok(certificate)
43}

45/// Constructs an `ImmutableFileRange` based on the provided start and end file numbers.
46pub fn immutable_file_range(
47    start: Option<ImmutableFileNumber>,
48    end: Option<ImmutableFileNumber>,
49) -> ImmutableFileRange {
50    match (start, end) {
51        (None, None) => ImmutableFileRange::Full,
52        (Some(start), None) => ImmutableFileRange::From(start),
53        (Some(start), Some(end)) => ImmutableFileRange::Range(start, end),
54        (None, Some(end)) => ImmutableFileRange::UpTo(end),
55    }
56}

74pub async fn verify_cardano_database(
75    step_number: u16,
76    progress_printer: &ProgressPrinter,
77    client: &Client,
78    certificate: &MithrilCertificate,
79    cardano_database_snapshot: &CardanoDatabaseSnapshot,
80    options: &ComputeCardanoDatabaseMessageOptions,
81    verified_digest: &VerifiedDigests,
82) -> MithrilResult<MKProof> {
83    progress_printer.report_step(step_number, "Verifying the cardano database")?;
84    let merkle_proof = CardanoDbUtils::wait_spinner(
85        progress_printer,
86        client.cardano_database_v2().verify_cardano_database(
87            certificate,
88            cardano_database_snapshot,
89            &options.immutable_file_range,
90            options.allow_missing,
91            &options.db_dir,
92            verified_digest,
93        ),
94    )
95    .await
96    .with_context(|| "Can not verify the cardano database")?;
97
98    Ok(merkle_proof)
99}

119/// Verifies that the Cardano database snapshot matches the associated certificate.
120pub async fn verify_message_matches_certificate(
121    logger: &Logger,
122    step_number: u16,
123    progress_printer: &ProgressPrinter,
124    certificate: &MithrilCertificate,
125    message: &ProtocolMessage,
126    cardano_db_snapshot: &CardanoDatabaseSnapshot,
127    db_dir: &Path,
128) -> MithrilResult<()> {
129    progress_printer.report_step(step_number, "Verifying the cardano db signature…")?;
130    if !certificate.match_message(message) {
131        debug!(
132            logger,
133            "Merkle root verification failed, removing unpacked files & directory."
134        );
135
136        if let Err(error) = std::fs::remove_dir_all(db_dir) {
137            warn!(
138                logger, "Error while removing unpacked files & directory";
139                "error" => error.to_string()
140            );
141        }
142
143        return Err(anyhow!(
144            "Certificate verification failed (cardano db snapshot hash = '{}').",
145            cardano_db_snapshot.hash.clone()
146        ));
147    }
148
149    Ok(())
150}

152pub fn log_download_information(
153    db_dir: &Path,
154    snapshot_hash: &str,
155    cardano_network: &str,
156    cardano_node_version: &str,
157    json_output: bool,
158    include_ancillary: bool,
159) -> MithrilResult<()> {
160    fn is_linux_arm() -> bool {
161        cfg!(target_os = "linux") && cfg!(target_arch = "aarch64")
162    }
163
164    let canonical_filepath = &db_dir
165        .canonicalize()
166        .with_context(|| format!("Could not get canonical filepath of '{}'", db_dir.display()))?;
167
168    let docker_cmd = CardanoDbUtils::get_docker_run_command(
169        canonical_filepath,
170        cardano_network,
171        cardano_node_version,
172        // Aggregators are only using UTxO-HD in memory format as other format are deprecated or non portable
173        LedgerFormat::InMemory,
174    );
175
176    let snapshot_converter_cmd = |flavor| {
177        format!(
178            "mithril-client tools utxo-hd snapshot-converter --db-directory {} --cardano-node-version {} --utxo-hd-flavor {} --commit",
179            db_dir.display(),
180            cardano_node_version,
181            flavor,
182        )
183    };
184
185    if json_output {
186        let mut json = serde_json::json!({
187            "timestamp": Utc::now().to_rfc3339(),
188            "db_directory": canonical_filepath,
189        });
190
191        if !is_linux_arm() {
192            json["run_docker_cmd"] = serde_json::Value::String(docker_cmd);
193
194            if include_ancillary {
195                json["snapshot_converter_cmd_to_lmdb"] =
196                    serde_json::Value::String(snapshot_converter_cmd("LMDB"));
197                json["snapshot_converter_cmd_to_legacy"] =
198                    serde_json::Value::String(snapshot_converter_cmd("Legacy"));
199            }
200        }
201
202        println!("{json}");
203    } else {
204        println!(
205            r###"Cardano database snapshot '{}' archives have been successfully unpacked. Immutable files have been successfully verified with Mithril.
206
207    Files in the directory '{}' can be used to run a Cardano node with version >= {cardano_node_version}.
208    "###,
209            snapshot_hash,
210            db_dir.display()
211        );
212
213        if !is_linux_arm() {
214            println!(
215                r###"If you are using the Cardano Docker image, you can restore a Cardano node with:
216
217    {docker_cmd}
218
219    "###
220            );
221
222            if include_ancillary {
223                println!(
224                    r###"Upgrade and replace the restored ledger state snapshot to 'LMDB' flavor by running the command:
225
226    {}
227    "###,
228                    snapshot_converter_cmd("LMDB"),
229                );
230
231                if !is_version_at_least_10_6_2_or_latest(cardano_node_version) {
232                    println!(
233                        r###"Or to 'Legacy' flavor by running the command:
234
235    {}
236    "###,
237                        snapshot_converter_cmd("Legacy"),
238                    );
239                }
240            }
241        }
242    }
243
244    Ok(())
245}
