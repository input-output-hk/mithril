20pub async fn fetch_certificate_and_verifying_chain(
21    step_number: u16,
22    progress_printer: &ProgressPrinter,
23    client: &Client,
24    certificate_hash: &str,
25) -> MithrilResult<MithrilCertificate> {
26    progress_printer.report_step(
27        step_number,
28        "Fetching the certificate and verifying the certificate chain…",
29    )?;
30    let certificate = client
31        .certificate()
32        .verify_chain(certificate_hash)
33        .await
34        .with_context(|| {
35            format!(
36                "Can not verify the certificate chain from certificate_hash: '{certificate_hash}'"
37            )
38        })?;
39
40    Ok(certificate)
41}

43/// Constructs an `ImmutableFileRange` based on the provided start and end file numbers.
44pub fn immutable_file_range(
45    start: Option<ImmutableFileNumber>,
46    end: Option<ImmutableFileNumber>,
47) -> ImmutableFileRange {
48    match (start, end) {
49        (None, None) => ImmutableFileRange::Full,
50        (Some(start), None) => ImmutableFileRange::From(start),
51        (Some(start), Some(end)) => ImmutableFileRange::Range(start, end),
52        (None, Some(end)) => ImmutableFileRange::UpTo(end),
53    }
54}

72pub async fn verify_cardano_database(
73    step_number: u16,
74    progress_printer: &ProgressPrinter,
75    client: &Client,
76    certificate: &MithrilCertificate,
77    cardano_database_snapshot: &CardanoDatabaseSnapshot,
78    options: &ComputeCardanoDatabaseMessageOptions,
79    verified_digest: &VerifiedDigests,
80) -> MithrilResult<MKProof> {
81    progress_printer.report_step(step_number, "Verifying the cardano database")?;
82    let merkle_proof = CardanoDbUtils::wait_spinner(
83        progress_printer,
84        client.cardano_database_v2().verify_cardano_database(
85            certificate,
86            cardano_database_snapshot,
87            &options.immutable_file_range,
88            options.allow_missing,
89            &options.db_dir,
90            verified_digest,
91        ),
92    )
93    .await
94    .with_context(|| "Can not verify the cardano database")?;
95
96    Ok(merkle_proof)
97}

117/// Verifies that the Cardano database snapshot matches the associated certificate.
118pub async fn verify_message_matches_certificate(
119    logger: &Logger,
120    step_number: u16,
121    progress_printer: &ProgressPrinter,
122    certificate: &MithrilCertificate,
123    message: &ProtocolMessage,
124    cardano_db_snapshot: &CardanoDatabaseSnapshot,
125    db_dir: &Path,
126) -> MithrilResult<()> {
127    progress_printer.report_step(step_number, "Verifying the cardano db signature…")?;
128    if !certificate.match_message(message) {
129        debug!(
130            logger,
131            "Merkle root verification failed, removing unpacked files & directory."
132        );
133
134        if let Err(error) = std::fs::remove_dir_all(db_dir) {
135            warn!(
136                logger, "Error while removing unpacked files & directory";
137                "error" => error.to_string()
138            );
139        }
140
141        return Err(anyhow!(
142            "Certificate verification failed (cardano db snapshot hash = '{}').",
143            cardano_db_snapshot.hash.clone()
144        ));
145    }
146
147    Ok(())
148}

150pub fn log_download_information(
151    db_dir: &Path,
152    snapshot_hash: &str,
153    cardano_network: &str,
154    cardano_node_version: &str,
155    json_output: bool,
156    include_ancillary: bool,
157) -> MithrilResult<()> {
158    fn is_linux_arm() -> bool {
159        cfg!(target_os = "linux") && cfg!(target_arch = "aarch64")
160    }
161
162    let canonical_filepath = &db_dir
163        .canonicalize()
164        .with_context(|| format!("Could not get canonical filepath of '{}'", db_dir.display()))?;
165
166    let docker_cmd = format!(
167        "docker run -v cardano-node-ipc:/ipc -v cardano-node-data:/data --mount type=bind,source=\"{}\",target=/data/db/ -e NETWORK={} ghcr.io/intersectmbo/cardano-node:{}",
168        canonical_filepath.display(),
169        cardano_network,
170        cardano_node_version
171    );
172
173    let snapshot_converter_cmd = |flavor| {
174        format!(
175            "mithril-client tools utxo-hd snapshot-converter --db-directory {} --cardano-node-version {} --utxo-hd-flavor {} --commit",
176            db_dir.display(),
177            cardano_node_version,
178            flavor,
179        )
180    };
181
182    if json_output {
183        let mut json = serde_json::json!({
184            "timestamp": Utc::now().to_rfc3339(),
185            "db_directory": canonical_filepath,
186        });
187
188        if !is_linux_arm() {
189            json["run_docker_cmd"] = serde_json::Value::String(docker_cmd);
190
191            if include_ancillary {
192                json["snapshot_converter_cmd_to_lmdb"] =
193                    serde_json::Value::String(snapshot_converter_cmd("LMDB"));
194                json["snapshot_converter_cmd_to_legacy"] =
195                    serde_json::Value::String(snapshot_converter_cmd("Legacy"));
196            }
197        }
198
199        println!("{json}");
200    } else {
201        println!(
202            r###"Cardano database snapshot '{}' archives have been successfully unpacked. Immutable files have been successfully verified with Mithril.
203
204    Files in the directory '{}' can be used to run a Cardano node with version >= {cardano_node_version}.
205    "###,
206            snapshot_hash,
207            db_dir.display()
208        );
209
210        if !is_linux_arm() {
211            println!(
212                r###"If you are using the Cardano Docker image, you can restore a Cardano node with:
213
214    {docker_cmd}
215
216    "###
217            );
218
219            if include_ancillary {
220                println!(
221                    r###"Upgrade and replace the restored ledger state snapshot to 'LMDB' flavor by running the command:
222
223    {}
224
225    Or to 'Legacy' flavor by running the command:
226
227    {}
228    "###,
229                    snapshot_converter_cmd("LMDB"),
230                    snapshot_converter_cmd("Legacy"),
231                );
232            }
233        }
234    }
235
236    Ok(())
237}
