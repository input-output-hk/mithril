5/// Trait for converting types into their records representations
6///
7/// Private as it is only meant to be used for types where we can't implement conversion using standard
8/// [From] trait (such as `Vec`).
9pub(crate) trait IntoRecords {
10    type Output;
11
12    fn into_records(self) -> Self::Output;
13}
14
15impl IntoRecords for Vec<CardanoBlockWithTransactions> {
16    type Output = (
17        Vec<CardanoBlockRecord>,
18        Vec<StorableCardanoTransactionRecord>,
19    );
20
21    fn into_records(self) -> Self::Output {
22        let transactions_records = self
23            .iter()
24            .flat_map(|b| {
25                b.transactions_hashes
26                    .clone()
27                    .into_iter()
28                    .map(|h| StorableCardanoTransactionRecord::new(h, &b.block_hash))
29            })
30            .collect();
31        let blocks_records = self.into_iter().map(Into::into).collect();
32
33        (blocks_records, transactions_records)
34    }
35}
