10/// Query to insert [CardanoTransactionRecord] in the sqlite database
11pub struct InsertCardanoTransactionQuery {
12    condition: WhereCondition,
13}
14
15impl InsertCardanoTransactionQuery {
16    /// Query that insert multiples records.
17    pub fn insert_many(
18        transactions_records: Vec<StorableCardanoTransactionRecord>,
19    ) -> StdResult<Self> {
20        let columns = "(transaction_hash,  block_hash)";
21        let values_columns: Vec<&str> = repeat_n("(?*, ?*)", transactions_records.len()).collect();
22
23        let values: StdResult<Vec<Value>> =
24            transactions_records.into_iter().try_fold(vec![], |mut vec, record| {
25                vec.append(&mut vec![
26                    Value::String(record.transaction_hash),
27                    Value::String(record.block_hash),
28                ]);
29                Ok(vec)
30            });
31        let condition = WhereCondition::new(
32            format!("{columns} values {}", values_columns.join(", ")).as_str(),
33            values?,
34        );
35
36        Ok(Self { condition })
37    }
38}
39
40impl Query for InsertCardanoTransactionQuery {
41    type Entity = StorableCardanoTransactionRecord;
42
43    fn filters(&self) -> WhereCondition {
44        self.condition.clone()
45    }
46
47    fn get_definition(&self, condition: &str) -> String {
48        let aliases = SourceAlias::new(&[("{:cardano_tx:}", "cardano_tx")]);
49        let projection = Self::Entity::get_projection().expand(aliases);
50
51        format!("insert or ignore into cardano_tx {condition} returning {projection}")
52    }
53}
