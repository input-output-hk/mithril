1use anyhow::Context;
2use mithril_common::{
3    StdResult,
4    entities::{SingleSignature, SingleSignatureAuthenticationStatus},
5    messages::{RegisterSignatureMessageHttp, TryFromMessageAdapter},
6};
7
8pub struct FromRegisterSingleSignatureAdapter;
9
10impl TryFromMessageAdapter<RegisterSignatureMessageHttp, SingleSignature>
11    for FromRegisterSingleSignatureAdapter
12{
13    fn try_adapt(
14        register_single_signature_message: RegisterSignatureMessageHttp,
15    ) -> StdResult<SingleSignature> {
16        let signature = SingleSignature {
17            party_id: register_single_signature_message.party_id,
18            signature: register_single_signature_message.signature.try_into().with_context(
19                || "'FromRegisterSingleSignatureAdapter' can not convert the single signature",
20            )?,
21            won_indexes: register_single_signature_message.won_indexes,
22            authentication_status: SingleSignatureAuthenticationStatus::Unauthenticated,
23        };
24
25        Ok(signature)
26    }
27}
