12/// `BufferedSingleSignatureRecord` record is the representation of a buffered single_signature
13/// that may be used for upcoming open messages.
14#[derive(Debug, PartialEq, Clone)]
15pub struct BufferedSingleSignatureRecord {
16    /// Party id.
17    pub party_id: String,
18
19    /// Signed entity type discriminant.
20    pub signed_entity_type_id: SignedEntityTypeDiscriminants,
21
22    /// Lottery indexes
23    pub lottery_indexes: Vec<LotteryIndex>,
24
25    /// The STM single signature of the message
26    pub signature: HexEncodedSingleSignature,
27
28    /// Date and time when the buffered single signature was created
29    pub created_at: DateTime<Utc>,
30}
31
32impl BufferedSingleSignatureRecord {
33    pub(crate) fn try_from_single_signature(
34        other: &SingleSignature,
35        signed_entity_id: SignedEntityTypeDiscriminants,
36    ) -> StdResult<Self> {
37        let record = BufferedSingleSignatureRecord {
38            signed_entity_type_id: signed_entity_id,
39            party_id: other.party_id.to_owned(),
40            lottery_indexes: other.won_indexes.to_owned(),
41            signature: other.signature.to_json_hex()?,
42            created_at: Utc::now(),
43        };
44
45        Ok(record)
46    }
47}

49#[cfg(test)]
50impl BufferedSingleSignatureRecord {
51    pub(crate) fn fake<T: Into<String>>(
52        party_id: T,
53        discriminant: SignedEntityTypeDiscriminants,
54    ) -> Self {
55        // Note: due to the unique constraint on the signature column, we want to make sure that
56        // the signature are different for party_id/discriminant pairs.
57        // We can't just reuse fake_data::single_signature as they are static.
58        Self::try_from_single_signature(
59            &SingleSignature::fake(party_id.into(), discriminant.to_string()),
60            discriminant,
61        )
62        .unwrap()
63    }
64
65    pub(crate) fn fakes<T: Into<String> + Clone>(
66        input: &[(T, SignedEntityTypeDiscriminants)],
67    ) -> Vec<Self> {
68        input
69            .iter()
70            .map(|(party_id, discriminants)| Self::fake(party_id.clone(), *discriminants))
71            .collect()
72    }
73
74    /// Returns a copy of the record with the date replaced by "1st of January 1970".
75    pub(crate) fn with_stripped_date(&self) -> Self {
76        Self {
77            created_at: DateTime::<Utc>::default(),
78            ..self.clone()
79        }
80    }
81}

83impl TryFrom<BufferedSingleSignatureRecord> for SingleSignature {
84    type Error = StdError;
85
86    fn try_from(value: BufferedSingleSignatureRecord) -> Result<Self, Self::Error> {
87        let signature = SingleSignature {
88            party_id: value.party_id,
89            won_indexes: value.lottery_indexes,
90            signature: value.signature.try_into()?,
91            authentication_status: Default::default(),
92        };
93
94        Ok(signature)
95    }

98impl SqLiteEntity for BufferedSingleSignatureRecord {
99    fn hydrate(row: sqlite::Row) -> Result<Self, HydrationError>
100    where
101        Self: Sized,
102    {
103        let signed_entity_type_id = usize::try_from(row.read::<i64, _>(0)).map_err(|e| {
104            panic!("Integer field signed_entity_type_id cannot be turned into usize: {e}")
105        })?;
106        let party_id = row.read::<&str, _>(1).to_string();
107        let lottery_indexes_str = row.read::<&str, _>(2);
108        let signature = row.read::<&str, _>(3).to_string();
109        let created_at = row.read::<&str, _>(4);
110
111        let record = Self {
112            party_id,
113            signed_entity_type_id: SignedEntityTypeDiscriminants::from_id(signed_entity_type_id).map_err(
114                |e| {
115                    HydrationError::InvalidData(format!(
116                        "Could not turn i64 ({signed_entity_type_id}) to SignedEntityTypeDiscriminants. Error: '{e}'"
117                    ))
118                },
119            )?,
120            lottery_indexes: serde_json::from_str(lottery_indexes_str).map_err(|e| {
121                HydrationError::InvalidData(format!(
122                    "Could not turn string '{lottery_indexes_str}' to Vec<LotteryIndex>. Error: {e}"
123                ))
124            })?,
125            signature,
126            created_at: DateTime::parse_from_rfc3339(created_at)
127                .map_err(|e| {
128                    HydrationError::InvalidData(format!(
129                        "Could not turn string '{created_at}' to rfc3339 Datetime. Error: {e}"
130                    ))
131                })?
132                .with_timezone(&Utc),
133        };
134
135        Ok(record)
136    }
137
138    fn get_projection() -> Projection {
139        let mut projection = Projection::default();
140        projection.add_field(
141            "signed_entity_type_id",
142            "{:buffered_single_signature:}.signed_entity_type_id",
143            "integer",
144        );
145        projection.add_field("party_id", "{:buffered_single_signature:}.party_id", "text");
146        projection.add_field(
147            "lottery_indexes",
148            "{:buffered_single_signature:}.lottery_indexes",
149            "text",
150        );
151        projection.add_field(
152            "signature",
153            "{:buffered_single_signature:}.signature",
154            "text",
155        );
156        projection.add_field(
157            "created_at",
158            "{:buffered_single_signature:}.created_at",
159            "text",
160        );
161
162        projection
163    }
