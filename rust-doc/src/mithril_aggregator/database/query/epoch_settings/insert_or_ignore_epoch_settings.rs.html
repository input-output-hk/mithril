1use sqlite::Value;
2
3use mithril_persistence::sqlite::{Query, SourceAlias, SqLiteEntity, WhereCondition};
4
5use crate::database::record::EpochSettingsRecord;
6
7/// Query to insert a [EpochSettingsRecord] into the sqlite database if it does not already exist.
8pub struct InsertOrIgnoreEpochSettingsQuery {
9    condition: WhereCondition,
10}
11
12impl InsertOrIgnoreEpochSettingsQuery {
13    pub fn one(epoch_settings: EpochSettingsRecord) -> Self {
14        Self {
15            condition: WhereCondition::new(
16                r#"(epoch_setting_id, protocol_parameters, 
17                cardano_transactions_signing_config, 
18                cardano_blocks_transactions_signing_config) 
19                values (?1, ?2, ?3, ?4)"#,
20                vec![
21                    Value::Integer(*epoch_settings.epoch_settings_id as i64),
22                    Value::String(
23                        serde_json::to_string(&epoch_settings.protocol_parameters).unwrap(),
24                    ),
25                    match &epoch_settings.cardano_transactions_signing_config {
26                        Some(config) => Value::String(serde_json::to_string(config).unwrap()),
27                        None => Value::Null,
28                    },
29                    match &epoch_settings.cardano_blocks_transactions_signing_config {
30                        Some(config) => Value::String(serde_json::to_string(config).unwrap()),
31                        None => Value::Null,
32                    },
33                ],
34            ),
35        }
36    }
37}
38
39impl Query for InsertOrIgnoreEpochSettingsQuery {
40    type Entity = EpochSettingsRecord;
41
42    fn filters(&self) -> WhereCondition {
43        self.condition.clone()
44    }
45
46    fn get_definition(&self, condition: &str) -> String {
47        // it is important to alias the fields with the same name as the table
48        // since the table cannot be aliased in a RETURNING statement in SQLite.
49        let projection = Self::Entity::get_projection()
50            .expand(SourceAlias::new(&[("{:epoch_setting:}", "epoch_setting")]));
51
52        format!("insert or ignore into epoch_setting {condition} returning {projection}")
53    }
54}

71    fn test_insert_epoch_setting_in_empty_db() {
72        let connection = main_db_connection().unwrap();
73
74        let expected_epoch_settings = EpochSettingsRecord {
75            epoch_settings_id: Epoch(3),
76            protocol_parameters: fake_data::protocol_parameters(),
77            cardano_transactions_signing_config: Some(CardanoTransactionsSigningConfig {
78                security_parameter: BlockNumber(24),
79                step: BlockNumber(62),
80            }),
81            cardano_blocks_transactions_signing_config: Some(
82                CardanoBlocksTransactionsSigningConfig {
83                    security_parameter: BlockNumber(48),
84                    step: BlockNumber(96),
85                },
86            ),
87        };
88        let record = connection
89            .fetch_first(InsertOrIgnoreEpochSettingsQuery::one(
90                expected_epoch_settings.clone(),
91            ))
92            .unwrap();
93
94        assert_eq!(Some(expected_epoch_settings), record);
95
96        let expected_epoch_settings_without_ctx_config = EpochSettingsRecord {
97            epoch_settings_id: Epoch(4),
98            protocol_parameters: fake_data::protocol_parameters(),
99            cardano_transactions_signing_config: None,
100            cardano_blocks_transactions_signing_config: None,
101        };
102        let record = connection
103            .fetch_first(InsertOrIgnoreEpochSettingsQuery::one(
104                expected_epoch_settings_without_ctx_config.clone(),
105            ))
106            .unwrap();
107
108        assert_eq!(Some(expected_epoch_settings_without_ctx_config), record);
