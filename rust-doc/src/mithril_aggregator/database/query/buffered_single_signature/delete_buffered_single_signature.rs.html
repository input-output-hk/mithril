1use sqlite::Value;
2
3use mithril_common::entities::{PartyId, SignedEntityTypeDiscriminants};
4use mithril_persistence::sqlite::{Query, SourceAlias, SqLiteEntity, WhereCondition};
5
6use crate::database::record::BufferedSingleSignatureRecord;
7
8/// Query to delete old [BufferedSingleSignatureRecord] from the sqlite database
9pub struct DeleteBufferedSingleSignatureQuery {
10    condition: WhereCondition,
11}
12
13impl DeleteBufferedSingleSignatureQuery {
14    pub fn by_discriminant_and_party_ids(
15        signed_entity_type_discriminant: SignedEntityTypeDiscriminants,
16        party_ids: Vec<PartyId>,
17    ) -> Self {
18        let ids_values = party_ids.into_iter().map(Value::String).collect();
19
20        Self {
21            condition: WhereCondition::new(
22                "signed_entity_type_id = ?*",
23                vec![Value::Integer(signed_entity_type_discriminant.index() as i64)],
24            )
25            .and_where(WhereCondition::where_in("party_id", ids_values)),
26        }
27    }
28}
29
30impl Query for DeleteBufferedSingleSignatureQuery {
31    type Entity = BufferedSingleSignatureRecord;
32
33    fn filters(&self) -> WhereCondition {
34        self.condition.clone()
35    }
36
37    fn get_definition(&self, condition: &str) -> String {
38        // it is important to alias the fields with the same name as the table
39        // since the table cannot be aliased in a RETURNING statement in SQLite.
40        let projection = Self::Entity::get_projection().expand(SourceAlias::new(&[(
41            "{:buffered_single_signature:}",
42            "buffered_single_signature",
43        )]));
44
45        format!("delete from buffered_single_signature where {condition} returning {projection}")
46    }
47}

49#[cfg(test)]
50mod tests {
51    use mithril_common::entities::SignedEntityTypeDiscriminants::{
52        CardanoTransactions, MithrilStakeDistribution,
53    };
54    use mithril_persistence::sqlite::ConnectionExtensions;
55
56    use crate::database::query::GetBufferedSingleSignatureQuery;
57    use crate::database::record::strip_buffered_sigs_date;
58    use crate::database::test_helper::{insert_buffered_single_signatures, main_db_connection};
59
60    use super::*;
61
62    #[test]
63    fn test_delete_buffered_single_signature_records_by_discriminant_and_party_ids() {
64        let connection = main_db_connection().unwrap();
65        let records = BufferedSingleSignatureRecord::fakes(&[
66            ("party_1", MithrilStakeDistribution),
67            ("party_2", MithrilStakeDistribution),
68            ("party_3", MithrilStakeDistribution),
69            ("party_1", CardanoTransactions),
70            ("party_2", CardanoTransactions),
71        ]);
72        insert_buffered_single_signatures(&connection, records.clone()).unwrap();
73
74        let cursor = connection
75            .fetch(
76                DeleteBufferedSingleSignatureQuery::by_discriminant_and_party_ids(
77                    MithrilStakeDistribution,
78                    vec!["party_1".into(), "party_3".into()],
79                ),
80            )
81            .unwrap();
82        assert_eq!(2, cursor.count());
83
84        let remaining_records: Vec<BufferedSingleSignatureRecord> = connection
85            .fetch_collect(GetBufferedSingleSignatureQuery::all())
86            .unwrap();
87        assert_eq!(
88            strip_buffered_sigs_date(&BufferedSingleSignatureRecord::fakes(&[
89                ("party_2", CardanoTransactions),
90                ("party_1", CardanoTransactions),
91                ("party_2", MithrilStakeDistribution),
92            ])),
93            strip_buffered_sigs_date(&remaining_records)
94        );
95
96        let cursor = connection
97            .fetch(
98                DeleteBufferedSingleSignatureQuery::by_discriminant_and_party_ids(
99                    CardanoTransactions,
100                    vec!["party_1".into(), "party_2".into()],
101                ),
102            )
103            .unwrap();
104        assert_eq!(2, cursor.count());
105
106        let remaining_records: Vec<BufferedSingleSignatureRecord> = connection
107            .fetch_collect(GetBufferedSingleSignatureQuery::all())
108            .unwrap();
109        assert_eq!(
110            strip_buffered_sigs_date(&BufferedSingleSignatureRecord::fakes(&[(
111                "party_2",
112                MithrilStakeDistribution
113            ),])),
114            strip_buffered_sigs_date(&remaining_records)
115        );
116    }
