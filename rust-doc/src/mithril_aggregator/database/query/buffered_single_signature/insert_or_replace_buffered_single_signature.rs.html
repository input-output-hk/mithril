7/// Query to insert or replace [BufferedSingleSignatureRecord] in the sqlite database
8pub struct InsertOrReplaceBufferedSingleSignatureRecordQuery {
9    condition: WhereCondition,
10}
11
12impl InsertOrReplaceBufferedSingleSignatureRecordQuery {
13    pub fn one(record: BufferedSingleSignatureRecord) -> Self {
14        let condition = WhereCondition::new(
15            "(signed_entity_type_id, party_id, lottery_indexes, signature, created_at) values (?*, ?*, ?*, ?*, ?*)",
16            vec![
17                Value::Integer(record.signed_entity_type_id.index() as i64),
18                Value::String(record.party_id),
19                Value::String(serde_json::to_string(&record.lottery_indexes).unwrap()),
20                Value::String(record.signature),
21                Value::String(record.created_at.to_rfc3339()),
22            ],
23        );
24
25        Self { condition }
26    }
27}
28
29impl Query for InsertOrReplaceBufferedSingleSignatureRecordQuery {
30    type Entity = BufferedSingleSignatureRecord;
31
32    fn filters(&self) -> WhereCondition {
33        self.condition.clone()
34    }
35
36    fn get_definition(&self, condition: &str) -> String {
37        // it is important to alias the fields with the same name as the table
38        // since the table cannot be aliased in a RETURNING statement in SQLite.
39        let projection = Self::Entity::get_projection().expand(SourceAlias::new(&[(
40            "{:buffered_single_signature:}",
41            "buffered_single_signature",
42        )]));
43
44        format!(
45            "insert or replace into buffered_single_signature {condition} returning {projection}"
46        )
47    }
48}

158                updated_record.clone(),

160            .unwrap();
161        let count = connection
162            .fetch(GetBufferedSingleSignatureQuery::all())
163            .unwrap()
