7#[derive(Deserialize, Serialize, Debug)]
8struct CardanoTransactionProofQueryParams {
9    transaction_hashes: String,
10}
11
12impl CardanoTransactionProofQueryParams {
13    pub fn split_transactions_hashes(&self) -> Vec<String> {
14        self.transaction_hashes
15            .split(',')
16            .map(|s| s.to_string())
17            .collect()
18    }
19
20    pub fn sanitize(&self) -> Vec<String> {
21        let mut transaction_hashes = self.split_transactions_hashes();
22        transaction_hashes.sort();
23        transaction_hashes.dedup();
24        transaction_hashes
25    }
26}

28pub fn routes(
29    router_state: &RouterState,
30) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
31    proof_cardano_transaction(router_state)
32}

34/// GET /proof/cardano-transaction
35fn proof_cardano_transaction(
36    router_state: &RouterState,
37) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
38    warp::path!("proof" / "cardano-transaction")
39        .and(warp::get())
40        .and(middlewares::with_client_metadata(router_state))
41        .and(warp::query::<CardanoTransactionProofQueryParams>())
42        .and(middlewares::with_logger(router_state))
43        .and(middlewares::with_signed_entity_service(router_state))
44        .and(middlewares::validators::with_prover_transactions_hash_validator(router_state))
45        .and(middlewares::with_prover_service(router_state))
46        .and(middlewares::with_metrics_service(router_state))
47        .and_then(handlers::proof_cardano_transaction)
48}

72    pub async fn proof_cardano_transaction(
73        client_metadata: ClientMetadata,
74        transaction_parameters: CardanoTransactionProofQueryParams,
75        logger: Logger,
76        signed_entity_service: Arc<dyn SignedEntityService>,
77        validator: ProverTransactionsHashValidator,
78        prover_service: Arc<dyn ProverService>,
79        metrics_service: Arc<MetricsService>,
80    ) -> Result<impl warp::Reply, Infallible> {
81        metrics_service
82            .get_proof_cardano_transaction_total_proofs_served_since_startup()
83            .increment(&[
84                client_metadata.origin_tag.as_deref().unwrap_or_default(),
85                client_metadata.client_type.as_deref().unwrap_or_default(),
86            ]);
87
88        let transaction_hashes = transaction_parameters.split_transactions_hashes();
89        debug!(
90            logger, ">> proof_cardano_transaction";
91            "transaction_hashes" => &transaction_parameters.transaction_hashes
92        );
93
94        if let Err(error) = validator.validate(&transaction_hashes) {
95            warn!(logger, "proof_cardano_transaction::bad_request");
96            return Ok(reply::bad_request(error.label, error.message));
97        }
98
99        let sanitized_hashes = transaction_parameters.sanitize();
100
101        // Fallback to 0, it should be impossible to have more than u32::MAX transactions.
102        metrics_service
103            .get_proof_cardano_transaction_total_transactions_served_since_startup()
104            .increment_by(
105                &[
106                    client_metadata.origin_tag.as_deref().unwrap_or_default(),
107                    client_metadata.client_type.as_deref().unwrap_or_default(),
108                ],
109                sanitized_hashes.len().try_into().unwrap_or(0),
110            );
111
112        match unwrap_to_internal_server_error!(
113            signed_entity_service
114                .get_last_cardano_transaction_snapshot()
115                .await,
116            logger => "proof_cardano_transaction::error"
117        ) {
118            Some(signed_entity) => {
119                let message = unwrap_to_internal_server_error!(
120                    build_response_message(prover_service, signed_entity, sanitized_hashes).await,
121                    logger => "proof_cardano_transaction"
122                );
123                Ok(reply::json(&message, StatusCode::OK))
124            }
125            None => {
126                warn!(logger, "proof_cardano_transaction::not_found");
127                Ok(reply::empty(StatusCode::NOT_FOUND))
128            }
129        }
130    }
131
132    pub async fn build_response_message(
133        prover_service: Arc<dyn ProverService>,
134        signed_entity: SignedEntity<CardanoTransactionsSnapshot>,
135        transaction_hashes: Vec<String>,
136    ) -> StdResult<CardanoTransactionsProofsMessage> {
137        let transactions_set_proofs = prover_service
138            .compute_transactions_proofs(
139                signed_entity.artifact.block_number,
140                transaction_hashes.as_slice(),
141            )
142            .await?;
143        let message = ToCardanoTransactionsProofsMessageAdapter::try_adapt(
144            signed_entity,
145            transactions_set_proofs,
146            transaction_hashes,
147        )?;
148
149        Ok(message)
150    }

178        state: RouterState,
179    ) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
180        let cors = warp::cors()

184
185        warp::any().and(routes(&state).with(cors))
186    }

435    fn sanitize_cardano_transaction_proof_query_params_remove_duplicate() {
436        let tx1 = fake_data::transaction_hashes()[0].to_string();
437        let tx2 = fake_data::transaction_hashes()[1].to_string();
438
439        // We are testing on an unordered list of transaction hashes
440        // as some rust dedup methods only remove consecutive duplicates
441        let params = CardanoTransactionProofQueryParams {
442            transaction_hashes: format!("{tx1},{tx2},{tx2},{tx1},{tx2}",),
443        };
444
445        assert_equivalent(params.sanitize(), vec![tx1, tx2]);
