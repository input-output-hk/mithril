8struct CardanoTransactionProofQueryParams {
9    transaction_hashes: String,
10}
11
12impl CardanoTransactionProofQueryParams {
13    pub fn split_transactions_hashes(&self) -> Vec<String> {
14        self.transaction_hashes.split(',').map(|s| s.to_string()).collect()
15    }
16
17    pub fn sanitize(&self) -> Vec<String> {
18        let mut transaction_hashes = self.split_transactions_hashes();
19        transaction_hashes.sort();
20        transaction_hashes.dedup();
21        transaction_hashes
22    }
23}

126    pub async fn proof_cardano_transaction(
127        client_metadata: ClientMetadata,
128        transaction_parameters: CardanoTransactionProofQueryParams,
129        logger: Logger,
130        signed_entity_service: Arc<dyn SignedEntityService>,
131        validator: ProverHashValidator,
132        prover_service: Arc<dyn LegacyProverService>,
133        metrics_service: Arc<MetricsService>,
134    ) -> Result<impl warp::Reply, Infallible> {
135        metrics_service
136            .get_proof_cardano_transaction_total_proofs_served_since_startup()
137            .increment(&[
138                client_metadata.origin_tag.as_deref().unwrap_or_default(),
139                client_metadata.client_type.as_deref().unwrap_or_default(),
140            ]);
141
142        let transaction_hashes = transaction_parameters.split_transactions_hashes();
143        debug!(
144            logger, ">> proof_cardano_transaction";
145            "transaction_hashes" => &transaction_parameters.transaction_hashes
146        );
147
148        if let Err(error) = validator.validate(&transaction_hashes) {
149            warn!(logger, "proof_cardano_transaction::bad_request");
150            return Ok(reply::bad_request(error.label, error.message));
151        }
152
153        let sanitized_hashes = transaction_parameters.sanitize();
154
155        // Fallback to 0, it should be impossible to have more than u32::MAX transactions.
156        metrics_service
157            .get_proof_cardano_transaction_total_transactions_served_since_startup()
158            .increment_by(
159                &[
160                    client_metadata.origin_tag.as_deref().unwrap_or_default(),
161                    client_metadata.client_type.as_deref().unwrap_or_default(),
162                ],
163                sanitized_hashes.len().try_into().unwrap_or(0),
164            );
165
166        match unwrap_to_internal_server_error!(
167            signed_entity_service
168                .get_last_cardano_transaction_snapshot()
169                .await,
170            logger => "proof_cardano_transaction::error"
171        ) {
172            Some(signed_entity) => {
173                let message = unwrap_to_internal_server_error!(
174                    build_response_message(prover_service, signed_entity, sanitized_hashes).await,
175                    logger => "proof_cardano_transaction"
176                );
177                Ok(reply::json(&message, StatusCode::OK))
178            }
179            None => {
180                warn!(logger, "proof_cardano_transaction::not_found");
181                Ok(reply::empty(StatusCode::NOT_FOUND))
182            }
183        }
184    }

312    pub async fn build_response_message(
313        prover_service: Arc<dyn LegacyProverService>,
314        signed_entity: SignedEntity<CardanoTransactionsSnapshot>,
315        transaction_hashes: Vec<String>,
316    ) -> StdResult<CardanoTransactionsProofsMessage> {
317        let transactions_set_proofs = prover_service
318            .compute_transactions_proofs(
319                signed_entity.artifact.block_number,
320                transaction_hashes.as_slice(),
321            )
322            .await?;
323        let message = ToCardanoTransactionsProofsMessageAdapter::try_adapt(
324            signed_entity,
325            transactions_set_proofs,
326            transaction_hashes,
327        )?;
328
329        Ok(message)
330    }

332    async fn build_response_message_for_v2_cardano_transaction(
333        prover_service: Arc<dyn LegacyProverService>,
334        signed_entity: SignedEntity<CardanoBlocksTransactionsSnapshot>,
335        transaction_hashes: Vec<String>,
336    ) -> StdResult<CardanoTransactionsProofsMessage> {
337        let transactions_set_proofs = prover_service
338            .compute_transactions_proofs(
339                signed_entity.artifact.block_number_signed,
340                transaction_hashes.as_slice(),
341            )
342            .await?;
343        let certified_transactions = transactions_set_proofs
344            .iter()
345            .flat_map(|proof| proof.transactions_hashes().to_vec())
346            .collect::<Vec<_>>();
347        let non_certified_transactions = transaction_hashes
348            .iter()
349            .filter(|hash| !certified_transactions.contains(hash))
350            .cloned()
351            .collect::<Vec<_>>();
352        let certified_transactions = transactions_set_proofs
353            .into_iter()
354            .map(TryInto::try_into)
355            .collect::<StdResult<Vec<CardanoTransactionsSetProofMessagePart>>>(
356        )?;
357        let message = CardanoTransactionsProofsMessage::new(
358            &signed_entity.certificate_id,
359            certified_transactions,
360            non_certified_transactions,
361            signed_entity.artifact.block_number_signed,
362        );
363
364        Ok(message)
365    }

367    async fn build_response_message_for_v2_cardano_block(

369        signed_entity: SignedEntity<CardanoBlocksTransactionsSnapshot>,
370        block_hashes: Vec<String>,
371    ) -> StdResult<CardanoTransactionsProofsMessage> {
372        // TODO(#2987): replace this temporary legacy call with the new prover once block proofs are available.
373        let transactions_set_proofs = prover_service
374            .compute_transactions_proofs(
375                signed_entity.artifact.block_number_signed,
376                block_hashes.as_slice(),
377            )
378            .await?;
379        let certified_transactions = transactions_set_proofs
380            .into_iter()
381            .map(TryInto::try_into)
382            .collect::<StdResult<Vec<_>>>()?;
383        let non_certified_transactions = block_hashes;
384
385        Ok(CardanoTransactionsProofsMessage::new(
386            &signed_entity.certificate_id,
387            certified_transactions,
388            non_certified_transactions,
389            signed_entity.artifact.block_number_signed,
390        ))

1043        let tx2 = fake_data::transaction_hashes()[1].to_string();
