14/// Match the given result and do an early return with an internal server error (500)
15/// if it was an Error. Else return the unwrapped value.
16#[macro_export]
17macro_rules! unwrap_to_internal_server_error {
18    ($code:expr, $logger:expr => $($warn_comment:tt)*) => {
19        match $code {
20            Ok(res) => res,
21            Err(err) => {
22                slog::warn!($logger, $($warn_comment)*; "error" => ?err);
23                return Ok($crate::http_server::routes::reply::server_error(
24                    err,
25                ));
26            }
27        }
28    };
29}
