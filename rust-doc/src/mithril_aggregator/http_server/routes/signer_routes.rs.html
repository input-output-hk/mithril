14pub fn routes(
15    router_state: &RouterState,
16) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
17    register_signer(router_state)
18        .or(registered_signers(router_state))
19        .or(signers_tickers(router_state))
20}

22/// POST /register-signer
23fn register_signer(
24    router_state: &RouterState,
25) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
26    warp::path!("register-signer")
27        .and(warp::post())
28        .and(middlewares::with_origin_tag(router_state))
29        .and(warp::header::optional::<String>(
30            MITHRIL_SIGNER_VERSION_HEADER,
31        ))
32        .and(warp::body::json())
33        .and(middlewares::with_logger(router_state))
34        .and(middlewares::with_signer_registerer(router_state))
35        .and(middlewares::with_event_transmitter(router_state))
36        .and(middlewares::with_epoch_service(router_state))
37        .and(middlewares::with_metrics_service(router_state))
38        .and_then(handlers::register_signer)
39}

41/// Get /signers/tickers
42fn signers_tickers(
43    router_state: &RouterState,
44) -> impl Filter<Extract = (impl warp::Reply,), Error = warp::Rejection> + Clone {
45    warp::path!("signers" / "tickers")
46        .and(warp::get())
47        .and(middlewares::with_logger(router_state))
48        .and(middlewares::extract_config(router_state, |config| {
49            config.network.to_string()
50        }))
51        .and(middlewares::with_signer_getter(router_state))
52        .and_then(handlers::signers_tickers)
53}

55/// Get /signers/registered/:epoch
56fn registered_signers(
57    router_state: &RouterState,

67async fn fetch_epoch_header_value(
68    epoch_service: EpochServiceWrapper,
69    logger: &slog::Logger,
70) -> String {
71    match epoch_service.read().await.epoch_of_current_data() {
72        Ok(epoch) => format!("{epoch}"),
73        Err(e) => {
74            warn!(logger, "Could not fetch epoch header value from Epoch service"; "error" => ?e);
75            String::new()
76        }
77    }
78}

80async fn compute_registration_epoch(
81    registered_at: &str,
82    epoch_service: EpochServiceWrapper,
83) -> StdResult<Epoch> {
84    if registered_at.to_lowercase() == "latest" {
85        epoch_service.read().await.epoch_of_current_data()
86    } else {
87        registered_at
88            .parse::<u64>()
89            .map(Epoch)
90            .with_context(|| "Invalid epoch: must be a number or 'latest'")
91    }
92}

112    /// Register Signer
113    #[allow(clippy::too_many_arguments)]
114    pub async fn register_signer(
115        origin_tag: Option<String>,
116        signer_node_version: Option<String>,
117        register_signer_message: RegisterSignerMessage,
118        logger: Logger,
119        signer_registerer: Arc<dyn SignerRegisterer>,
120        event_transmitter: Arc<TransmitterService<EventMessage>>,
121        epoch_service: EpochServiceWrapper,
122        metrics_service: Arc<MetricsService>,
123    ) -> Result<impl warp::Reply, Infallible> {
124        debug!(logger, ">> register_signer"; "payload" => ?register_signer_message);
125
126        metrics_service
127            .get_signer_registration_total_received_since_startup()
128            .increment(&[origin_tag.as_deref().unwrap_or_default()]);
129
130        let registration_epoch = register_signer_message.epoch;
131
132        let signer = match FromRegisterSignerAdapter::try_adapt(register_signer_message) {
133            Ok(signer) => signer,
134            Err(err) => {
135                warn!(logger,"register_signer::payload decoding error"; "error" => ?err);
136                return Ok(reply::bad_request(
137                    "Could not decode signer payload".to_string(),
138                    err.to_string(),
139                ));
140            }
141        };
142
143        let epoch_str = fetch_epoch_header_value(epoch_service, &logger).await;
144
145        match signer_registerer
146            .register_signer(registration_epoch, &signer)
147            .await
148        {
149            Ok(signer_with_stake) => {
150                event_transmitter.send(EventMessage::signer_registration(
151                    "HTTP::signer_register",
152                    &signer_with_stake,
153                    signer_node_version,
154                    epoch_str.as_str(),
155                ));
156
157                Ok(reply::empty(StatusCode::CREATED))
158            }
159            Err(SignerRegistrationError::ExistingSigner(signer_with_stake)) => {
160                debug!(logger, "register_signer::already_registered");
161
162                event_transmitter.send(EventMessage::signer_registration(
163                    "HTTP::signer_register",
164                    &signer_with_stake,
165                    signer_node_version,
166                    epoch_str.as_str(),
167                ));
168
169                Ok(reply::empty(StatusCode::CREATED))
170            }
171            Err(SignerRegistrationError::FailedSignerRegistration(err)) => {
172                warn!(logger,"register_signer::failed_signer_registration"; "error" => ?err);
173                Ok(reply::bad_request(
174                    "failed_signer_registration".to_string(),
175                    err.to_string(),
176                ))
177            }
178            Err(SignerRegistrationError::RegistrationRoundNotYetOpened) => {
179                warn!(logger, "register_signer::registration_round_not_yed_opened");
180                Ok(reply::server_error(
181                    SignerRegistrationError::RegistrationRoundNotYetOpened,
182                ))
183            }
184            Err(err) => {
185                warn!(logger,"register_signer::error"; "error" => ?err);
186                Ok(reply::server_error(err))
187            }
188        }
189    }

191    /// Get Registered Signers for a given epoch
192    pub async fn registered_signers(
193        registered_at: String,
194        logger: Logger,
195        epoch_service: EpochServiceWrapper,
196        verification_key_store: Arc<dyn VerificationKeyStorer>,
197    ) -> Result<impl warp::Reply, Infallible> {
198        let registered_at_epoch =
199            match compute_registration_epoch(&registered_at, epoch_service).await {
200                Ok(epoch) => epoch,
201                Err(err) => {
202                    warn!(logger,"registered_signers::invalid_epoch"; "error" => ?err);
203                    return Ok(reply::bad_request(
204                        "invalid_epoch".to_string(),
205                        err.to_string(),
206                    ));
207                }
208            };
209
210        // The given epoch is the epoch at which the signer registered, the store works on
211        // the recording epoch so we need to offset.
212        match verification_key_store
213            .get_signers(registered_at_epoch.offset_to_recording_epoch())
214            .await
215        {
216            Ok(Some(signers)) => {
217                let message = SignerRegistrationsMessage::new(registered_at_epoch, signers);
218                Ok(reply::json(&message, StatusCode::OK))
219            }
220            Ok(None) => {
221                warn!(logger, "registered_signers::not_found");
222                Ok(reply::empty(StatusCode::NOT_FOUND))
223            }
224            Err(err) => {
225                warn!(logger,"registered_signers::error"; "error" => ?err);
226                Ok(reply::server_error(err))
227            }
228        }
229    }

231    pub async fn signers_tickers(
232        logger: Logger,
233        network: String,
234        signer_getter: Arc<dyn SignerGetter>,
235    ) -> Result<impl warp::Reply, Infallible> {
236        match signer_getter.get_all().await {
237            Ok(signers) => {
238                let signers: Vec<_> = signers
239                    .into_iter()
240                    .map(|s| SignerTickerListItemMessage {
241                        party_id: s.signer_id,
242                        pool_ticker: s.pool_ticker,
243                        has_registered: s.last_registered_at.is_some(),
244                    })
245                    .collect();
246                Ok(reply::json(
247                    &SignersTickersMessage { network, signers },
248                    StatusCode::OK,
249                ))
250            }
251            Err(err) => {
252                warn!(logger,"registered_signers::error"; "error" => ?err);
253                Ok(reply::server_error(err))
254            }
255        }
256    }
