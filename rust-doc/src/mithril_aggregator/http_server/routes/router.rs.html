114/// Routes
115pub fn routes(
116    state: Arc<RouterState>,
117) -> impl Filter<Extract = (impl Reply,), Error = Rejection> + Clone {
118    let cors = warp::cors()
119        .allow_any_origin()
120        .allow_headers(vec![
121            "content-type",
122            MITHRIL_API_VERSION_HEADER,
123            MITHRIL_ORIGIN_TAG_HEADER,
124            MITHRIL_CLIENT_TYPE_HEADER,
125        ])
126        .allow_methods(vec![Method::GET, Method::POST, Method::OPTIONS]);
127
128    warp::any()
129        .and(warp::path(SERVER_BASE_PATH))
130        .and(
131            certificate_routes::routes(&state)
132                .or(artifact_routes::snapshot::routes(&state))
133                .or(artifact_routes::cardano_database::routes(&state))
134                .or(artifact_routes::mithril_stake_distribution::routes(&state))
135                .or(artifact_routes::cardano_stake_distribution::routes(&state))
136                .or(artifact_routes::cardano_transaction::routes(&state))
137                .or(proof_routes::routes(&state))
138                .or(signer_routes::routes(&state))
139                .or(signatures_routes::routes(&state))
140                .or(epoch_routes::routes(&state))
141                .or(statistics_routes::routes(&state))
142                .or(root_routes::routes(&state))
143                .or(status::routes(&state)),
144        )
145        .recover(handle_custom)
146        .and(middlewares::with_api_version_provider(&state))
147        .map(|reply, api_version_provider: Arc<APIVersionProvider>| {
148            warp::reply::with_header(
149                reply,
150                MITHRIL_API_VERSION_HEADER,
151                &api_version_provider.compute_current_version().unwrap().to_string(),
152            )
153        })
154        .with(cors)
155        .with(middlewares::log_route_call(&state))
156}
157
158pub async fn handle_custom(reject: Rejection) -> Result<impl Reply, Rejection> {
159    if reject.is_not_found() {
160        Ok(StatusCode::NOT_FOUND)
161    } else {
162        Err(reject)
163    }
164}

176    #[tokio::test]
177    async fn test_404_response_should_include_status_code_and_headers() {
178        let container = Arc::new(initialize_dependencies!().await);
179        let state = RouterState::new_with_dummy_config(container);
180        let routes = routes(Arc::new(state));
181
182        let response = warp::test::request()
183            .path("/aggregator/a-route-that-does-not-exist")
184            // We need to set the Origin header to trigger the CORS middleware
185            .header("Origin", "http://localhost")
186            .reply(&routes)
187            .await;
188        let response_headers = response.headers();
189
190        assert_eq!(response.status(), StatusCode::NOT_FOUND);
191        assert!(
192            response_headers.get(MITHRIL_API_VERSION_HEADER).is_some(),
193            "API version header should be present, headers: {response_headers:?}",
194        );
195        assert!(
196            response_headers.get("access-control-allow-origin").is_some(),
197            "CORS headers should be present, headers: {response_headers:?}",
198        );
199    }

201    #[tokio::test]
202    async fn test_authorized_request_headers() {
203        fn request_with_access_control_request_headers(headers: String) -> RequestBuilder {
204            warp::test::request()
205                .method("OPTIONS")
206                .path("/aggregator")
207                // We need to set the Origin header to trigger the CORS middleware
208                .header("Origin", "http://localhost")
209                .header("access-control-request-method", "GET")
210                .header("access-control-request-headers", headers)
211        }
212
213        let container = Arc::new(initialize_dependencies!().await);
214        let state = RouterState::new_with_dummy_config(container);
215        let routes = routes(Arc::new(state));
216
217        assert!(
218            !request_with_access_control_request_headers("unauthorized_header".to_string())
219                .reply(&routes)
220                .await
221                .status()
222                .is_success()
223        );
224
225        assert!(request_with_access_control_request_headers(format!(
226            "{MITHRIL_API_VERSION_HEADER},{MITHRIL_ORIGIN_TAG_HEADER},{MITHRIL_CLIENT_TYPE_HEADER}"
227        ))
228        .reply(&routes)
229        .await
230        .status()
231        .is_success());
