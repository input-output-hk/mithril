11    use mithril_common::entities::SignedEntityType;
12    use mithril_common::signable_builder::{Artifact, SignedEntity};
13
14    pub fn create_signed_entity<T>(
15        signed_entity_type: SignedEntityType,
16        record: T,
17    ) -> SignedEntity<T>
18    where
19        T: Artifact,
20    {
21        let mut entities = create_signed_entities(signed_entity_type, vec![record]);
22        entities.pop().unwrap()
23    }
24
25    pub fn create_signed_entities<T>(
26        signed_entity_type: SignedEntityType,
27        records: Vec<T>,
28    ) -> Vec<SignedEntity<T>>
29    where
30        T: Artifact,
31    {
32        records
33            .into_iter()
34            .enumerate()
35            .map(|(idx, record)| SignedEntity {
36                signed_entity_id: format!("{idx}"),
37                signed_entity_type: signed_entity_type.to_owned(),
38                certificate_id: format!("certificate-{idx}"),
39                artifact: record,
40                created_at: DateTime::parse_from_rfc3339("2023-01-19T13:43:05.618857482Z")
41                    .unwrap()
42                    .with_timezone(&Utc),
43            })
44            .collect()
45    }
