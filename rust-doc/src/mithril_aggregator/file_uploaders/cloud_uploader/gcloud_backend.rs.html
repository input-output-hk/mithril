22/// Google Cloud Platform file uploader using `gcloud-storage` crate
23pub struct GCloudBackendUploader {
24    bucket: String,
25    use_cdn_domain: bool,
26    storage_client: GcpStorageClient,
27    logger: Logger,
28}
29
30impl GCloudBackendUploader {
31    /// Creates a new instance of `GCloudBackendUploader`
32    pub async fn try_new(
33        bucket: String,
34        use_cdn_domain: bool,
35        credentials_json_env_var: String,
36        logger: Logger,
37    ) -> StdResult<Self> {
38        const BASE_ERROR_CONTEXT: &str =
39            "Failed to create Google Cloud Storage client for file uploading";
40        let unparsed_credentials_json = std::env::var(&credentials_json_env_var)
41            .with_context(|| {
42                format!("Environment variable `{credentials_json_env_var}` must be set")
43            })
44            .with_context(|| BASE_ERROR_CONTEXT)?;
45        let credentials_file = CredentialsFile::new_from_str(&unparsed_credentials_json)
46            .await
47            .with_context(|| BASE_ERROR_CONTEXT)?;
48        let config = ClientConfig::default()
49            .with_credentials(credentials_file)
50            .await
51            .with_context(|| BASE_ERROR_CONTEXT)?;
52
53        Ok(Self {
54            bucket,
55            use_cdn_domain,
56            storage_client: GcpStorageClient::new(config),
57            logger,
58        })
59    }
60}

62#[async_trait]
63impl CloudBackendUploader for GCloudBackendUploader {
64    async fn file_exists(&self, remote_file_path: &CloudRemotePath) -> StdResult<Option<FileUri>> {
65        info!(self.logger, "Reading file metadata {remote_file_path}");
66        let request = GetObjectRequest {
67            bucket: self.bucket.clone(),
68            object: remote_file_path.to_string(),
69            ..Default::default()
70        };
71
72        let file_uri = match self
73            .storage_client
74            .get_object(&request)
75            .await
76            .with_context(|| "remote reading file metadata failure")
77        {
78            Ok(_) => {
79                info!(self.logger, "Found file metadata {remote_file_path}");
80
81                Some(remote_file_path.to_gcloud_storage_location(&self.bucket, self.use_cdn_domain))
82            }
83            Err(_) => {
84                info!(self.logger, "Missing file metadata {remote_file_path}");
85
86                None
87            }
88        };
89
90        Ok(file_uri)
91    }
92
93    async fn upload_file(
94        &self,
95        local_file_path: &Path,
96        remote_file_path: &CloudRemotePath,
97    ) -> StdResult<FileUri> {
98        info!(
99            self.logger,
100            "Uploading {} to {remote_file_path}",
101            local_file_path.display()
102        );
103        let file = tokio::fs::File::open(local_file_path).await.with_context(|| {
104            format!("Opening file failure, path: {}", local_file_path.display())
105        })?;
106        let stream = FramedRead::new(file, BytesCodec::new());
107
108        self.storage_client
109            .upload_streamed_object(
110                &UploadObjectRequest {
111                    bucket: self.bucket.clone(),
112                    ..Default::default()
113                },
114                stream,
115                &UploadType::Simple(Media::new(remote_file_path.to_string())),
116            )
117            .await
118            .with_context(|| "Remote uploading failure")?;
119        info!(
120            self.logger,
121            "Uploaded {} to {remote_file_path}",
122            local_file_path.display()
123        );
124
125        Ok(remote_file_path.to_gcloud_storage_location(&self.bucket, self.use_cdn_domain))
126    }
127
128    async fn make_file_public(&self, remote_file_path: &CloudRemotePath) -> StdResult<()> {
129        let new_bucket_access_control = ObjectAccessControlCreationConfig {
130            entity: "allUsers".to_string(),
131            role: ObjectACLRole::READER,
132        };
133        info!(
134            self.logger, "Updating acl for {remote_file_path}";
135            "inserted_acl" => ?new_bucket_access_control
136        );
137        self.storage_client
138            .insert_object_access_control(&InsertObjectAccessControlRequest {
139                bucket: self.bucket.clone(),
140                object: gcp_percent_encode(&remote_file_path.to_string()),
141                acl: new_bucket_access_control,
142                ..Default::default()
143            })
144            .await
145            .with_context(|| "Updating acl failure")?;
146
147        info!(self.logger, "Updated acl for {remote_file_path}");
148
149        Ok(())
150    }
151}
