10/// A type alias for the party index used in KES cryptographic material.
11pub type KesPartyIndexForTest = u64;
12
13/// A struct to hold KES cryptographic material for testing purposes.
14pub(crate) struct KesCryptographicMaterialForTest {
15    #[allow(dead_code)]
16    pub party_id: ProtocolPartyId,
17    pub operational_certificate_file: PathBuf,
18    pub kes_secret_key_file: PathBuf,
19}
20
21/// Create KES cryptographic material for testing purposes.
22pub fn create_kes_cryptographic_material(
23    party_idx: KesPartyIndexForTest,
24    kes_period: KesPeriod,
25    test_directory: &str,
26) -> KesCryptographicMaterialForTest {
27    let temp_dir = std::env::temp_dir()
28        .join("create_kes_cryptographic_material")
29        .join(format!("{test_directory}_{party_idx}"));
30    fs::create_dir_all(&temp_dir).expect("temp dir creation should not fail");
31    let keypair = ColdKeyGenerator::create_deterministic_keypair([party_idx as u8; 32]);
32    let mut dummy_buffer = [0u8; Sum6Kes::SIZE + 4];
33    let mut dummy_seed = [party_idx as u8; 32];
34    let (mut kes_secret_key, kes_verification_key) =
35        Sum6Kes::keygen(&mut dummy_buffer, &mut dummy_seed);
36    for _ in 0..kes_period {
37        kes_secret_key
38            .update()
39            .expect("KES secret key update should not fail");
40    }
41    let mut kes_bytes = Sum6KesBytes([0u8; Sum6Kes::SIZE + 4]);
42    kes_bytes.0.copy_from_slice(&kes_secret_key.clone_sk());
43    let operational_certificate = OpCert::new(kes_verification_key, 0, 0, keypair);
44    let kes_secret_key_file = temp_dir.join(format!("kes{party_idx}.skey"));
45    kes_bytes
46        .to_file(&kes_secret_key_file)
47        .expect("KES secret key file export should not fail");
48    let operational_certificate_file = temp_dir.join(format!("pool{party_idx}.cert"));
49    operational_certificate
50        .to_file(&operational_certificate_file)
51        .expect("operational certificate file export should not fail");
52    let party_id = operational_certificate
53        .compute_protocol_party_id()
54        .expect("compute protocol party id should not fail");
55
56    KesCryptographicMaterialForTest {
57        party_id,
58        operational_certificate_file,
59        kes_secret_key_file,
60    }
61}
