407    fn assert_decode_all<T: Fn(&str) -> Result<(), String>>(encoded_types: &[&str], decode_fun: T) {
408        let errors: Vec<String> = encoded_types
409            .iter()
410            .filter_map(|encoded_type| match decode_fun(encoded_type) {
411                Ok(..) => None,
412                Err(error_src) => {
413                    let error = format!(
414                        "> The encoded key does not match anymore the structure of the type: `{}`\
415                        \n  * error: {error_src}\
416                        \n  * encoded_string: {encoded_type}",
417                        type_name::<T>()
418                    );
419
420                    Some(error)
421                }
422            })
423            .collect();
424
425        assert!(
426            errors.is_empty(),
427            "At least one of the encoded values could not be decoded:\n{}",
428            errors.join("\n")
429        );
430    }
431
432    fn assert_can_deserialize_using_key_decode_hex<T: Serialize + DeserializeOwned>(
433        encoded_types: &[&str],
434    ) {
435        assert_decode_all(encoded_types, |encoded_type| {
436            key_decode_hex::<T>(encoded_type)
437                .map(|_| ())
438                .map_err(|e| format!("{e:?}"))
439        })
440    }
441
442    fn assert_can_convert_to_protocol_key<
443        T: ProtocolKeyCodec<T> + Serialize + DeserializeOwned + TryToBytes + TryFromBytes,
444    >(
445        encoded_types: &[&str],
446    ) {
447        assert_decode_all(
448            encoded_types,
449            |encoded_type| match ProtocolKey::<T>::try_from(encoded_type) {
450                Ok(_) => Ok(()),
451                Err(error) => Err(format!("{error:?}")),
452            },
453        )
454    }
