14/// Signer with Stake Message
15#[derive(Clone, PartialEq, Eq, Default, Serialize, Deserialize)]
16pub struct SignerWithStakeMessagePart {
17    /// The unique identifier of the signer
18    ///
19    /// Used only for testing when SPO pool id is not certified
20    pub party_id: PartyId,
21
22    /// The public key used to authenticate signer signature
23    pub verification_key: HexEncodedVerificationKey,
24
25    /// The encoded signer 'Mithril verification key' signature (signed by the
26    /// Cardano node KES secret key).
27    ///
28    /// None is used only for testing when SPO pool id is not certified
29    #[serde(skip_serializing_if = "Option::is_none")]
30    pub verification_key_signature: Option<HexEncodedVerificationKeySignature>,
31
32    /// The encoded operational certificate of stake pool operator attached to
33    /// the signer node.
34    ///
35    /// None is used only for testing when SPO pool id is not certified
36    #[serde(skip_serializing_if = "Option::is_none")]
37    pub operational_certificate: Option<HexEncodedOpCert>,
38
39    /// The number of evolutions of the KES key since the start KES period of the operational certificate at the time of signature.
40    #[serde(rename = "kes_period", skip_serializing_if = "Option::is_none")]
41    pub kes_evolutions: Option<KesEvolutions>,
42
43    /// The signer stake
44    pub stake: Stake,
45}
46
47impl SignerWithStakeMessagePart {
48    /// Convert a set of signers into message parts
49    pub fn from_signers(signers: Vec<SignerWithStake>) -> Vec<Self> {
50        signers.into_iter().map(|signer| signer.into()).collect()
51    }
52
53    /// Convert a set of signer message parts into a set of signers with stake
54    pub fn try_into_signers(messages: Vec<Self>) -> StdResult<Vec<SignerWithStake>> {
55        messages
56            .into_iter()
57            .map(SignerWithStakeMessagePart::try_into)
58            .collect()
59    }
60}
61
62impl TryInto<SignerWithStake> for SignerWithStakeMessagePart {
63    type Error = StdError;
64
65    fn try_into(self) -> Result<SignerWithStake, Self::Error> {
66        let verification_key_signature: Option<ProtocolSignerVerificationKeySignature> = self
67            .verification_key_signature
68            .map(|f| f.try_into())
69            .transpose()
70            .with_context(|| {
71                format!(
72                    "Error while parsing verification key signature message, party_id = '{}'",
73                    self.party_id
74                )
75            })?;
76        let operational_certificate: Option<ProtocolOpCert> = self
77            .operational_certificate
78            .map(|f| f.try_into())
79            .transpose()
80            .with_context(|| {
81                format!(
82                    "Error while parsing operational certificate message, party_id = '{}'.",
83                    self.party_id
84                )
85            })?;
86        let value = SignerWithStake {
87            party_id: self.party_id,
88            verification_key: self.verification_key.try_into()?,
89            verification_key_signature,
90            kes_evolutions: self.kes_evolutions,
91            operational_certificate,
92            stake: self.stake,
93        };
94        Ok(value)
95    }
96}

98impl From<SignerWithStake> for SignerWithStakeMessagePart {
