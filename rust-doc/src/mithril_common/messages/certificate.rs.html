13/// Message structure of a certificate
14#[derive(Clone, PartialEq, Serialize, Deserialize)]
15pub struct CertificateMessage {
16    /// Hash of the current certificate
17    /// Computed from the other fields of the certificate
18    /// aka H(Cp,n))
19    pub hash: String,
20
21    /// Hash of the previous certificate in the chain
22    /// This is either the hash of the first certificate of the epoch in the chain
23    /// Or the first certificate of the previous epoch in the chain (if the certificate is the first of its epoch)
24    /// aka H(FC(n))
25    pub previous_hash: String,
26
27    /// Epoch of the Cardano chain
28    pub epoch: Epoch,
29
30    /// The signed entity type of the message.
31    /// aka BEACON(p,n)
32    pub signed_entity_type: SignedEntityType,
33
34    /// Certificate metadata
35    /// aka METADATA(p,n)
36    pub metadata: CertificateMetadataMessagePart,
37
38    /// Structured message that is used to create the signed message
39    /// aka MSG(p,n) U AVK(n-1)
40    pub protocol_message: ProtocolMessage,
41
42    /// Message that is signed by the signers
43    /// aka H(MSG(p,n) || AVK(n-1))
44    pub signed_message: String,
45
46    /// Aggregate verification key
47    /// The AVK used to sign during the current epoch
48    /// aka AVK(n-2)
49    pub aggregate_verification_key: String,
50
51    /// STM multi signature created from a quorum of single signatures from the signers
52    /// aka MULTI_SIG(H(MSG(p,n) || AVK(n-1)))
53    pub multi_signature: String,
54
55    /// Genesis signature created from the original stake distribution
56    /// aka GENESIS_SIG(AVK(-1))
57    pub genesis_signature: String,
58}
59
60impl CertificateMessage {
61    /// Check that the certificate signed message match the given protocol message.
62    pub fn match_message(&self, message: &ProtocolMessage) -> bool {
63        message.compute_hash() == self.signed_message
64    }
65}

67impl Debug for CertificateMessage {
68    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
69        let should_be_exhaustive = f.alternate();
70        let mut debug = f.debug_struct("Certificate");
71        debug
72            .field("hash", &self.hash)
73            .field("previous_hash", &self.previous_hash)
74            .field("epoch", &format_args!("{:?}", self.epoch))
75            .field(
76                "signed_entity_type",
77                &format_args!("{:?}", self.signed_entity_type),
78            )
79            .field("metadata", &format_args!("{:?}", self.metadata))
80            .field(
81                "protocol_message",
82                &format_args!("{:?}", self.protocol_message),
83            )
84            .field("signed_message", &self.signed_message);
85
86        match should_be_exhaustive {
87            true => debug
88                .field(
89                    "aggregate_verification_key",
90                    &self.aggregate_verification_key,
91                )
92                .field("multi_signature", &self.multi_signature)
93                .field("genesis_signature", &self.genesis_signature)
94                .finish(),
95            false => debug.finish_non_exhaustive(),
96        }
97    }
98}

100impl TryFrom<CertificateMessage> for Certificate {
101    type Error = StdError;
102
103    fn try_from(certificate_message: CertificateMessage) -> Result<Self, Self::Error> {
104        let metadata = CertificateMetadata {
105            network: certificate_message.metadata.network,
106            protocol_version: certificate_message.metadata.protocol_version,
107            protocol_parameters: certificate_message.metadata.protocol_parameters,
108            initiated_at: certificate_message.metadata.initiated_at,
109            sealed_at: certificate_message.metadata.sealed_at,
110            signers: certificate_message.metadata.signers,
111        };
112
113        let certificate = Certificate {
114            hash: certificate_message.hash,
115            previous_hash: certificate_message.previous_hash,
116            epoch: certificate_message.epoch,
117            metadata,
118            protocol_message: certificate_message.protocol_message,
119            signed_message: certificate_message.signed_message,
120            aggregate_verification_key: certificate_message
121                .aggregate_verification_key
122                .try_into()
123                .with_context(|| {
124                "Can not convert message to certificate: can not decode the aggregate verification key"
125            })?,
126            signature: if certificate_message.genesis_signature.is_empty() {
127                CertificateSignature::MultiSignature(
128                    certificate_message.signed_entity_type,
129                    certificate_message
130                        .multi_signature
131                        .try_into()
132                        .with_context(|| {
133                            "Can not convert message to certificate: can not decode the multi-signature"
134                        })?,
135                )
136            } else {
137                CertificateSignature::GenesisSignature(
138                    certificate_message
139                        .genesis_signature
140                        .try_into()
141                        .with_context(|| {
142                            "Can not convert message to certificate: can not decode the genesis signature"
143                        })?,
144                )
145            },
146        };
147
148        Ok(certificate)
149    }
150}

152impl TryFrom<Certificate> for CertificateMessage {
153    type Error = StdError;
154
155    fn try_from(certificate: Certificate) -> Result<Self, Self::Error> {
156        let signed_entity_type = certificate.signed_entity_type();
157        let metadata = CertificateMetadataMessagePart {
158            network: certificate.metadata.network,
159            protocol_version: certificate.metadata.protocol_version,
160            protocol_parameters: certificate.metadata.protocol_parameters,
161            initiated_at: certificate.metadata.initiated_at,
162            sealed_at: certificate.metadata.sealed_at,
163            signers: certificate.metadata.signers,
164        };
165
166        let (multi_signature, genesis_signature) = match certificate.signature {
167            CertificateSignature::GenesisSignature(signature) => (
168                String::new(),
169                signature.to_bytes_hex().with_context(|| {
170                    "Can not convert certificate to message: can not encode the genesis signature"
171                })?,
172            ),
173            CertificateSignature::MultiSignature(_, signature) => (
174                signature.to_json_hex().with_context(|| {
175                    "Can not convert certificate to message: can not encode the multi-signature"
176                })?,
177                String::new(),
178            ),
179        };
180
181        let message = CertificateMessage {
182            hash: certificate.hash,
183            previous_hash: certificate.previous_hash,
184            epoch: certificate.epoch,
185            signed_entity_type,
186            metadata,
187            protocol_message: certificate.protocol_message,
188            signed_message: certificate.signed_message,
189            aggregate_verification_key: certificate
190                .aggregate_verification_key
191                .to_json_hex()
192                .with_context(|| {
193                    "Can not convert certificate to message: can not encode aggregate verification key"
194                })?,
195            multi_signature,
196            genesis_signature,
197        };
198
199        Ok(message)
200    }
201}
