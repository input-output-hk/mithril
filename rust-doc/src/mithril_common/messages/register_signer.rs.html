9/// Register Signer Message
10#[derive(Clone, PartialEq, Eq, Serialize, Deserialize)]
11pub struct RegisterSignerMessage {
12    /// Epoch at which registration is sent
13    pub epoch: Epoch,
14
15    /// The unique identifier of the signer
16    ///
17    /// Used only for testing when SPO pool id is not certified
18    pub party_id: PartyId,
19
20    /// The public key used to authenticate signer signature
21    pub verification_key: HexEncodedVerificationKey,
22
23    /// The encoded signer 'Mithril verification key' signature (signed by the
24    /// Cardano node KES secret key).
25    ///
26    /// None is used only for testing when SPO pool id is not certified
27    #[serde(skip_serializing_if = "Option::is_none")]
28    pub verification_key_signature: Option<HexEncodedVerificationKeySignature>,
29
30    /// The encoded operational certificate of stake pool operator attached to
31    /// the signer node.
32    ///
33    /// None is used only for testing when SPO pool id is not certified
34    #[serde(skip_serializing_if = "Option::is_none")]
35    pub operational_certificate: Option<HexEncodedOpCert>,
36
37    /// The number of evolutions of the KES key since the start KES period of the operational certificate at the time of signature.
38    #[serde(rename = "kes_period", skip_serializing_if = "Option::is_none")]
39    pub kes_evolutions: Option<KesEvolutions>,
40}
41
42impl Debug for RegisterSignerMessage {
43    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
44        let should_be_exhaustive = f.alternate();
45        let mut debug = f.debug_struct("Signer");
46        debug.field("party_id", &self.party_id).field("epoch", &self.epoch);
47
48        match should_be_exhaustive {
49            true => debug
50                .field(
51                    "verification_key",
52                    &format_args!("{:?}", self.verification_key),
53                )
54                .field(
55                    "verification_key_signature",
56                    &format_args!("{:?}", self.verification_key_signature),
57                )
58                .field(
59                    "operational_certificate",
60                    &format_args!("{:?}", self.operational_certificate),
61                )
62                .field("kes_evolutions", &format_args!("{:?}", self.kes_evolutions))
63                .finish(),
64            false => debug.finish_non_exhaustive(),
65        }
66    }
67}
