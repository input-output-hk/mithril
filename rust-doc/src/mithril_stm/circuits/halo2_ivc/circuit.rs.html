10#[derive(Clone, Debug)]
11pub struct IvcCircuit {
12    // Persistent values throughout an ivc stream. This is the root of trust for an ivc stream.
13    pub global: Value<Global>,
14    // State values from the last aggregated certificate
15    pub state: Value<State>,
16    // Witness (mainly the next certificate to be aggregated) for deriving the next state
17    pub witness: Value<Witness>,
18    // Snark proof of the next certificate
19    pub cert_proof: Value<Vec<u8>>,
20    // Latest IVC proof
21    pub self_proof: Value<Vec<u8>>,
22    // Latest Accumulator
23    pub acc: Value<Accumulator<S>>,
24    // Domain and ConstraintSystem associated with certificate circuit VerifyingKey
25    pub cert_domain_cs: (EvaluationDomain<F>, ConstraintSystem<F>),
26    // Domain and ConstraintSystem associated with ivc circuit VerifyingKey
27    pub self_domain_cs: (EvaluationDomain<F>, ConstraintSystem<F>),
28}
29
30impl IvcCircuit {
31    #[allow(clippy::too_many_arguments)]
32    pub fn new(
33        global: Global,
34        state: State,
35        witness: Witness,
36        cert_proof: Vec<u8>,
37        self_proof: Vec<u8>,
38        acc: Accumulator<S>,
39        cert_vk: &VerifyingKey<F, KZGCommitmentScheme<E>>,
40        self_vk: &VerifyingKey<F, KZGCommitmentScheme<E>>,
41    ) -> Self {
42        IvcCircuit {
43            global: Value::known(global),
44            state: Value::known(state),
45            witness: Value::known(witness),
46            cert_proof: Value::known(cert_proof),
47            self_proof: Value::known(self_proof),
48            acc: Value::known(acc),
49            cert_domain_cs: (cert_vk.get_domain().clone(), cert_vk.cs().clone()),
50            self_domain_cs: (self_vk.get_domain().clone(), self_vk.cs().clone()),
51        }
52    }
53
54    // Create a default ivc circuit for generating ivc proving key and verifying key
55    pub fn unknown(cert_vk: &VerifyingKey<F, KZGCommitmentScheme<E>>) -> Self {
56        let mut self_cs = ConstraintSystem::default();
57        configure_ivc_circuit(&mut self_cs);
58        let self_domain = EvaluationDomain::new(self_cs.degree() as u32, K);
59
60        IvcCircuit {
61            global: Value::unknown(),
62            state: Value::unknown(),
63            witness: Value::unknown(),
64            cert_proof: Value::unknown(),
65            self_proof: Value::unknown(),
66            acc: Value::unknown(),
67            cert_domain_cs: (cert_vk.get_domain().clone(), cert_vk.cs().clone()),
68            self_domain_cs: (self_domain, self_cs),
69        }
70    }
71}
72
73impl Circuit<F> for IvcCircuit {
74    type Config = IvcConfig;
75    type FloorPlanner = SimpleFloorPlanner;
76    type Params = ();
77
78    fn without_witnesses(&self) -> Self {
79        unreachable!()
80    }
81
82    fn configure(meta: &mut ConstraintSystem<F>) -> Self::Config {
83        configure_ivc_circuit(meta)
84    }
85
86    fn synthesize(
87        &self,
88        config: Self::Config,
89        mut layouter: impl Layouter<F>,
90    ) -> Result<(), Error> {
91        let ivc_gadget = IvcGadget::new(&config);
92
93        // Assign global and constraint it as public input
94        let global = ivc_gadget.assign_global_as_public_input(
95            &mut layouter,
96            &self.global,
97            &self.cert_domain_cs,
98            &self.self_domain_cs,
99        )?;
100        // Assign previous state
101        let state = ivc_gadget.assign_state(&mut layouter, &self.state)?;
102        // Assign witness for the new certificate to be aggregated
103        let witness = ivc_gadget.assign_witness(&mut layouter, &self.witness)?;
104
105        // If state.counter = 0, we are aggregating the genesis certificate
106        let is_genesis = ivc_gadget.is_genesis(&mut layouter, &state)?;
107        let is_not_genesis = ivc_gadget.native_gadget.not(&mut layouter, &is_genesis)?;
108
109        // Verify genesis certificate
110        ivc_gadget.assert_genesis(&mut layouter, &is_not_genesis, &global, &witness)?;
111
112        // Verify certificate chain link between the last aggregated certificate and the new certificate to obtain the next state
113        let next_state = ivc_gadget.transition(
114            &mut layouter,
115            &is_genesis,
116            &is_not_genesis,
117            &global,
118            &state,
119            &witness,
120        )?;
121        // Constrain the next state as public input
122        ivc_gadget.constrain_state_as_public_input(&mut layouter, &next_state)?;
123
124        // Verify (prepare) cert_proof and previous ivc_proof and update accumulator
125        let next_acc = ivc_gadget.verify_prepare(
126            &mut layouter,
127            &global,
128            &is_not_genesis,
129            &state,
130            &witness,
131            &self.cert_proof,
132            &self.self_proof,
133            &self.acc,
134        )?;
135        // Constrain the next accumulator as public input
136        ivc_gadget
137            .verifier_gadget
138            .constrain_as_public_input(&mut layouter, &next_acc)?;
139
140        ivc_gadget.core_decomp_chip.load(&mut layouter)?;
141        ivc_gadget.sha2_256_chip.load(&mut layouter)
142    }
143}
