12#[derive(Debug, Clone)]
13pub struct SigningKey(JubjubScalar);
14
15impl SigningKey {
16    pub fn generate(rng: &mut (impl RngCore + CryptoRng)) -> Self {
17        let sk = JubjubScalar::random(rng);
18        SigningKey(sk)
19    }
20
21    pub fn sign(&self, msg: &[JubjubBase], rng: &mut (impl RngCore + CryptoRng)) -> Signature {
22        let g = JubjubSubgroup::generator();
23        let vk = g * self.0;
24        let r = JubjubScalar::random(rng);
25        let cap_r = g * r;
26        let (vk_x, vk_y) = get_coordinates(vk);
27        let (cap_r_x, cap_r_y) = get_coordinates(cap_r);
28
29        let mut to_hash = vec![DST_SCHNORR_SIGNATURE, vk_x, vk_y, cap_r_x, cap_r_y];
30        to_hash.extend_from_slice(msg);
31
32        let c = PoseidonHash::hash(&to_hash);
33        let c_scalar = jubjub_base_to_scalar(c);
34        let s = r - self.0 * c_scalar;
35
36        Signature { s, c }
37    }
38}

40#[derive(Debug, Clone, Copy, Default)]
41pub struct VerificationKey(pub JubjubSubgroup);
42
43impl From<&SigningKey> for VerificationKey {
44    fn from(sk: &SigningKey) -> Self {
45        let g = JubjubSubgroup::generator();
46        let vk = g * sk.0;
47        VerificationKey(vk)
48    }
49}
50
51impl VerificationKey {
52    pub fn to_field(&self) -> [JubjubBase; 2] {
53        let (x, y) = get_coordinates(self.0);
54        [x, y]
55    }
56
57    pub fn to_bytes(&self) -> [u8; 64] {
58        let (x, y) = get_coordinates(self.0);
59        let mut bytes = [0u8; 64];
60        bytes[0..32].copy_from_slice(&x.to_bytes_le());
61        bytes[32..64].copy_from_slice(&y.to_bytes_le());
62        bytes
63    }
64
65    pub fn from_bytes(bytes: &[u8]) -> Result<Self, SignatureError> {
66        let bytes = bytes.get(0..64).ok_or(SignatureError::SerializationError)?;
67        let mut u_bytes = [0u8; 32];
68        u_bytes.copy_from_slice(&bytes[0..32]);
69        let mut v_bytes = [0u8; 32];
70        v_bytes.copy_from_slice(&bytes[32..64]);
71
72        let u = JubjubBase::from_bytes_le(&u_bytes)
73            .into_option()
74            .ok_or(SignatureError::SerializationError)?;
75        let v = JubjubBase::from_bytes_le(&v_bytes)
76            .into_option()
77            .ok_or(SignatureError::SerializationError)?;
78        if !bool::from(is_on_curve(u, v)) {
79            return Err(SignatureError::SerializationError);
80        }
81
82        let point = JubjubSubgroup::from_raw_unchecked(u, v);
83        if !bool::from(Jubjub::from(point).is_prime_order()) {
84            return Err(SignatureError::SerializationError);
85        }
86
87        Ok(VerificationKey(point))
88    }
89}

91#[derive(Debug, Clone, PartialEq, Eq)]
92pub struct Signature {
93    pub s: JubjubScalar,
94    pub c: JubjubBase,
95}
96
97impl Signature {
98    /// Verify a signature against a verification key.
99    pub fn verify(&self, msg: &[JubjubBase], vk: &VerificationKey) -> Result<(), SignatureError> {
100        let g = JubjubSubgroup::generator();
101        let c_scalar = jubjub_base_to_scalar(self.c);
102
103        let (vk_x, vk_y) = get_coordinates(vk.0);
104        let cap_r_prime = g * self.s + vk.0 * c_scalar;
105        let (cap_r_x_prime, cap_r_y_prime) = get_coordinates(cap_r_prime);
106
107        let mut to_hash = vec![DST_SCHNORR_SIGNATURE, vk_x, vk_y, cap_r_x_prime, cap_r_y_prime];
108        to_hash.extend_from_slice(msg);
109
110        let c_prime = PoseidonHash::hash(&to_hash);
111        if c_prime != self.c {
112            return Err(SignatureError::VerificationFailed);
113        }
114
115        Ok(())
116    }
117}

122    use rand_core::OsRng;

124    /// Test signing functionality.
125    #[test]
126    fn test_signature_schnorr_verification_valid() {
127        let mut rng = OsRng;
128        let sk = SigningKey::generate(&mut rng);
129        let msg = JubjubBase::random(&mut rng);
130
131        // Sign the message
132        let signature = sk.sign(&[msg], &mut rng);
133
134        // Ensure the components of the signature are non-default values
135        assert_ne!(
136            signature.s,
137            JubjubScalar::ZERO,
138            "Schnorr signature s component should not be zero."
139        );
140        assert_ne!(
141            signature.c,
142            JubjubBase::ZERO,
143            "Schnorr signature c component should not be zero."
144        );
145
146        signature.verify(&[msg], &VerificationKey::from(&sk)).unwrap();
147    }
148
149    #[test]
150    fn test_signature_schnorr_verification_invalid_signature() {
151        let mut rng = OsRng;
152        let sk = SigningKey::generate(&mut rng);
153        let msg = JubjubBase::random(&mut rng);
154        let vk: VerificationKey = (&sk).into();
155
156        // Generate signature and tamper with it
157        let mut signature = sk.sign(&[msg], &mut rng);
158        signature.s = JubjubScalar::random(&mut rng); // Modify `s` component
159
160        // Verify the modified signature
161        let result = signature.verify(&[msg], &vk);
162        assert!(
163            result.is_err(),
164            "Invalid Schnorr signature should fail verification, but it passed."
165        );
166    }
