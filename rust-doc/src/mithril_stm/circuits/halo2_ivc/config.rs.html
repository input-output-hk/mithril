12#[derive(Debug, Clone)]
13pub struct IvcConfig {
14    pub(crate) native_config: NativeConfig,
15    pub(crate) core_decomp_config: P2RDecompositionConfig,
16    pub(crate) jubjub_config: EccConfig,
17    pub(crate) bls12_381_config: ForeignEccConfig<C>,
18    pub(crate) poseidon_config: PoseidonConfig<F>,
19    pub(crate) sha256_config: Sha256Config,
20}
21
22pub fn configure_ivc_circuit(meta: &mut ConstraintSystem<F>) -> IvcConfig {
23    let nb_advice_cols = [
24        NB_EDWARDS_COLS,
25        NB_POSEIDON_ADVICE_COLS,
26        NB_SHA256_ADVICE_COLS,
27        nb_foreign_ecc_chip_columns::<F, C, C, NG>(),
28    ]
29    .into_iter()
30    .max()
31    .unwrap_or(0);
32
33    let nb_fixed_cols = [NB_ARITH_FIXED_COLS, NB_POSEIDON_FIXED_COLS, NB_SHA256_FIXED_COLS]
34        .into_iter()
35        .max()
36        .unwrap_or(0);
37
38    let advice_columns: Vec<_> = (0..nb_advice_cols).map(|_| meta.advice_column()).collect();
39    let fixed_columns: Vec<_> = (0..nb_fixed_cols).map(|_| meta.fixed_column()).collect();
40    let committed_instance_column = meta.instance_column();
41    let instance_column = meta.instance_column();
42
43    let native_config = NativeChip::configure(
44        meta,
45        &(
46            advice_columns[..NB_ARITH_COLS].try_into().unwrap(),
47            fixed_columns[..NB_ARITH_FIXED_COLS].try_into().unwrap(),
48            [committed_instance_column, instance_column],
49        ),
50    );
51    let core_decomp_config = {
52        let pow2_config = Pow2RangeChip::configure(meta, &advice_columns[1..NB_ARITH_COLS]);
53        P2RDecompositionChip::configure(meta, &(native_config.clone(), pow2_config))
54    };
55
56    let jubjub_config =
57        EccChip::<Jubjub>::configure(meta, &advice_columns[..NB_EDWARDS_COLS].try_into().unwrap());
58
59    let base_config = FieldChip::<F, CBase, C, NG>::configure(meta, &advice_columns);
60    let bls12_381_config =
61        ForeignEccChip::<F, C, C, NG, NG>::configure(meta, &base_config, &advice_columns);
62
63    let poseidon_config = PoseidonChip::configure(
64        meta,
65        &(
66            advice_columns[..NB_POSEIDON_ADVICE_COLS].try_into().unwrap(),
67            fixed_columns[..NB_POSEIDON_FIXED_COLS].try_into().unwrap(),
68        ),
69    );
70
71    let sha256_config = Sha256Chip::configure(
72        meta,
73        &(
74            advice_columns[..NB_SHA256_ADVICE_COLS].try_into().unwrap(),
75            fixed_columns[..NB_SHA256_FIXED_COLS].try_into().unwrap(),
76        ),
77    );
78
79    IvcConfig {
80        native_config,
81        core_decomp_config,
82        jubjub_config,
83        bls12_381_config,
84        poseidon_config,
85        sha256_config,
86    }
87}
