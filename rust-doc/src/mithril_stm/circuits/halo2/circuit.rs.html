28#[derive(Clone, Default, Debug)]
29pub struct StmCircuit {
30    // k in mithril: the required number of distinct lottery indices slots needed to create a valid multi-signature
31    quorum: u32,
32    // m in mithril: the number of lotteries that a user can participate in to sign a message
33    num_lotteries: u32,
34    merkle_tree_depth: u32,
35}
36
37impl StmCircuit {
38    fn checked_len_u32(actual: usize) -> u32 {
39        u32::try_from(actual).unwrap_or(u32::MAX)
40    }
41
42    /// Validates global circuit parameters before synthesis.
43    ///
44    /// Enforces `quorum < num_lotteries` returning
45    /// `StmCircuitError::InvalidCircuitParameters` when violated.
46    pub(crate) fn validate_parameters(&self) -> StmResult<()> {
47        if self.quorum >= self.num_lotteries {
48            return Err(anyhow!(StmCircuitError::InvalidCircuitParameters {
49                quorum: self.quorum,
50                num_lotteries: self.num_lotteries,
51            }));
52        }
53
54        Ok(())
55    }
56
57    /// Validates that witness vector length matches the configured quorum.
58    ///
59    /// This precondition prevents shape mismatches; failures return
60    /// `StmCircuitError::WitnessLengthMismatch`.
61    pub(crate) fn validate_witness_length(&self, actual: usize) -> StmResult<()> {
62        let expected_quorum = self.quorum as usize;
63        if actual != expected_quorum {
64            return Err(anyhow!(StmCircuitError::WitnessLengthMismatch {
65                expected_quorum: self.quorum,
66                actual: Self::checked_len_u32(actual),
67            }));
68        }
69
70        Ok(())
71    }
72
73    /// Validates witness lottery indices against circuit constraints.
74    ///
75    /// The circuit uses 32-bit comparison constraints (`lower_than(..., 32)`), so each
76    /// index must fit in `u32` and must satisfy `index < num_lotteries`.
77    pub(crate) fn validate_lottery_index(&self, index: LotteryIndex) -> StmResult<()> {
78        let max_supported = u32::MAX as LotteryIndex;
79        if index > max_supported {
80            return Err(anyhow!(StmCircuitError::LotteryIndexTooLarge {
81                index,
82                max_supported,
83            }));
84        }
85
86        if index >= self.num_lotteries as LotteryIndex {
87            return Err(anyhow!(StmCircuitError::LotteryIndexOutOfBounds {
88                index,
89                num_lotteries: self.num_lotteries,
90            }));
91        }
92
93        Ok(())
94    }
95
96    /// Validates Merkle sibling path length against `merkle_tree_depth`.
97    ///
98    /// This guards against inconsistent witness paths and returns
99    /// `StmCircuitError::MerkleSiblingLengthMismatch` on mismatch.
100    pub(crate) fn validate_merkle_sibling_length(&self, actual: usize) -> StmResult<()> {
101        let expected_depth = self.merkle_tree_depth as usize;
102        if actual != expected_depth {
103            return Err(anyhow!(StmCircuitError::MerkleSiblingLengthMismatch {
104                expected_depth: self.merkle_tree_depth,
105                actual: Self::checked_len_u32(actual),
106            }));
107        }
108
109        Ok(())
110    }
111
112    /// Validates Merkle position bits length against `merkle_tree_depth`.
113    ///
114    /// Under the current witness shape, this cannot fail independently from sibling-length
115    /// validation because both lengths derive from `x.siblings`; returns `StmCircuitError::MerklePositionLengthMismatch`.
116    pub(crate) fn validate_merkle_position_length(&self, actual: usize) -> StmResult<()> {
117        let expected_depth = self.merkle_tree_depth as usize;
118        if actual != expected_depth {
119            return Err(anyhow!(StmCircuitError::MerklePositionLengthMismatch {
120                expected_depth: self.merkle_tree_depth,
121                actual: Self::checked_len_u32(actual),
122            }));
123        }
124
125        Ok(())
126    }
127
128    /// Constructs a new `StmCircuit` from Mithril `Parameters`.  
129    ///  
130    /// This constructor only validates that the quorum `k` and the number of lotteries `m`  
131    /// fit into a `u32` by performing fallible `u64 -> u32` conversions. If either value  
132    /// exceeds `u32::MAX`, it returns an error.  
133    ///  
134    /// It does **not** enforce semantic constraints such as `k < m` or other parameter  
135    /// relationships; those invariants are validated later during circuit synthesis via  
136    /// `validate_parameters`.
137    pub fn try_new(stm_params: &Parameters, merkle_tree_depth: u32) -> StmResult<Self> {
138        Ok(Self {
139            quorum: stm_params
140                .k
141                .try_into()
142                .with_context(|| format!(
143                    "Failed to cast quorum as a u32. Its value ({}) is too large for the circuit.", 
144                    stm_params.k)
145                )?,
146            num_lotteries: stm_params
147                .m
148                .try_into()
149                .with_context(|| format!(
150                    "Failed to cast number of lotteries as a u32. Its value ({}) is too large for the circuit.", 
151                    stm_params.m),
152                )?,
153            merkle_tree_depth,
154        })
155    }
156}

158impl Relation for StmCircuit {
159    type Instance = (MerkleRoot, SignedMessageWithoutPrefix);
160    type Witness = Vec<(MTLeaf, MerklePath, UniqueSchnorrSignature, LotteryIndex)>;
161
162    fn format_instance(instance: &Self::Instance) -> Result<Vec<CircuitBase>, Error> {
163        Ok(vec![instance.0.into(), instance.1.into()])
164    }
165
166    fn circuit(
167        &self,
168        std_lib: &ZkStdLib,
169        layouter: &mut impl Layouter<CircuitBase>,
170        instance: Value<Self::Instance>,
171        witness: Value<Self::Witness>,
172    ) -> Result<(), Error> {
173        self.validate_parameters().map_err(to_synthesis_error)?;
174        let witness = witness
175            .map_with_result(|witness| -> StmResult<_> {
176                self.validate_witness_length(witness.len())?;
177                witness
178                    .iter()
179                    .try_for_each(|(_, _, _, index)| self.validate_lottery_index(*index))?;
180                Ok(witness)
181            })
182            .map_err(to_synthesis_error)?
183            .transpose_vec(self.quorum as usize);
184
185        let merkle_root: AssignedNative<CircuitBase> =
186            std_lib.assign_as_public_input(layouter, instance.map(|(x, _)| x.into()))?;
187        let msg: AssignedNative<CircuitBase> =
188            std_lib.assign_as_public_input(layouter, instance.map(|(_, x)| x.into()))?;
189
190        // Compute H_1(merkle_root, msg)
191        let hash = std_lib.hash_to_curve(layouter, &[merkle_root.clone(), msg.clone()])?;
192
193        let generator: AssignedNativePoint<CircuitCurve> = std_lib.jubjub().assign_fixed(
194            layouter,
195            <CircuitCurve as CircuitCurveTrait>::CryptographicGroup::generator(),
196        )?;
197
198        let domain_separation_tag_signature: AssignedNative<_> =
199            std_lib.assign_fixed(layouter, CircuitBase::from(DOMAIN_SEPARATION_TAG_SIGNATURE))?;
200        let domain_separation_tag_lottery: AssignedNative<_> =
201            std_lib.assign_fixed(layouter, CircuitBase::from(DOMAIN_SEPARATION_TAG_LOTTERY))?;
202        let lottery_prefix = std_lib.poseidon(
203            layouter,
204            &[
205                domain_separation_tag_lottery.clone(),
206                merkle_root.clone(),
207                msg.clone(),
208            ],
209        )?;
210
211        let mut pre_index: AssignedNative<_> =
212            std_lib.assign(layouter, Value::known(CircuitBase::ZERO))?;
213        for (i, wit) in witness.into_iter().enumerate() {
214            let index: AssignedNative<CircuitBase> = std_lib.assign(
215                layouter,
216                wit.clone().map(|(_, _, _, i)| CircuitBase::from(i)),
217            )?;
218
219            // Check index order
220            if i > 0 {
221                let is_less = std_lib.lower_than(layouter, &pre_index, &index, 32)?;
222                std_lib.assert_true(layouter, &is_less)?;
223            }
224
225            pre_index = index.clone();
226
227            let vk = std_lib
228                .jubjub()
229                .assign(layouter, wit.clone().map(|(x, _, _, _)| x.0.0.0))?;
230
231            let target: AssignedNative<CircuitBase> =
232                std_lib.assign(layouter, wit.clone().map(|(x, _, _, _)| x.1.into()))?;
233
234            // Assign sibling Values.
235            let assigned_merkle_siblings = std_lib.assign_many(
236                layouter,
237                wit.clone()
238                    .map_with_result(|(_, x, _, _)| -> StmResult<_> {
239                        self.validate_merkle_sibling_length(x.siblings.len())?;
240                        Ok(x.siblings.iter().map(|sibling| sibling.1.into()).collect::<Vec<_>>())
241                    })
242                    .map_err(to_synthesis_error)?
243                    .transpose_vec(self.merkle_tree_depth as usize)
244                    .as_slice(),
245            )?;
246
247            // Assign sibling Position.
248            let assigned_merkle_positions = std_lib.assign_many(
249                layouter,
250                wit.clone()
251                    .map_with_result(|(_, x, _, _)| -> StmResult<_> {
252                        self.validate_merkle_position_length(x.siblings.len())?;
253                        Ok(x.siblings
254                            .iter()
255                            .map(|sibling| CircuitBase::from(sibling.0))
256                            .collect::<Vec<_>>())
257                    })
258                    .map_err(to_synthesis_error)?
259                    .transpose_vec(self.merkle_tree_depth as usize)
260                    .as_slice(),
261            )?;
262
263            // Assert merkle positions are binary values.
264            let assigned_merkle_positions = assigned_merkle_positions
265                .iter()
266                .map(|pos| std_lib.convert(layouter, pos))
267                .collect::<Result<Vec<AssignedBit<CircuitBase>>, Error>>()?;
268
269            let sigma_value = wit
270                .clone()
271                .map_with_result(|(_, _, sig, _)| {
272                    let (u, v) = sig.commitment_point.get_coordinates();
273                    PrimeOrderProjectivePoint::from_coordinates(u, v).map(|point| point.0)
274                })
275                .map_err(to_synthesis_error)?;
276            let sigma: AssignedNativePoint<_> = std_lib.jubjub().assign(layouter, sigma_value)?;
277            let s: AssignedScalarOfNativeCurve<CircuitCurve> = std_lib
278                .jubjub()
279                .assign(layouter, wit.clone().map(|(_, _, sig, _)| sig.response.0))?;
280            let c_native = std_lib.assign(
281                layouter,
282                wit.map(|(_, _, sig, _)| CircuitBase::from(sig.challenge)),
283            )?;
284            let c: AssignedScalarOfNativeCurve<CircuitCurve> =
285                std_lib.jubjub().convert(layouter, &c_native)?;
286
287            verify_merkle_path(
288                std_lib,
289                layouter,
290                &vk,
291                &target,
292                &merkle_root,
293                &assigned_merkle_siblings,
294                &assigned_merkle_positions,
295            )?;
296
297            verify_unique_signature(
298                std_lib,
299                layouter,
300                &domain_separation_tag_signature,
301                &generator,
302                &vk,
303                &s,
304                &c,
305                &c_native,
306                &hash,
307                &sigma,
308            )?;
309
310            verify_lottery(std_lib, layouter, &lottery_prefix, &sigma, &index, &target)?;
311        }
312
313        // m can be put as a public instance or a constant
314        let m = std_lib.assign_fixed(layouter, CircuitBase::from(self.num_lotteries as u64))?;
315        let is_less = std_lib.lower_than(layouter, &pre_index, &m, 32)?;
316
317        std_lib.assert_true(layouter, &is_less)
318    }

321        ZkStdLibArch {
322            jubjub: true,
323            poseidon: true,
324            sha2_256: false,

330            base64: false,
331            nr_pow2range_cols: 2,
332            automaton: false,
333            blake2b: false,
334        }

336
337    fn write_relation<W: std::io::Write>(&self, writer: &mut W) -> std::io::Result<()> {
338        writer.write_all(&self.quorum.to_le_bytes())?;
339        writer.write_all(&self.num_lotteries.to_le_bytes())?;
340        writer.write_all(&self.merkle_tree_depth.to_le_bytes())
341    }
342
343    fn read_relation<R: std::io::Read>(reader: &mut R) -> std::io::Result<Self> {
344        // Buffers to read 4 bytes for each `u32` field.
345        let mut quorum_bytes = [0u8; 4];
346        let mut num_lotteries_bytes = [0u8; 4];
347        let mut merkle_tree_depth_bytes = [0u8; 4];
348
349        // Read the values into their corresponding buffers.
350        reader.read_exact(&mut quorum_bytes)?;
351        reader.read_exact(&mut num_lotteries_bytes)?;
352        reader.read_exact(&mut merkle_tree_depth_bytes)?;
353
354        // Convert the byte arrays back into `u32` values.
355        let quorum = u32::from_le_bytes(quorum_bytes);
356        let num_lotteries = u32::from_le_bytes(num_lotteries_bytes);
357        let merkle_tree_depth = u32::from_le_bytes(merkle_tree_depth_bytes);
358
359        // Construct and return the `StmCircuit` instance.
360        Ok(Self {
361            quorum,
362            num_lotteries,
363            merkle_tree_depth,
364        })
365    }

368#[cfg(test)]
369mod dst_alignment_tests {
370    use crate::circuits::halo2::types::CircuitBase;
371    use crate::signature_scheme::{
372        BaseFieldElement, DOMAIN_SEPARATION_TAG_LOTTERY, DOMAIN_SEPARATION_TAG_SIGNATURE,
373        compute_poseidon_digest,
374    };
375
376    use midnight_circuits::{hash::poseidon::PoseidonChip, instructions::hash::HashCPU};
377
378    const REFERENCE_SIGNATURE_DOMAIN_TAG: BaseFieldElement =
379        BaseFieldElement(CircuitBase::from_raw([0x5349_474E_5F44_5354, 0, 0, 0]));
380    const REFERENCE_LOTTERY_DOMAIN_TAG: BaseFieldElement =
381        BaseFieldElement(CircuitBase::from_raw([0x4C4F_5454_5F44_5354, 0, 0, 0]));
382
383    #[test]
384    fn signature_and_lottery_domain_tags_do_not_collide() {
385        assert_ne!(
386            DOMAIN_SEPARATION_TAG_SIGNATURE, DOMAIN_SEPARATION_TAG_LOTTERY,
387            "signature and lottery domain separation tags must be distinct"
388        );
389    }
390
391    #[test]
392    fn signature_digest_matches_reference_signature_domain_tag_formula() {
393        let signature_transcript_inputs = vec![
394            BaseFieldElement::from(11),
395            BaseFieldElement::from(22),
396            BaseFieldElement::from(33),
397            BaseFieldElement::from(44),
398        ];
399
400        let mut stm_inputs = vec![DOMAIN_SEPARATION_TAG_SIGNATURE];
401        stm_inputs.extend_from_slice(&signature_transcript_inputs);
402        let signature_digest_via_stm = compute_poseidon_digest(&stm_inputs);
403
404        let mut signature_digest_manual_inputs =
405            vec![CircuitBase::from(REFERENCE_SIGNATURE_DOMAIN_TAG)];
406        signature_digest_manual_inputs.extend(
407            signature_transcript_inputs
408                .iter()
409                .map(|value| CircuitBase::from(*value)),
410        );
411
412        let signature_digest_via_reference_formula = BaseFieldElement(
413            PoseidonChip::<CircuitBase>::hash(&signature_digest_manual_inputs),
414        );
415
416        assert_eq!(
417            signature_digest_via_stm, signature_digest_via_reference_formula,
418            "signature digest computed via STM helper must match reference signature-domain formula"
419        );
420    }
421
422    #[test]
423    fn lottery_prefix_matches_reference_lottery_domain_tag_formula() {
424        let merkle_root = CircuitBase::from(123u64);
425        let msg = CircuitBase::from(456u64);
426
427        let lottery_prefix_via_stm_constant = PoseidonChip::<CircuitBase>::hash(&[
428            CircuitBase::from(DOMAIN_SEPARATION_TAG_LOTTERY),
429            merkle_root,
430            msg,
431        ]);
432
433        let lottery_prefix_via_reference_formula = PoseidonChip::<CircuitBase>::hash(&[
434            CircuitBase::from(REFERENCE_LOTTERY_DOMAIN_TAG),
435            merkle_root,
436            msg,
437        ]);
438
439        assert_eq!(
440            BaseFieldElement(lottery_prefix_via_stm_constant),
441            BaseFieldElement(lottery_prefix_via_reference_formula),
442            "lottery prefix must use the lottery domain separation tag as first Poseidon input"
443        );
444    }
