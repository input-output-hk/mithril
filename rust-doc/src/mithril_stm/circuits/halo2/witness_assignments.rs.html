21/// Assigned Merkle authentication path values consumed by the Merkle path gadget.
22pub(crate) struct AssignedMerklePath {

24    pub(crate) siblings: Vec<AssignedNative<CircuitBase>>,

29/// Assigned witness values consumed by the Merkle, lottery, and signature gadgets.
30pub(crate) struct AssignedWitnessEntry {
31    /// Assigned verification key point for the current witness leaf.
32    pub(crate) verification_key: AssignedNativePoint<CircuitCurve>,
33    /// Assigned lottery target value committed in the witness leaf.
34    pub(crate) lottery_target_value: AssignedNative<CircuitBase>,
35    /// Assigned Merkle authentication path for the current witness leaf.
36    pub(crate) merkle_path: AssignedMerklePath,
37    /// Assigned lottery index associated with the current signature witness.
38    pub(crate) lottery_index: AssignedNative<CircuitBase>,
39}

53/// Assigns one witness entry into the Halo2 value layer consumed by circuit gadgets.
54pub(crate) fn assign_witness_entry(
55    circuit: &StmCircuit,
56    std_lib: &ZkStdLib,
57    layouter: &mut impl Layouter<CircuitBase>,
58    witness_entry: Value<CircuitWitnessEntry>,
59) -> Result<AssignedWitnessEntry, Error> {
60    let verification_key = assign_verification_key(std_lib, layouter, witness_entry.clone())?;
61
62    let lottery_target_value = std_lib.assign(
63        layouter,
64        witness_entry
65            .clone()
66            .map(|entry| entry.leaf.lottery_target_value().into()),
67    )?;
68    let merkle_path = assign_merkle_path(circuit, std_lib, layouter, witness_entry.clone())?;
69
70    let lottery_index = std_lib.assign(
71        layouter,
72        witness_entry.map(|entry| CircuitBase::from(entry.lottery_index)),
73    )?;
74
75    Ok(AssignedWitnessEntry {
76        verification_key,
77        lottery_target_value,
78        merkle_path,
79        lottery_index,
80    })
81}

83/// Assigns the witness verification key as a Jubjub point usable by downstream gadgets.
84fn assign_verification_key(
85    std_lib: &ZkStdLib,
86    layouter: &mut impl Layouter<CircuitBase>,
87    witness_entry: Value<CircuitWitnessEntry>,
88) -> Result<AssignedNativePoint<CircuitCurve>, Error> {
89    std_lib.jubjub().assign(
90        layouter,
91        witness_entry.map(|entry| entry.leaf.verification_key_point().0),
92    )
93}

95/// Assigns and validates the Merkle authentication path carried by one witness entry.
96fn assign_merkle_path(
97    circuit: &StmCircuit,
98    std_lib: &ZkStdLib,
99    layouter: &mut impl Layouter<CircuitBase>,
100    witness_entry: Value<CircuitWitnessEntry>,
101) -> Result<AssignedMerklePath, Error> {
102    let siblings = std_lib.assign_many(
103        layouter,
104        witness_entry
105            .clone()
106            .map_with_result(|entry| -> StmResult<_> {
107                let merkle_path = entry.merkle_path;
108                circuit.validate_merkle_sibling_length(merkle_path.siblings.len())?;
109                Ok(merkle_path
110                    .siblings
111                    .iter()
112                    .map(|sibling| sibling.1.into())
113                    .collect::<Vec<_>>())
114            })
115            .map_err(to_synthesis_error)?
116            .transpose_vec(circuit.merkle_tree_depth() as usize)
117            .as_slice(),
118    )?;
119
120    let positions = std_lib.assign_many(
121        layouter,
122        witness_entry
123            .clone()
124            .map_with_result(|entry| -> StmResult<_> {
125                let merkle_path = entry.merkle_path;
126                circuit.validate_merkle_position_length(merkle_path.siblings.len())?;
127                Ok(merkle_path
128                    .siblings
129                    .iter()
130                    .map(|sibling| CircuitBase::from(sibling.0))
131                    .collect::<Vec<_>>())
132            })
133            .map_err(to_synthesis_error)?
134            .transpose_vec(circuit.merkle_tree_depth() as usize)
135            .as_slice(),
136    )?;
137
138    let positions = positions
139        .iter()
140        .map(|position| std_lib.convert(layouter, position))
141        .collect::<Result<Vec<_>, Error>>()?;
142
143    Ok(AssignedMerklePath {
144        siblings,
145        positions,
146    })
147}

149/// Assigns unique Schnorr signature components extracted from one circuit witness entry.
150pub(crate) fn assign_signature_components(
151    std_lib: &ZkStdLib,
152    layouter: &mut impl Layouter<CircuitBase>,
153    witness_entry: Value<CircuitWitnessEntry>,
154) -> Result<AssignedSignatureComponents, Error> {
155    let commitment_point_value = witness_entry
156        .clone()
157        .map_with_result(|entry| {
158            let signature = entry.unique_schnorr_signature;
159            let (u, v) = signature.commitment_point.get_coordinates();
160            PrimeOrderProjectivePoint::from_coordinates(u, v).map(|point| point.0)
161        })
162        .map_err(to_synthesis_error)?;
163    let commitment_point = std_lib.jubjub().assign(layouter, commitment_point_value)?;
164    let response = std_lib.jubjub().assign(
165        layouter,
166        witness_entry
167            .clone()
168            .map(|entry| entry.unique_schnorr_signature.response.0),
169    )?;
170    let challenge_in_base_field = std_lib.assign(
171        layouter,
172        witness_entry.map(|entry| CircuitBase::from(entry.unique_schnorr_signature.challenge)),
173    )?;
174    let challenge_as_scalar = std_lib.jubjub().convert(layouter, &challenge_in_base_field)?;
175
176    Ok(AssignedSignatureComponents {
177        commitment_point,
178        response,
179        challenge_in_base_field,
180        challenge_as_scalar,
181    })
