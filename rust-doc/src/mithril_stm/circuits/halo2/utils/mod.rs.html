6/// Errors returned by Halo2 utility conversion helpers.
7#[derive(Debug, Error)]
8pub(crate) enum Halo2UtilsError {
9    /// Parsing the field modulus constant failed.
10    #[error("failed to parse prime field modulus from hex")]
11    FieldModulusParse,
12    /// Converting a reduced integer to a field element failed.
13    #[error("failed to convert reduced integer to field element")]
14    FieldElementConversion,
15}
16
17/// Splits a field element into `(lower, upper)` limbs at `num_bits` using LE encoding.
18///
19/// The input is interpreted as a little-endian integer before bit slicing.
20pub(crate) fn split_field_element_into_le_limbs<F: PrimeField>(
21    value: &F,
22    num_bits: u32,
23) -> Result<(F, F), Halo2UtilsError> {
24    let value_big = BigUint::from_bytes_le(value.to_repr().as_ref());
25    let lower_mask = (BigUint::one() << num_bits) - BigUint::one();
26    let lower_big = value_big.clone() & &lower_mask;
27    let upper_big = value_big >> num_bits;
28    let lower = big_unsigned_integer_to_field_element::<F>(lower_big)?;
29    let upper = big_unsigned_integer_to_field_element::<F>(upper_big)?;
30    Ok((lower, upper))
31}
32
33/// Parses the prime field modulus (`F::MODULUS`) into a `BigUint`.
34fn field_modulus_as_biguint<F: PrimeField>() -> Result<BigUint, Halo2UtilsError> {
35    BigUint::from_str_radix(&F::MODULUS[2..], 16).map_err(|_| Halo2UtilsError::FieldModulusParse)
36}
37
38/// Reduces an unsigned integer modulo `p` and converts it into the field element type.
39///
40/// Conversion uses `from_str_vartime` on the reduced decimal representation.
41fn big_unsigned_integer_to_field_element<F: PrimeField>(e: BigUint) -> Result<F, Halo2UtilsError> {
42    let modulus = field_modulus_as_biguint::<F>()?;
43    let e = e % modulus;
44    F::from_str_vartime(&e.to_str_radix(10)[..]).ok_or(Halo2UtilsError::FieldElementConversion)
45}

50#[cfg(test)]
51mod merkle_path_adapter {
52    use crate::circuits::halo2::types::{MerklePath as Halo2MerklePath, Position};
53    use crate::membership_commitment::MerklePath as StmMerklePath;
54    use crate::signature_scheme::BaseFieldElement;
55    use digest::Digest;
56    use thiserror::Error;
57
58    #[derive(Debug, Error)]
59    pub enum MerklePathAdapterError {
60        #[error("invalid merkle digest length")]
61        InvalidDigestLength,
62        #[error("non-canonical merkle digest")]
63        NonCanonicalDigest,
64    }
65
66    impl<D: Digest> TryFrom<&StmMerklePath<D>> for Halo2MerklePath {
67        type Error = MerklePathAdapterError;
68
69        fn try_from(stm_path: &StmMerklePath<D>) -> Result<Self, Self::Error> {
70            let mut siblings = Vec::with_capacity(stm_path.values.len());
71
72            for (i, value) in stm_path.values.iter().enumerate() {
73                let bytes: [u8; 32] = value
74                    .as_slice()
75                    .try_into()
76                    .map_err(|_| MerklePathAdapterError::InvalidDigestLength)?;
77                let node = BaseFieldElement::from_bytes(&bytes)
78                    .ok()
79                    .map(|base| base.0)
80                    .ok_or(MerklePathAdapterError::NonCanonicalDigest)?;
81                let bit = (stm_path.index >> i) & 1;
82                // At level `i`, `bit = (index >> i) & 1`: `0` means current is left, `1` means right.
83                // STM uses `H(current || sibling)` for `bit == 0`, else `H(sibling || current)`.
84                // Map `0 -> Position::Right` and `1 -> Position::Left` so Halo2 folds identically.
85                let position = if bit == 0 {
86                    Position::Right
87                } else {
88                    Position::Left
89                };
90                siblings.push((position, node));
91            }
92
93            Ok(Halo2MerklePath::new(siblings))
94        }
95    }
