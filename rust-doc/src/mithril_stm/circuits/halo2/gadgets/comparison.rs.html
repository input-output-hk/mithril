10/// Compares two assigned 255-bit values and returns a bit witnessing whether `x < y`.
11pub(super) fn lower_than_native(
12    std_lib: &ZkStdLib,
13    layouter: &mut impl Layouter<CircuitBase>,
14    x: &AssignedNative<CircuitBase>,
15    y: &AssignedNative<CircuitBase>,
16) -> Result<AssignedBit<CircuitBase>, Error> {
17    let (x_low_assigned, x_high_assigned) = decompose_unsafe(std_lib, layouter, x)?;
18    let (y_low_assigned, y_high_assigned) = decompose_unsafe(std_lib, layouter, y)?;
19
20    let is_equal_high = std_lib.is_equal(layouter, &x_high_assigned, &y_high_assigned)?;
21    let is_less_low = std_lib.lower_than(layouter, &x_low_assigned, &y_low_assigned, 127)?;
22    let is_less_high = std_lib.lower_than(layouter, &x_high_assigned, &y_high_assigned, 128)?;
23
24    let low_less = std_lib.and(layouter, &[is_equal_high, is_less_low])?;
25    std_lib.or(layouter, &[is_less_high, low_less])
26}
