1//! Adapters from STM-side structures into the Halo2 circuit witness contract.
2//!
3//! Integrators should treat `witness.rs` as the circuit-facing boundary and use this module when
4//! STM data needs explicit conversion into that shape.
5
6use digest::Digest;
7use thiserror::Error;
8
9use crate::circuits::halo2::witness::{MerklePath as Halo2MerklePath, Position};
10use crate::membership_commitment::MerklePath as StmMerklePath;
11use crate::signature_scheme::BaseFieldElement;
12
13/// Errors returned when adapting STM Merkle paths to Halo2 witness paths.
14#[derive(Debug, Error)]
15pub enum MerklePathAdapterError {
16    /// A Merkle digest did not contain the expected 32-byte field encoding.
17    #[error("invalid merkle digest length")]
18    InvalidDigestLength,
19    /// A 32-byte digest was not a canonical circuit base-field element encoding.
20    #[error("non-canonical merkle digest")]
21    NonCanonicalDigest,
22}
23
24impl<D: Digest> TryFrom<&StmMerklePath<D>> for Halo2MerklePath {
25    type Error = MerklePathAdapterError;
26
27    fn try_from(stm_path: &StmMerklePath<D>) -> Result<Self, Self::Error> {
28        let mut siblings = Vec::with_capacity(stm_path.values.len());
29
30        for (i, value) in stm_path.values.iter().enumerate() {
31            let bytes: [u8; 32] = value
32                .as_slice()
33                .try_into()
34                .map_err(|_| MerklePathAdapterError::InvalidDigestLength)?;
35            let node = BaseFieldElement::from_bytes(&bytes)
36                .ok()
37                .map(|base| base.into())
38                .ok_or(MerklePathAdapterError::NonCanonicalDigest)?;
39            let bit = (stm_path.index >> i) & 1;
40            // At level `i`, `bit = (index >> i) & 1`: `0` means current is left, `1` means right.
41            // STM uses `H(current || sibling)` for `bit == 0`, else `H(sibling || current)`.
42            // Map `0 -> Position::Right` and `1 -> Position::Left` so Halo2 folds identically.
43            let position = if bit == 0 {
44                Position::Right
45            } else {
46                Position::Left
47            };
48            siblings.push((position, node));
49        }
50
51        Ok(Halo2MerklePath::new(siblings))
52    }
53}

55#[cfg(test)]
56mod tests {
57    use super::*;
58    use crate::circuits::halo2::witness::Position;
59    use crate::hash::poseidon::MidnightPoseidonDigest;
60    use crate::membership_commitment::MerklePath as StmMerklePath;
61    use crate::signature_scheme::BaseFieldElement;
62
63    #[test]
64    fn converts_valid_merkle_path_and_preserves_position_mapping() {
65        let left_node = BaseFieldElement::from(5u64).to_bytes().to_vec();
66        let right_node = BaseFieldElement::from(9u64).to_bytes().to_vec();
67        let stm_path = StmMerklePath::<MidnightPoseidonDigest>::new(vec![left_node, right_node], 1);
68
69        let halo2_path = Halo2MerklePath::try_from(&stm_path)
70            .expect("valid STM merkle path should convert successfully");
71
72        assert_eq!(halo2_path.siblings.len(), 2);
73        assert_eq!(halo2_path.siblings[0].0, Position::Left);
74        assert_eq!(halo2_path.siblings[1].0, Position::Right);
75        assert_eq!(halo2_path.siblings[0].1, 5u64.into());
76        assert_eq!(halo2_path.siblings[1].1, 9u64.into());
77    }
78
79    #[test]
80    fn rejects_invalid_digest_length() {
81        let stm_path = StmMerklePath::<MidnightPoseidonDigest>::new(vec![vec![0u8; 31]], 0);
82
83        let error = Halo2MerklePath::try_from(&stm_path)
84            .expect_err("invalid digest length should be rejected");
85
86        assert!(matches!(error, MerklePathAdapterError::InvalidDigestLength));
87    }
88
89    #[test]
90    fn rejects_non_canonical_digest() {
91        let stm_path = StmMerklePath::<MidnightPoseidonDigest>::new(vec![vec![0xff; 32]], 0);
92
93        let error = Halo2MerklePath::try_from(&stm_path)
94            .expect_err("non-canonical digest should be rejected");
95
96        assert!(matches!(error, MerklePathAdapterError::NonCanonicalDigest));
97    }
