1//! Circuit witness and instance boundary types for the Halo2 STM circuit.
2//!
3//! This module defines the exact public/private shapes consumed by the relation, while
4//! keeping them separate from lower-level circuit primitives in `types.rs`.
5//! Integrators should treat these types as the circuit-facing contract.
6

16/// Lottery target value used by the circuit for signer eligibility checks.
17pub type LotteryTargetValue = CircuitBaseField;
18/// Signed message value used by the circuit transcript, without any domain prefix.
19pub type SignedMessageWithoutPrefix = CircuitBaseField;
20/// Merkle root public input committed by the STM membership commitment tree.
21pub type MerkleRoot = CircuitBaseField;
22/// Circuit statement/instance type, representing the public inputs to the STM SNARK circuit.
23pub(crate) type CircuitInstance = (MerkleRoot, SignedMessageWithoutPrefix);

25/// Merkle-tree leaf material used by Halo2 witness construction.
26///
27/// The first field stores the signer's verification key, and the second
28/// field stores the lottery target value associated with that signer.
29#[derive(Debug, Copy, Clone, PartialEq, Eq)]
30pub struct CircuitMerkleTreeLeaf(pub SchnorrVerificationKey, pub LotteryTargetValue);
31
32impl CircuitMerkleTreeLeaf {
33    /// Returns the signer's verification key stored in this witness leaf.
34    pub fn verification_key(&self) -> SchnorrVerificationKey {
35        self.0
36    }
37
38    /// Returns the curve point backing the signer's verification key.
39    pub fn verification_key_point(&self) -> PrimeOrderProjectivePoint {
40        self.0.0
41    }
42
43    /// Returns the lottery target value associated with this witness leaf.
44    pub fn lottery_target_value(&self) -> LotteryTargetValue {
45        self.1
46    }
47}

49/// Position of a sibling node relative to the current hash in a Merkle path.
50#[derive(Clone, Copy, Debug, PartialEq, Eq)]
51pub enum Position {
52    /// The sibling node is on the left side of the current hash.
53    Left,
54    /// The sibling node is on the right side of the current hash.
55    Right,
56}
57
58impl From<Position> for CircuitBase {
59    fn from(position: Position) -> Self {
60        match position {
61            Position::Left => Self::ZERO,
62            Position::Right => Self::ONE,
63        }
64    }
65}
66
67impl From<Position> for CircuitBaseField {
68    fn from(position: Position) -> Self {
69        CircuitBase::from(position).into()
70    }
71}

73/// Merkle authentication path used by the Halo2 circuit witness.
74///
75/// Each entry stores sibling position and sibling hash value for one tree level.
76#[derive(Clone, Debug)]
77pub struct MerklePath {
78    /// Ordered list of `(position, sibling_hash)` from leaf level to root level.
79    pub siblings: Vec<(Position, CircuitBaseField)>,
80}
81
82impl MerklePath {
83    /// Creates a new Merkle path from ordered sibling entries.
84    pub fn new(siblings: Vec<(Position, CircuitBaseField)>) -> Self {
85        Self { siblings }
86    }
87}

89/// Single circuit witness entry consumed by the Halo2 relation.
90#[derive(Clone, Debug)]
91pub struct CircuitWitnessEntry {
92    /// Merkle leaf material committed for the current signer.
93    pub(crate) leaf: CircuitMerkleTreeLeaf,
94    /// Merkle authentication path opening `leaf` to the public root.
95    pub(crate) merkle_path: MerklePath,
96    /// Unique Schnorr signature carried by this witness entry.
97    pub(crate) unique_schnorr_signature: UniqueSchnorrSignature,
98    /// Lottery slot index claimed by this witness entry.
99    pub(crate) lottery_index: LotteryIndex,
100}
101
102/// Full circuit witness consumed by the Halo2 relation.
103pub(crate) type CircuitWitness = Vec<CircuitWitnessEntry>;
