1use thiserror::Error;

3/// Circuit-scoped errors for Halo2 STM validation and execution.
4#[cfg_attr(not(test), allow(dead_code))]
5#[derive(Debug, Error, Clone, PartialEq, Eq)]
6pub enum StmCircuitError {
7    /// Invalid relation parameters: quorum must be strictly lower than number of lotteries.
8    #[error(
9        "Circuit::validate_parameters failed: quorum ({quorum}) must be lower than num_lotteries ({num_lotteries})"
10    )]
11    InvalidCircuitParameters { quorum: u32, num_lotteries: u32 },
12
13    /// Witness vector length does not match the configured quorum.
14    #[error(
15        "Circuit::validate_witness_length failed: expected quorum {expected_quorum}, got {actual}"
16    )]
17    WitnessLengthMismatch { expected_quorum: u32, actual: u32 },
18
19    /// Merkle sibling path length does not match the configured Merkle depth.
20    #[error(
21        "Circuit::validate_merkle_sibling_length failed: expected depth {expected_depth}, got {actual}"
22    )]
23    MerkleSiblingLengthMismatch { expected_depth: u32, actual: u32 },
24
25    /// Merkle position path length does not match the configured Merkle depth.
26    #[error(
27        "Circuit::validate_merkle_position_length failed: expected depth {expected_depth}, got {actual}"
28    )]
29    MerklePositionLengthMismatch { expected_depth: u32, actual: u32 },
30
31    /// Merkle tree depth does not fit fixture sizing constraints.
32    #[error("Invalid merkle tree depth ({depth})")]
33    InvalidMerkleTreeDepth { depth: u32 },
34
35    /// Selected leaf index is out of bounds for the current tree.
36    #[error("Invalid selected leaf index ({index}) for {num_leaves} leaves")]
37    InvalidSelectedLeafIndex { index: u32, num_leaves: u32 },
38
39    /// Empty lottery indices were provided where at least one is required.
40    #[error("Empty indices")]
41    EmptyIndices,
42
43    /// Witness must contain at least two entries.
44    #[error("Witness too short (got {actual}, expected at least 2)")]
45    WitnessTooShort { actual: u32 },
46
47    /// No distinct witness entries were found.
48    #[error("No distinct witness entries")]
49    NoDistinctWitnessEntries,
50
51    /// Tried to build a witness from an empty signer set.
52    #[error("Empty signer leaves")]
53    EmptySignerLeaves,
54
55    /// Signer leaf index is out of bounds.
56    #[error("Invalid signer leaf index ({index}) for {num_signers} signers")]
57    InvalidSignerFixtureIndex { index: u32, num_signers: u32 },
58
59    /// Failed to decode lottery target from field bytes.
60    #[error("Invalid lottery target bytes")]
61    InvalidLotteryTargetBytes,
62
63    /// Failed to decode challenge bytes into a base field element.
64    #[error("Invalid challenge bytes")]
65    InvalidChallengeBytes,
66
67    /// Challenge bytes decode but do not match native challenge value.
68    #[error("Challenge endianness mismatch")]
69    ChallengeEndiannessMismatch,
70
71    /// Merkle root digest has an invalid byte length.
72    #[error("Invalid merkle root digest length ({actual})")]
73    InvalidMerkleRootDigestLength { actual: u32 },
74
75    /// Merkle root digest is not a canonical base field element encoding.
76    #[error("Non-canonical merkle root digest")]
77    NonCanonicalMerkleRootDigest,
78
79    /// STM Merkle-path verification failed for selected leaf.
80    #[error("Merkle path verification failed")]
81    MerklePathVerificationFailed,
82
83    /// Failed to create the local assets directory for persisted circuit params.
84    #[error("Failed to create params assets directory")]
85    ParamsAssetsDirCreate,
86
87    /// In-memory circuit key cache lock is poisoned.
88    #[error("Circuit keys cache lock poisoned ({operation})")]
89    CircuitKeysCacheLockPoisoned { operation: &'static str },
90
91    /// Signature generation failed while preparing witness.
92    #[error("Signature generation failed")]
93    SignatureGenerationFailed,
94
95    /// Signature verification failed while preparing witness.
96    #[error("Signature verification failed")]
97    SignatureVerificationFailed,
98
99    /// Proof was generated but rejected by the verifier.
100    #[error("Proof verification rejected")]
101    VerificationRejected,
102}
