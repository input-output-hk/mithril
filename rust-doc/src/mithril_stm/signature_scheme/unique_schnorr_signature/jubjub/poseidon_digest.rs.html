6/// Domain Separation Tag (DST) for the Poseidon hash used in signature contexts.
7pub(crate) const DST_SIGNATURE: JubjubBase = JubjubBase::from_raw([
8    0x5349_474E_5F44_5354, // "SIGN_DST" (ASCII), little-endian u64

15/// Domain Separation Tag (DST) for the lottery check. It is used as a prefix when computing
16/// the eligibility value of a signature.
17// TODO: remove this allow dead_code directive when function is called or future_snark is activated
18#[allow(dead_code)]
19pub const DOMAIN_SEPARATION_TAG_LOTTERY: BaseFieldElement =
20    BaseFieldElement(JubjubBase::from_raw([3, 3, 0, 0]));
21
22/// Computes a Poseidon digest over the provided base field elements
23/// Returns a scalar field element as the digest
24pub(crate) fn compute_poseidon_digest(input: &[BaseFieldElement]) -> BaseFieldElement {
25    let mut poseidon_input = vec![DST_SIGNATURE];
26    poseidon_input.extend(input.iter().map(|i| i.0).collect::<Vec<JubjubBase>>());
27
28    BaseFieldElement(PoseidonChip::<JubjubBase>::hash(&poseidon_input))
29}
