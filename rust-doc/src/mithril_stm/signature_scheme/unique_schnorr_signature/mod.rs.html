1//! Unique Schnorr Signature module
2//!
3//! This module implements a variant of the Schnorr signature algorithm.
4//! Specifically, it extends the classic scheme by appending a deterministic
5//! value derived solely from the message and the signing key. This ungrindable
6//! value produces a unique, reproducible identification tag for each signature,
7//! which can be leveraged in lottery-based schemes such as Mithril multi-signatures.
8
9mod error;
10mod jubjub;
11mod signature;
12mod signing_key;
13mod verification_key;

37        #[test]
38        fn verification_key(seed in any::<[u8;32]>()) {
39            // Valid generation check
40            let sk = SchnorrSigningKey::generate(&mut ChaCha20Rng::from_seed(seed));
41            let g = PrimeOrderProjectivePoint::create_generator();
42            let vk = sk.0 * g;
43            let vk_from_sk = SchnorrVerificationKey::new_from_signing_key(sk);
44            assert_eq!(vk, vk_from_sk.0);
45        }

47        #[test]
48        fn valid_signing_verification(
49            msg in prop::collection::vec(any::<u8>(), 1..128),
50            seed in any::<[u8;32]>(),
51        ) {
52            let sk = SchnorrSigningKey::generate(&mut ChaCha20Rng::from_seed(seed));
53            let vk = SchnorrVerificationKey::new_from_signing_key(sk.clone());
54            let base_input = BaseFieldElement::try_from(msg.as_slice()).unwrap();
55
56            let sig_result = sk.sign(&[base_input], &mut ChaCha20Rng::from_seed(seed));
57            assert!(sig_result.is_ok(), "Signature generation failed");
58
59            let sig = sig_result.unwrap();
60
61            assert!(sig.verify(&[base_input], &vk).is_ok(), "Verification failed.");
62        }

64        #[test]
65        fn invalid_signature(msg in prop::collection::vec(any::<u8>(), 1..128), seed in any::<[u8;32]>()) {
66            let mut rng = ChaCha20Rng::from_seed(seed);
67            let sk1 = SchnorrSigningKey::generate(&mut rng);
68            let vk1 = SchnorrVerificationKey::new_from_signing_key(sk1);
69            let sk2 = SchnorrSigningKey::generate(&mut rng);
70            let base_input = BaseFieldElement::try_from(msg.as_slice()).unwrap();
71            let fake_sig = sk2.sign(&[base_input], &mut rng).unwrap();
72
73            let error = fake_sig.verify(&[base_input], &vk1).expect_err("Fake signature should not be verified");
74
75            assert!(
76                matches!(
77                    error.downcast_ref::<UniqueSchnorrSignatureError>(),
78                    Some(UniqueSchnorrSignatureError::SignatureInvalid(_))
79                ),
80                "Unexpected error: {error:?}"
81            );
82        }

84        #[test]
85        fn signing_key_to_from_bytes(seed in any::<[u8;32]>()) {
86            let mut rng = ChaCha20Rng::from_seed(seed);
87            let sk = SchnorrSigningKey::generate(&mut rng);
88            let mut sk_bytes = sk.to_bytes();
89
90            // Valid conversion
91            let recovered_sk = SchnorrSigningKey::from_bytes(&sk_bytes).unwrap();
92            assert_eq!(sk.0, recovered_sk.0, "Recovered signing key does not match with the original!");
93
94            // Not enough bytes
95            let mut short_bytes = [0u8; 31];
96            short_bytes.copy_from_slice(sk_bytes.get(..31).unwrap());
97            let result = SchnorrSigningKey::from_bytes(&short_bytes).expect_err("From bytes conversion of signing key should fail");
98            assert!(
99                matches!(
100                    result.downcast_ref::<UniqueSchnorrSignatureError>(),
101                    Some(UniqueSchnorrSignatureError::Serialization)
102                ),
103                "Unexpected error: {result:?}"
104            );
105
106            // Invalid bytes
107            sk_bytes[31] |= 0xff;
108            let result = SchnorrSigningKey::from_bytes(&sk_bytes).expect_err("From bytes conversion of signing key should fail");
109            assert!(
110                matches!(
111                    result.downcast_ref::<UniqueSchnorrSignatureError>(),
112                    Some(UniqueSchnorrSignatureError::ScalarFieldElementSerialization)
113                ),
114                "Unexpected error: {result:?}"
115            );
116        }

153        #[test]
154        fn signature_to_from_bytes(msg in prop::collection::vec(any::<u8>(), 1..128), seed in any::<[u8;32]>()) {
155            let mut rng = ChaCha20Rng::from_seed(seed);
156            let sk = SchnorrSigningKey::generate(&mut rng);

158            let signature = sk.sign(&[base_input], &mut ChaCha20Rng::from_seed(seed)).unwrap();
159            let signature_bytes = signature.to_bytes();

161            // Valid conversion
162            let recovered_signature = UniqueSchnorrSignature::from_bytes(&signature_bytes).unwrap();

165            // Test invalid `commitment_point`
