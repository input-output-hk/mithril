13/// Check if the given point is on the curve using its coordinates
14pub fn is_on_curve(point: JubjubExtended) -> bool {
15    let point_affine_representation = JubjubAffine::from(point);
16    let (x, y) = (
17        point_affine_representation.get_u(),
18        point_affine_representation.get_v(),
19    );
20    let x_square = x.square();
21    let y_square = y.square();
22
23    let lhs = y_square - x_square;
24    let rhs = JubjubBase::ONE + EDWARDS_D * x_square * y_square;
25
26    lhs == rhs
27}

29/// Extract the coordinates of given points in an Extended form
30///
31/// This is mainly used to feed the Poseidon hash function, the order is maintained
32/// from input to output which is important for the hash function
33pub fn get_coordinates_several_points(points: &[JubjubExtended]) -> Vec<(JubjubBase, JubjubBase)> {
34    let mut points_affine =
35        vec![JubjubAffine::from_raw_unchecked(JubjubBase::ZERO, JubjubBase::ZERO); points.len()];
36    JubjubExtended::batch_normalize(points, &mut points_affine);
37
38    points_affine
39        .into_iter()
40        .map(|p| (p.get_u(), p.get_v()))
41        .collect::<Vec<(JubjubBase, JubjubBase)>>()
42}

44/// Generate a random non zero value from the scalar field of Jubjub
45///
46/// Tries to generate 100 times a non zero value and returns an error if it fails to do so
47pub fn generate_non_zero_scalar<R: RngCore + CryptoRng>(rng: &mut R) -> StmResult<JubjubScalar> {
48    for _ in 0..100 {
49        let random_scalar = JubjubScalar::random(&mut *rng);
50        if random_scalar != JubjubScalar::ZERO {
51            return Ok(random_scalar);
52        }
53    }
54    Err(anyhow!(SchnorrSignatureError::RandomScalarGenerationError))
55        .with_context(|| "Failed to generate a non zero signing key after 100 attempts.")
56}
