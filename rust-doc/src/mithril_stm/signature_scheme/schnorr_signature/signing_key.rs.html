11/// Schnorr Signing key, it is essentially a random scalar of the Jubjub scalar field
12#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
13pub struct SchnorrSigningKey(pub(crate) ScalarFieldElement);
14
15impl SchnorrSigningKey {
16    /// Generate a random scalar value to use as signing key
17    pub fn generate<R: RngCore + CryptoRng>(rng: &mut R) -> StmResult<Self> {
18        let scalar = ScalarFieldElement::new_random_nonzero_scalar(rng)
19            .with_context(|| "Failed to generate a non zero Schnorr signing key.")?;
20        Ok(SchnorrSigningKey(scalar))
21    }
22
23    /// This function is an adapted version of the Schnorr signature scheme that includes
24    /// the computation of a deterministic value (called commitment_point) based on the message and the signing key
25    /// and works with the Jubjub elliptic curve and the Poseidon hash function.
26    ///
27    /// Input:
28    ///     - a message: some bytes
29    ///     - a secret key: an element of the scalar field of the Jubjub curve
30    /// Output:
31    ///     - a signature of the form (commitment_point, response, challenge):
32    ///         - commitment_point is deterministic depending only on the message and secret key
33    ///         - the response and challenge depends on a random value generated during the signature
34    ///
35    /// The protocol computes:
36    ///     - commitment_point = secret_key * H(Sha256(msg))
37    ///     - random_scalar, a random value
38    ///     - random_point_1 = random_scalar * H(Sha256(msg))
39    ///     - random_point_2 = random_scalar * prime_order_generator_point, where generator is a generator of the prime-order subgroup of Jubjub
40    ///     - challenge = Poseidon(DST || H(Sha256(msg)) || verification_key || commitment_point || random_point_1 || random_point_2)
41    ///     - response = random_scalar - challenge * signing_key
42    ///
43    /// Output the signature (commitment_point, response, challenge)
44    ///
45    pub fn sign<R: RngCore + CryptoRng>(
46        &self,
47        msg: &[u8],
48        rng: &mut R,
49    ) -> StmResult<SchnorrSignature> {
50        // Use the subgroup generator to compute the curve points
51        let prime_order_generator_point = PrimeOrderProjectivePoint::create_generator();
52        let verification_key = SchnorrVerificationKey::new_from_signing_key(self.clone())
53            .with_context(|| "Could not generate verification key from signing key.")?;
54
55        // First hashing the message to a scalar then hashing it to a curve point
56        let msg_hash_point = ProjectivePoint::hash_to_projective_point(msg);
57
58        let commitment_point = self.0 * msg_hash_point;
59
60        let random_scalar = ScalarFieldElement::new_random_nonzero_scalar(rng)
61            .with_context(|| "Random scalar generation failed during signing.")?;
62
63        let random_point_1 = random_scalar * msg_hash_point;
64        let random_point_2 = random_scalar * prime_order_generator_point;
65
66        // Since the hash function takes as input scalar elements
67        // We need to convert the EC points to their coordinates
68        // The order must be preserved
69        let points_coordinates: Vec<BaseFieldElement> = [
70            msg_hash_point,
71            ProjectivePoint::from(verification_key.0),
72            commitment_point,
73            random_point_1,
74            ProjectivePoint::from(random_point_2),
75        ]
76        .iter()
77        .flat_map(|point| {
78            let (u, v) = point.get_coordinates();
79            [u, v]
80        })
81        .collect();
82
83        let challenge = compute_truncated_digest(&points_coordinates);
84        let challenge_times_sk = challenge * self.0;
85        let response = random_scalar - challenge_times_sk;
86
87        Ok(SchnorrSignature {
88            commitment_point,
89            response,
90            challenge,
91        })
92    }
93
94    /// Convert a `SchnorrSigningKey` into bytes.
95    pub fn to_bytes(&self) -> [u8; 32] {
96        self.0.to_bytes()
97    }
98
99    /// Convert bytes into a `SchnorrSigningKey`.
100    ///
101    /// The bytes must represent a Jubjub scalar or the conversion will fail
102    pub fn from_bytes(bytes: &[u8]) -> StmResult<Self> {
103        if bytes.len() < 32 {
104            return Err(anyhow!(SchnorrSignatureError::Serialization)).with_context(
105                || "Not enough bytes provided to re-construct a Schnorr signing key.",
106            );
107        }
108        let scalar_field_element = ScalarFieldElement::from_bytes(bytes)
109            .with_context(|| "Could not construct Schnorr signing key from given bytes.")?;
110        Ok(SchnorrSigningKey(scalar_field_element))
111    }
112}
