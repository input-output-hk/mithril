23/// Structure responsible of creating a signer
24#[derive(Debug, Clone, Serialize, Deserialize)]
25pub struct Initializer {
26    /// Stake of the participant
27    pub stake: Stake,
28    /// Protocol parameters.
29    #[serde(rename = "params")]
30    pub parameters: Parameters,
31    /// Signing key for concatenation proof system.
32    #[serde(rename = "sk")]
33    pub bls_signing_key: BlsSigningKey,
34    /// Verification key for concatenation proof system.
35    #[serde(rename = "pk")]
36    pub bls_verification_key_proof_of_possession: VerificationKeyProofOfPossessionForConcatenation,
37    /// Signing key for snark proof system.
38    #[cfg(feature = "future_snark")]
39    #[serde(skip_serializing_if = "Option::is_none")]
40    pub schnorr_signing_key: Option<SchnorrSigningKey>,
41    /// Verification key for snark proof system.
42    #[cfg(feature = "future_snark")]
43    #[serde(skip_serializing_if = "Option::is_none")]
44    pub schnorr_verification_key: Option<VerificationKeyForSnark>,
45}

47impl Initializer {
48    /// Creates a new initializer
49    pub fn new<R: RngCore + CryptoRng>(parameters: Parameters, stake: Stake, rng: &mut R) -> Self {
50        let bls_signing_key = BlsSigningKey::generate(rng);
51        let bls_verification_key_proof_of_possession =
52            VerificationKeyProofOfPossessionForConcatenation::from(&bls_signing_key);
53        #[cfg(feature = "future_snark")]
54        let (schnorr_signing_key, schnorr_verification_key) = {
55            let sk = SchnorrSigningKey::generate(rng);
56            let vk = VerificationKeyForSnark::new_from_signing_key(sk.clone());
57            (Some(sk), Some(vk))
58        };
59        Self {
60            stake,
61            parameters,
62            bls_signing_key,
63            bls_verification_key_proof_of_possession,
64            #[cfg(feature = "future_snark")]
65            schnorr_signing_key,
66            #[cfg(feature = "future_snark")]
67            schnorr_verification_key,
68        }
69    }
70
71    /// Attempts to generate a new signer from the current registration state.
72    ///
73    /// # Process
74    /// 1. Verifies that registration is closed (determined by total stake threshold)
75    /// 2. Confirms the initializer is registered and retrieves its signer index
76    /// 3. Constructs the Merkle tree commitment for each proof system (concatenation and snark)
77    /// 4. Creates the underlying proof system signer
78    ///
79    /// # Errors
80    /// Returns an error if:
81    /// - Registration is not yet closed
82    /// - The initializer is not registered
83    pub fn try_create_signer<D: MembershipDigest>(
84        self,
85        closed_key_registration: &ClosedKeyRegistration,
86    ) -> StmResult<Signer<D>> {
87        let registration_entry = RegistrationEntry::new(
88            self.bls_verification_key_proof_of_possession,
89            self.stake,
90            #[cfg(feature = "future_snark")]
91            self.schnorr_verification_key,
92        )?;
93
94        let signer_index = closed_key_registration
95            .get_signer_index_for_registration(
96                &(
97                    registration_entry,
98                    closed_key_registration.total_stake,
99                    self.parameters.phi_f,
100                )
101                    .try_into()?,
102            )
103            .ok_or_else(|| anyhow!(RegisterError::UnregisteredInitializer))?;
104
105        let key_registration_commitment_for_concatenation = closed_key_registration
106            .to_merkle_tree::<D::ConcatenationHash, RegistrationEntryForConcatenation>()
107            .to_merkle_tree_batch_commitment();
108        let concatenation_proof_signer = ConcatenationProofSigner::new(
109            registration_entry.get_stake(),
110            closed_key_registration.total_stake,
111            self.parameters,
112            self.bls_signing_key,
113            self.bls_verification_key_proof_of_possession.vk,
114            key_registration_commitment_for_concatenation,
115        );
116
117        #[cfg(feature = "future_snark")]
118        let snark_proof_signer = {
119            match (self.schnorr_signing_key, self.schnorr_verification_key) {
120                (Some(schnorr_signing_key), Some(schnorr_verification_key)) => {
121                    let key_registration_commitment_for_snark = closed_key_registration
122                        .to_merkle_tree::<D::SnarkHash, RegistrationEntryForSnark>()
123                        .to_merkle_tree_commitment();
124                    let lottery_target_value = ClosedRegistrationEntry::try_from((
125                        registration_entry,
126                        closed_key_registration.total_stake,
127                        self.parameters.phi_f,
128                    ))?
129                    .get_lottery_target_value()
130                    .ok_or(RegisterError::SnarkProofSignerCreation)
131                    .with_context(|| "missing lottery target value")?;
132                    Some(SnarkProofSigner::new(
133                        self.parameters,
134                        schnorr_signing_key,
135                        schnorr_verification_key,
136                        lottery_target_value,
137                        key_registration_commitment_for_snark,
138                    ))
139                }
140                _ => None,
141            }
142        };
143
144        // Create and return signer
145        Ok(Signer::new(
146            signer_index,
147            concatenation_proof_signer,
148            closed_key_registration.clone(),
149            self.parameters,
150            registration_entry.get_stake(),
151            #[cfg(feature = "future_snark")]
152            snark_proof_signer,
153        ))
154    }

156    /// Extract the verification key with proof of possession.
157    pub fn get_verification_key_proof_of_possession_for_concatenation(
158        &self,
159    ) -> VerificationKeyProofOfPossessionForConcatenation {
160        self.bls_verification_key_proof_of_possession
161    }
162
163    /// Extract the verification key for snark.
164    #[cfg(feature = "future_snark")]
165    pub fn get_verification_key_for_snark(&self) -> Option<VerificationKeyForSnark> {
166        self.schnorr_verification_key
167    }
168
169    /// Remove the SNARK-related keys (Schnorr signing and verification keys) from this
170    /// initializer.
171    ///
172    /// This is used during eras that do not yet support SNARK proofs to ensure the
173    /// initializer's registration entry matches the closed key registration built from
174    /// signers without SNARK verification keys.
175    #[cfg(feature = "future_snark")]
176    pub fn strip_snark_keys(&mut self) {
177        self.schnorr_signing_key = None;
178        self.schnorr_verification_key = None;
179    }

181    /// Convert to bytes using CBOR encoding with a version prefix.
182    pub fn to_bytes(&self) -> StmResult<Vec<u8>> {
183        codec::to_cbor_bytes(self)
184    }
185
186    /// Convert a slice of bytes to an `Initializer`.
187    ///
188    /// Supports both CBOR-encoded (version-prefixed) and legacy big-endian byte formats.
189    ///
190    /// # Error
191    /// The function fails if the given bytes cannot be decoded in either format.
192    pub fn from_bytes(bytes: &[u8]) -> StmResult<Initializer> {
193        codec::from_versioned_bytes(bytes, Self::from_bytes_legacy)
194    }
195
196    /// Decode an `Initializer` from the legacy big-endian byte-packed format.
197    ///
198    /// # Layout
199    /// * Stake (u64)
200    /// * Parameters
201    /// * BLS signing key
202    /// * BLS verification key (including PoP)
203    /// * [Future Snark - Schnorr Signing Key]
204    /// * [Future Snark - Schnorr Verification Key]
205    fn from_bytes_legacy(bytes: &[u8]) -> StmResult<Initializer> {
206        let mut u64_bytes = [0u8; 8];
207        u64_bytes.copy_from_slice(bytes.get(..8).ok_or(RegisterError::SerializationError)?);
208        let stake = u64::from_be_bytes(u64_bytes);
209        let params =
210            Parameters::from_bytes(bytes.get(8..32).ok_or(RegisterError::SerializationError)?)?;
211        let bls_signing_key =
212            BlsSigningKey::from_bytes(bytes.get(32..64).ok_or(RegisterError::SerializationError)?)?;
213        let bls_verification_key_proof_of_possession =
214            VerificationKeyProofOfPossessionForConcatenation::from_bytes(
215                bytes.get(64..256).ok_or(RegisterError::SerializationError)?,
216            )?;
217
218        #[cfg(feature = "future_snark")]
219        let (schnorr_signing_key, schnorr_verification_key) = {
220            let schnorr_signing_key =
221                bytes.get(256..288).map(SchnorrSigningKey::from_bytes).transpose()?;
222            let schnorr_verification_key = bytes
223                .get(288..352)
224                .map(VerificationKeyForSnark::from_bytes)
225                .transpose()?;
226
227            match (&schnorr_signing_key, &schnorr_verification_key) {
228                (Some(_), None) | (None, Some(_)) => {
229                    return Err(RegisterError::SerializationError.into());
230                }
231                _ => {}
232            }
233            (schnorr_signing_key, schnorr_verification_key)
234        };
235
236        Ok(Self {
237            stake,
238            parameters: params,
239            bls_signing_key,
240            bls_verification_key_proof_of_possession,
241            #[cfg(feature = "future_snark")]
242            schnorr_signing_key,
243            #[cfg(feature = "future_snark")]
244            schnorr_verification_key,
245        })
246    }
247}

249impl PartialEq for Initializer {
250    fn eq(&self, other: &Self) -> bool {
251        let base_eq = self.stake == other.stake
252            && self.parameters == other.parameters
253            && self.bls_signing_key.to_bytes() == other.bls_signing_key.to_bytes()
254            && self.get_verification_key_proof_of_possession_for_concatenation()
255                == other.get_verification_key_proof_of_possession_for_concatenation();
256
257        #[cfg(feature = "future_snark")]
258        let base_eq = base_eq
259            && self.schnorr_signing_key == other.schnorr_signing_key
260            && self.schnorr_verification_key == other.schnorr_verification_key;
261
262        base_eq
263    }
264}
