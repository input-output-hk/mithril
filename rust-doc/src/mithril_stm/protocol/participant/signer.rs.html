9/// Single signature generator. Contains the signer's registration index and the signature
10/// generators of each proof system.
11#[derive(Debug, Clone)]
12pub struct Signer<D: MembershipDigest> {
13    /// Index of the signer in registration
14    pub signer_index: SignerIndex,
15    /// Single signature generator of concatenation proof system
16    pub(crate) concatenation_proof_signer: ConcatenationProofSigner<D>,
17    /// Stake associated with the signer
18    stake: Stake,
19    /// Closed key registration containing closed registration entries and total stake
20    pub closed_key_registration: ClosedKeyRegistration,
21    /// Parameters of the protocol
22    pub parameters: Parameters,
23    /// Single signature generator of the future snark proof system
24    #[cfg(feature = "future_snark")]
25    pub(crate) snark_proof_signer: Option<SnarkProofSigner<D>>,
26}
27
28impl<D: MembershipDigest> Signer<D> {
29    /// Creates a new single signature generator
30    pub(crate) fn new(
31        signer_index: SignerIndex,
32        concatenation_proof_signer: ConcatenationProofSigner<D>,
33        closed_key_registration: ClosedKeyRegistration,
34        parameters: Parameters,
35        stake: Stake,
36        #[cfg(feature = "future_snark")] snark_proof_signer: Option<SnarkProofSigner<D>>,
37    ) -> Self {
38        Self {
39            signer_index,
40            concatenation_proof_signer,
41            stake,
42            closed_key_registration,
43            parameters,
44            #[cfg(feature = "future_snark")]
45            snark_proof_signer,
46        }
47    }
48
49    /// Creates and returns a single signature for the given message. First tries to create a
50    /// concatenation proof signature. Tries to create a snark proof signature as well if
51    /// * Concatenation proof signature creation is successful,
52    /// * the future snark proof system is enabled,
53    /// * a snark proof signer is available.
54    pub fn create_single_signature(&self, message: &[u8]) -> StmResult<SingleSignature> {
55        let concatenation_signature =
56            self.concatenation_proof_signer.create_single_signature(message)?;
57
58        // TODO: Update this rng. We might want to use a different RNG for the snark signature generation, or pass it as an argument to the function.
59        #[cfg(feature = "future_snark")]
60        let snark_signature = if let Some(snark_signer) = &self.snark_proof_signer {
61            let mut rng = rand_core::OsRng;
62            snark_signer.create_single_signature(message, &mut rng).ok()
63        } else {
64            None
65        };
66
67        Ok(SingleSignature {
68            concatenation_signature,
69            signer_index: self.signer_index,
70            #[cfg(feature = "future_snark")]
71            snark_signature,
72        })
73    }
74
75    /// Signing function that returns an Option type.
76    pub fn sign(&self, message: &[u8]) -> Option<SingleSignature> {
77        let result = self.create_single_signature(message);
78        result.ok()
79    }
80
81    /// Gets the BLS verification key.
82    pub fn get_bls_verification_key(&self) -> VerificationKeyForConcatenation {
83        self.concatenation_proof_signer.get_verification_key()
84    }
85
86    /// Gets the stake associated with the signer.
87    pub fn get_stake(&self) -> Stake {
88        self.stake
89    }
90    /// Gets the lottery target value.
91    #[cfg(feature = "future_snark")]
92    pub fn get_lottery_target_value(&self) -> Option<LotteryTargetValue> {
93        self.snark_proof_signer
94            .as_ref()
95            .map(|signer| signer.get_lottery_target_value())
96    }
97
98    /// Gets the Schnorr verification key.
99    #[cfg(feature = "future_snark")]
100    pub fn get_schnorr_verification_key(&self) -> Option<VerificationKeyForSnark> {
101        self.snark_proof_signer
102            .as_ref()
103            .map(|signer| signer.get_verification_key())
104    }
105}
