7/// CBOR-friendly envelope for `SingleSignatureWithRegisteredParty` serialization.
8///
9/// Used as an intermediate representation because `SingleSignatureWithRegisteredParty`
10/// has a custom tuple-based `Serialize` implementation that is incompatible with
11/// ciborium's derived `Deserialize` (which expects map format). Each sub-component
12/// is stored as pre-serialized bytes to avoid this mismatch.
13#[derive(Serialize, Deserialize)]
14struct SingleSignatureWithRegisteredPartyCborEnvelope {
15    signature_bytes: Vec<u8>,
16    registration_entry_bytes: Vec<u8>,
17}
18
19/// Signature with its registered party.
20#[derive(Debug, Clone, Hash, Deserialize, Eq, PartialEq, Ord, PartialOrd)]
21pub struct SingleSignatureWithRegisteredParty {
22    /// Stm signature
23    pub sig: SingleSignature,
24    /// Registered party
25    pub reg_party: ClosedRegistrationEntry,
26}
27
28impl SingleSignatureWithRegisteredParty {
29    /// Convert `SingleSignatureWithRegisteredParty` to CBOR bytes with a version prefix.
30    ///
31    /// Uses an intermediate envelope struct to avoid ciborium incompatibility
32    /// with the custom tuple-based `Serialize` implementation.
33    pub fn to_bytes(&self) -> StmResult<Vec<u8>> {
34        let envelope = SingleSignatureWithRegisteredPartyCborEnvelope {
35            signature_bytes: self.sig.to_bytes()?,
36            registration_entry_bytes: self.reg_party.to_bytes()?,
37        };
38        codec::to_cbor_bytes(&envelope)
39    }
40
41    /// Extract a `SingleSignatureWithRegisteredParty` from a byte slice.
42    ///
43    /// Supports both the new versioned CBOR format and the legacy byte-packed format.
44    pub fn from_bytes<D: MembershipDigest>(
45        bytes: &[u8],
46    ) -> StmResult<SingleSignatureWithRegisteredParty> {
47        if codec::has_cbor_v1_prefix(bytes) {
48            let envelope: SingleSignatureWithRegisteredPartyCborEnvelope =
49                codec::from_cbor_bytes(&bytes[1..])?;
50            let sig = SingleSignature::from_bytes::<D>(&envelope.signature_bytes)?;
51            let reg_party =
52                ClosedRegistrationEntry::from_bytes(&envelope.registration_entry_bytes)?;
53            Ok(SingleSignatureWithRegisteredParty { sig, reg_party })
54        } else {
55            Self::from_bytes_legacy::<D>(bytes)
56        }
57    }
58
59    /// Extract a `SingleSignatureWithRegisteredParty` from a byte slice using the legacy format.
60    /// # Layout
61    /// * RegParty length (u64 big-endian)
62    /// * RegParty
63    /// * Signature length (u64 big-endian)
64    /// * Signature
65    fn from_bytes_legacy<D: MembershipDigest>(
66        bytes: &[u8],
67    ) -> StmResult<SingleSignatureWithRegisteredParty> {
68        let mut u64_bytes = [0u8; 8];
69
70        u64_bytes.copy_from_slice(bytes.get(0..8).ok_or(SignatureError::SerializationError)?);
71        let size_reg_party = u64::from_be_bytes(u64_bytes) as usize;
72        let reg_party = ClosedRegistrationEntry::from_bytes(
73            bytes
74                .get(8..8 + size_reg_party)
75                .ok_or(SignatureError::SerializationError)?,
76        )?;
77
78        let sig_offset = 8 + size_reg_party;
79        u64_bytes.copy_from_slice(
80            bytes
81                .get(sig_offset..sig_offset + 8)
82                .ok_or(SignatureError::SerializationError)?,
83        );
84        let size_sig = u64::from_be_bytes(u64_bytes) as usize;
85        let sig = SingleSignature::from_bytes::<D>(
86            bytes
87                .get(sig_offset + 8..sig_offset + 8 + size_sig)
88                .ok_or(SignatureError::SerializationError)?,
89        )?;
90
91        Ok(SingleSignatureWithRegisteredParty { sig, reg_party })
92    }
93}

95impl Serialize for SingleSignatureWithRegisteredParty {
96    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
97    where
98        S: Serializer,
99    {
100        let mut tuple = serializer.serialize_tuple(2)?;
101        tuple.serialize_element(&self.sig)?;
102        tuple.serialize_element(&self.reg_party)?;
103        tuple.end()
104    }
105}
