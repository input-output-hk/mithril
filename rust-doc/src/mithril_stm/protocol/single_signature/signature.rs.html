17/// Single signature created by a single party who has won the lottery.
18/// Contains the underlying signature for the proof system and the registration index of the signer.
19#[derive(Debug, Clone, Serialize, Deserialize)]
20pub struct SingleSignature {
21    /// Underlying signature for concatenation proof system.
22    #[serde(flatten)]
23    pub(crate) concatenation_signature: SingleSignatureForConcatenation,
24    /// Merkle tree index of the signer.
25    pub signer_index: SignerIndex,
26    /// Underlying signature for snark proof system.
27    #[cfg(feature = "future_snark")]
28    #[serde(skip_serializing_if = "Option::is_none", default)]
29    pub(crate) snark_signature: Option<SingleSignatureForSnark>,
30}

32impl SingleSignature {
33    /// Verify a `SingleSignature` by validating the underlying single signature for proof system.
34    pub fn verify<D: MembershipDigest>(
35        &self,
36        params: &Parameters,
37        pk: &VerificationKeyForConcatenation,
38        stake: &Stake,
39        avk: &AggregateVerificationKey<D>,
40        msg: &[u8],
41        #[cfg(feature = "future_snark")] snark_registration_entry: Option<
42            RegistrationEntryForSnark,
43        >,
44    ) -> StmResult<()> {
45        #[cfg(feature = "future_snark")]
46        if let (Some(snark_signature), Some(entry), Some(snark_avk)) = (
47            &self.snark_signature,
48            snark_registration_entry,
49            avk.to_snark_aggregate_verification_key(),
50        ) {
51            snark_signature.verify(&entry.0, msg, snark_avk)?;
52        }
53
54        self.concatenation_signature.verify(
55            params,
56            pk,
57            stake,
58            avk.to_concatenation_aggregate_verification_key(),
59            msg,
60        )
61    }

63    /// Verify that all lottery indices of the concatenation signature are valid.
64    ///
65    /// This only applies to the concatenation proof system, which checks that the signature indices
66    /// won the lottery and are within the valid range. The SNARK proof system does not need
67    /// equivalent checking here, its lottery validity is handled during the aggregation process.
68    pub(crate) fn check_indices(
69        &self,
70        params: &Parameters,
71        stake: &Stake,
72        msg: &[u8],
73        total_stake: &Stake,
74    ) -> StmResult<()> {
75        self.concatenation_signature
76            .check_indices(params, stake, msg, total_stake)
77    }

79    /// Convert a `SingleSignature` into bytes
80    ///
81    /// # Layout
82    /// * Concatenation proof system single signature bytes:
83    /// *   Length of indices
84    /// *   Indices
85    /// *   Sigma
86    /// * Merkle index of the signer.
87    /// * (Optional) - `future_snark` Snark proof system single signature bytes:
88    /// *   Schnorr signature bytes
89    /// *   Length of indices
90    /// *   Indices
91    pub fn to_bytes(&self) -> Vec<u8> {
92        let mut output = Vec::new();
93        let indices = self.get_concatenation_signature_indices();
94        output.extend_from_slice(&(indices.len() as u64).to_be_bytes());
95
96        for index in indices {
97            output.extend_from_slice(&index.to_be_bytes());
98        }
99
100        output.extend_from_slice(&self.get_concatenation_signature_sigma().to_bytes());
101
102        output.extend_from_slice(&self.signer_index.to_be_bytes());
103
104        #[cfg(feature = "future_snark")]
105        if let Some(snark_signature) = &self.snark_signature {
106            let unique_schnorr_signature = snark_signature.get_schnorr_signature();
107            output.extend_from_slice(&unique_schnorr_signature.to_bytes());
108            let snark_indices = snark_signature.get_indices();
109            output.extend_from_slice(&(snark_indices.len() as u64).to_be_bytes());
110            for index in snark_indices {
111                output.extend_from_slice(&index.to_be_bytes());
112            }
113        }
114
115        output
116    }

118    /// Extract a `SingleSignature` from a byte slice.
119    pub fn from_bytes<D: MembershipDigest>(bytes: &[u8]) -> StmResult<SingleSignature> {
120        let mut u64_bytes = [0u8; 8];
121
122        u64_bytes.copy_from_slice(bytes.get(0..8).ok_or(SignatureError::SerializationError)?);
123        let nr_indexes = u64::from_be_bytes(u64_bytes) as usize;
124
125        let mut indexes = Vec::new();
126        for i in 0..nr_indexes {
127            u64_bytes.copy_from_slice(
128                bytes
129                    .get(8 + i * 8..16 + i * 8)
130                    .ok_or(SignatureError::SerializationError)?,
131            );
132            indexes.push(u64::from_be_bytes(u64_bytes));
133        }
134
135        let offset = 8 + nr_indexes * 8;
136        let sigma = BlsSignature::from_bytes(
137            bytes
138                .get(offset..offset + 48)
139                .ok_or(SignatureError::SerializationError)?,
140        )?;
141
142        u64_bytes.copy_from_slice(
143            bytes
144                .get(offset + 48..offset + 56)
145                .ok_or(SignatureError::SerializationError)?,
146        );
147        let signer_index = u64::from_be_bytes(u64_bytes);
148
149        #[cfg(feature = "future_snark")]
150        let snark_signature = {
151            let snark_offset = offset + 56;
152            if snark_offset < bytes.len() {
153                let schnorr_signature = crate::UniqueSchnorrSignature::from_bytes(
154                    bytes
155                        .get(snark_offset..snark_offset + 96)
156                        .ok_or(SignatureError::SerializationError)?,
157                )?;
158                let mut snark_idx_offset = snark_offset + 96;
159                u64_bytes.copy_from_slice(
160                    bytes
161                        .get(snark_idx_offset..snark_idx_offset + 8)
162                        .ok_or(SignatureError::SerializationError)?,
163                );
164                let nr_snark_indices = u64::from_be_bytes(u64_bytes) as usize;
165                snark_idx_offset += 8;
166
167                let mut snark_indices = Vec::with_capacity(nr_snark_indices);
168                for i in 0..nr_snark_indices {
169                    u64_bytes.copy_from_slice(
170                        bytes
171                            .get(snark_idx_offset + i * 8..snark_idx_offset + (i + 1) * 8)
172                            .ok_or(SignatureError::SerializationError)?,
173                    );
174                    snark_indices.push(u64::from_be_bytes(u64_bytes));
175                }
176
177                Some(SingleSignatureForSnark::new(
178                    schnorr_signature,
179                    snark_indices,
180                ))
181            } else {
182                None
183            }
184        };
185
186        Ok(SingleSignature {
187            concatenation_signature: SingleSignatureForConcatenation::new(sigma, indexes),
188            signer_index,
189            #[cfg(feature = "future_snark")]
190            snark_signature,
191        })
192    }
