14/// `Clerk` can verify and aggregate `SingleSignature`s and verify `AggregateSignature`s.
15/// Clerks can only be generated with the registration closed.
16/// This avoids that a Merkle Tree is computed before all parties have registered.
17#[derive(Debug, Clone)]
18pub struct Clerk<D: MembershipDigest> {
19    pub(crate) closed_reg: ClosedKeyRegistration<D>,
20    pub(crate) params: Parameters,
21}
22
23impl<D: MembershipDigest> Clerk<D> {
24    /// Create a new `Clerk` from a closed registration instance.
25    pub fn new_clerk_from_closed_key_registration(
26        params: &Parameters,
27        closed_reg: &ClosedKeyRegistration<D>,
28    ) -> Self {
29        Self {
30            params: *params,
31            closed_reg: closed_reg.clone(),
32        }
33    }

35    /// Create a new `Clerk` from a closed registration instance.

44    /// Create a Clerk from a signer.
45    pub fn new_clerk_from_signer(signer: &Signer<D>) -> Self {
46        let closed_reg = signer
47            .get_closed_key_registration()
48            .clone()
49            .expect("Core signer does not include closed registration. Clerk, and so, the Stm certificate cannot be built without closed registration!")
50            ;
51
52        Self {
53            params: signer.get_parameters(),
54            closed_reg,
55        }
56    }
57
58    /// Create a Clerk from a signer.
59    #[deprecated(since = "0.5.0", note = "Use `new_clerk_from_signer` instead")]
60    pub fn from_signer(signer: &Signer<D>) -> Self {
61        Self::new_clerk_from_signer(signer)
62    }

64    /// Aggregate a set of signatures.
65    #[deprecated(since = "0.5.3", note = "Use `aggregate_signatures_with_type` instead")]
66    pub fn aggregate_signatures(
67        &self,
68        sigs: &[SingleSignature],
69        msg: &[u8],
70    ) -> StmResult<AggregateSignature<D>> {
71        self.aggregate_signatures_with_type(sigs, msg, AggregateSignatureType::default())
72    }
73
74    /// Aggregate a set of signatures with a given proof type.
75    pub fn aggregate_signatures_with_type(
76        &self,
77        sigs: &[SingleSignature],
78        msg: &[u8],
79        aggregate_signature_type: AggregateSignatureType,
80    ) -> StmResult<AggregateSignature<D>> {
81        match aggregate_signature_type {
82            AggregateSignatureType::Concatenation => Ok(AggregateSignature::Concatenation(
83                ConcatenationProof::aggregate_signatures(self, sigs, msg).with_context(|| {
84                    format!(
85                        "Signatures failed to aggregate for type {}",
86                        AggregateSignatureType::Concatenation
87                    )
88                })?,
89            )),
90            #[cfg(feature = "future_snark")]
91            AggregateSignatureType::Future => Err(anyhow!(
92                AggregationError::UnsupportedProofSystem(aggregate_signature_type)
93            )),
94        }
95    }

97    /// Aggregate a set of signatures for their corresponding indices.
98    ///
99    /// This function first deduplicates the repeated signatures, and if there are enough signatures, it collects the merkle tree indexes of unique signatures.
100    /// The list of merkle tree indexes is used to create a batch proof, to prove that all signatures are from eligible signers.
101    ///
102    /// It returns an instance of `AggregateSignature`.
103    #[deprecated(since = "0.5.0", note = "Use `aggregate_signatures` instead")]
104    #[allow(deprecated)]
105    pub fn aggregate(
106        &self,
107        sigs: &[SingleSignature],
108        msg: &[u8],
109    ) -> StmResult<AggregateSignature<D>> {
110        Self::aggregate_signatures(self, sigs, msg)
111    }

144    /// Given a slice of `sig_reg_list`, this function returns a new list of `sig_reg_list` with only valid indices.
145    /// In case of conflict (having several signatures for the same index)
146    /// it selects the smallest signature (i.e. takes the signature with the smallest scalar).
147    /// The function selects at least `self.k` indexes.
148    ///  # Error
149    /// If there is no sufficient signatures, then the function fails.
150    // todo: We need to agree on a criteria to dedup (by default we use a BTreeMap that guarantees keys order)
151    // todo: not good, because it only removes index if there is a conflict (see benches)
152    pub fn select_valid_signatures_for_k_indices(
153        params: &Parameters,
154        msg: &[u8],
155        sigs: &[SingleSignatureWithRegisteredParty],
156        avk: &AggregateVerificationKey<D>,
157    ) -> StmResult<Vec<SingleSignatureWithRegisteredParty>> {
158        let mut sig_by_index: BTreeMap<Index, &SingleSignatureWithRegisteredParty> =
159            BTreeMap::new();
160        let mut removal_idx_by_vk: HashMap<&SingleSignatureWithRegisteredParty, Vec<Index>> =
161            HashMap::new();
162
163        for sig_reg in sigs.iter() {
164            if sig_reg
165                .sig
166                .verify(params, &sig_reg.reg_party.0, &sig_reg.reg_party.1, avk, msg)
167                .is_err()
168            {
169                continue;
170            }
171            for index in sig_reg.sig.indexes.iter() {
172                let mut insert_this_sig = false;
173                if let Some(&previous_sig) = sig_by_index.get(index) {
174                    let sig_to_remove_index = if sig_reg.sig.sigma < previous_sig.sig.sigma {
175                        insert_this_sig = true;
176                        previous_sig
177                    } else {
178                        sig_reg
179                    };
180
181                    if let Some(indexes) = removal_idx_by_vk.get_mut(sig_to_remove_index) {
182                        indexes.push(*index);
183                    } else {
184                        removal_idx_by_vk.insert(sig_to_remove_index, vec![*index]);
185                    }
186                } else {
187                    insert_this_sig = true;
188                }
189
190                if insert_this_sig {
191                    sig_by_index.insert(*index, sig_reg);
192                }
193            }
194        }
195
196        let mut dedup_sigs: HashSet<SingleSignatureWithRegisteredParty> = HashSet::new();
197        let mut count: u64 = 0;
198
199        for (_, &sig_reg) in sig_by_index.iter() {
200            if dedup_sigs.contains(sig_reg) {
201                continue;
202            }
203            let mut deduped_sig = sig_reg.clone();
204            if let Some(indexes) = removal_idx_by_vk.get(sig_reg) {
205                deduped_sig.sig.indexes = deduped_sig
206                    .sig
207                    .indexes
208                    .clone()
209                    .into_iter()
210                    .filter(|i| !indexes.contains(i))
211                    .collect();
212            }
213
214            let size: Result<u64, _> = deduped_sig.sig.indexes.len().try_into();
215            if let Ok(size) = size {
216                if dedup_sigs.contains(&deduped_sig) {
217                    panic!("Should not reach!");
218                }
219                dedup_sigs.insert(deduped_sig);
220                count += size;
221
222                if count >= params.k {
223                    return Ok(dedup_sigs.into_iter().collect());
224                }
225            }
226        }
227        Err(anyhow!(AggregationError::NotEnoughSignatures(
228            count, params.k
229        )))
230    }

235    use proptest::prelude::*;
236    use rand_chacha::ChaCha20Rng;
237    use rand_core::{RngCore, SeedableRng};

239    use crate::{

244    use super::AggregationError;

247
248    proptest! {
249        #![proptest_config(ProptestConfig::with_cases(50))]
250
251        #[test]
252        fn test_dedup(
253            seed in any::<[u8; 32]>(),
254            msg in any::<[u8;16]>(),
255            nparties in 1_usize..10,
256            m in 1_u64..20,
257            k in 1_u64..10,
258            phi_f in 0.1_f64..1.0,
259            num_invalid_sigs_per_party in 0_usize..3,
260        ) {
261            let params = Parameters { m, k, phi_f };
262            let mut rng = ChaCha20Rng::from_seed(seed);
263
264            // False messages
265            let mut false_msgs = Vec::new();
266            for _ in 0..num_invalid_sigs_per_party {
267                let mut false_msg = vec![0u8; 32];
268                rng.fill_bytes(&mut false_msg);
269                if false_msg == msg {
270                    false_msg[0] = msg[0].wrapping_add(1);
271                }
272                false_msgs.push(false_msg);
273            }
274
275            let mut key_registration = KeyRegistration::init();
276            let mut initializers = Vec::new();
277
278            for i in 0..nparties {
279                let stake = (i as u64 + 1) * 10;
280                let initializer = Initializer::new(params, stake, &mut rng);
281                key_registration.register(initializer.stake, initializer.pk).unwrap();
282                initializers.push(initializer);
283            }
284
285            let closed_registration: ClosedKeyRegistration<D> = key_registration.close();
286
287            let signers: Vec<_> = initializers
288                .into_iter()
289                .map(|init| init.create_signer(closed_registration.clone()).unwrap())
290                .collect();
291
292            let clerk = Clerk::new_clerk_from_signer(&signers[0]);
293            let avk = clerk.compute_aggregate_verification_key();
294
295            let mut all_sigs = Vec::new();
296            for signer in &signers {
297                // Add invalid signatures
298                for false_msg in &false_msgs {
299                    if let Some(sig) = signer.sign(false_msg) {
300                        all_sigs.push(sig);
301                    }
302                }
303                // Add valid signatures
304                if let Some(sig) = signer.sign(&msg) {
305                    all_sigs.push(sig);
306                }
307            }
308
309            let sig_reg_list = all_sigs
310                .iter()
311                .map(|sig| SingleSignatureWithRegisteredParty {
312                    sig: sig.clone(),
313                    reg_party: clerk.closed_reg.reg_parties[sig.signer_index as usize],
314                })
315                .collect::<Vec<SingleSignatureWithRegisteredParty>>();
316
317            let dedup_result =
318                Clerk::select_valid_signatures_for_k_indices(&params, &msg, &sig_reg_list, &avk);
319
320            match dedup_result {
321                Ok(valid_sigs) => {
322                    assert!(!valid_sigs.is_empty(), "Should have at least one valid signature");
323
324                    for passed_sigs in valid_sigs {
325                        let signer = &signers[passed_sigs.sig.signer_index as usize];
326                        let verify_result = passed_sigs.sig.verify(
327                            &params,
328                            &signer.get_verification_key(),
329                            &signer.get_stake(),
330                            &avk,
331                            &msg,
332                        );
333                        assert!(verify_result.is_ok(), "All returned signatures should verify: {:?}", verify_result);
334                    }
335                }
336                Err(error) => {
337                    assert!(
338                        matches!(
339                            error.downcast_ref::<AggregationError>(),
340                            Some(AggregationError::NotEnoughSignatures(..))
341                        ),
342                        "Expected NotEnoughSignatures, got: {:?}", error
343                    );
344                }
345            }
346        }
347    }
