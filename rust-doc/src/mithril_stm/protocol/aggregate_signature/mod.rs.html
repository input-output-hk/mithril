37    fn setup_parties(params: Parameters, stake: Vec<Stake>) -> Vec<Signer<D>> {
38        let mut kr = KeyRegistration::init();
39        let mut trng = TestRng::deterministic_rng(ChaCha);
40        let mut rng = ChaCha20Rng::from_seed(trng.random());
41
42        #[allow(clippy::needless_collect)]
43        let ps = stake
44            .into_iter()
45            .map(|stake| {
46                let p = Initializer::new(params, stake, &mut rng);
47                kr.register(stake, p.pk).unwrap();
48                p
49            })
50            .collect::<Vec<_>>();
51        let closed_reg = kr.close();
52        ps.into_iter()
53            .map(|p| p.create_signer(closed_reg.clone()).unwrap())
54            .collect()
55    }

57    /// Generate a vector of stakes that should sum to `honest_stake`
58    /// when ignoring the indices in `adversaries`
59    fn arb_honest_for_adversaries(
60        num_parties: usize,
61        honest_stake: Stake,
62        adversaries: HashMap<usize, Stake>,
63    ) -> impl Strategy<Value = Vec<Stake>> {
64        vec(1..honest_stake, num_parties).prop_map(move |parties| {
65            let honest_sum = parties.iter().enumerate().fold(0, |acc, (i, s)| {
66                if !adversaries.contains_key(&i) {
67                    acc + s
68                } else {
69                    acc
70                }
71            });
72
73            parties
74                .iter()
75                .enumerate()
76                .map(|(i, s)| {
77                    if let Some(a) = adversaries.get(&i) {
78                        *a
79                    } else {
80                        (*s * honest_stake) / honest_sum
81                    }
82                })
83                .collect()
84        })
85    }

87    /// Generate a vector of `num_parties` stakes summing to `num_parties * total_stake`,
88    /// plus a subset S of 0..num_parties such that the sum of the stakes at indices
89    /// in S is adversary_stake * N
90    fn arb_parties_with_adversaries(
91        num_parties: usize,
92        num_adversaries: usize,
93        total_stake: Stake,
94        adversary_stake: Stake,
95    ) -> impl Strategy<Value = (HashSet<usize>, Vec<Stake>)> {
96        hash_map(0..num_parties, 1..total_stake, num_adversaries).prop_flat_map(
97            move |adversaries| {
98                let adversary_sum: Stake = adversaries.values().sum();
99                let adversaries_normed = adversaries
100                    .iter()
101                    .map(|(a, stake)| (*a, (stake * adversary_stake) / adversary_sum))
102                    .collect();
103
104                let adversaries = adversaries.into_keys().collect();
105                (
106                    Just(adversaries),
107                    arb_honest_for_adversaries(
108                        num_parties,
109                        total_stake - adversary_stake,
110                        adversaries_normed,
111                    ),
112                )
113            },
114        )
115    }

117    fn find_signatures(msg: &[u8], ps: &[Signer<D>], is: &[usize]) -> Vec<SingleSignature> {
118        let mut sigs = Vec::new();
119        for i in is {
120            if let Some(sig) = ps[*i].sign(msg) {
121                sigs.push(sig);
122            }
123        }
124        sigs
125    }

127    /// Pick N between min and max, and then
128    /// generate a vector of N stakes summing to N * tstake,
129    /// plus a subset S of 0..N such that the sum of the stakes at indices
130    /// in S is astake * N
131    fn arb_parties_adversary_stake(
132        min: usize,
133        max: usize,
134        tstake: Stake,
135        astake: Stake,
136    ) -> impl Strategy<Value = (HashSet<usize>, Vec<Stake>)> {
137        (min..max)
138            .prop_flat_map(|n| (Just(n), 1..=n / 2))
139            .prop_flat_map(move |(n, nadv)| {
140                arb_parties_with_adversaries(n, nadv, tstake * n as Stake, astake * n as Stake)
141            })
142    }

150    /// Run the protocol up to aggregation. This will produce a valid aggregation of signatures.
151    /// The following tests mutate this aggregation so that the proof is no longer valid.
152    fn arb_proof_setup(max_parties: usize) -> impl Strategy<Value = ProofTest> {
153        any::<[u8; 16]>().prop_flat_map(move |msg| {
154            (2..max_parties).prop_map(move |n| {
155                let params = Parameters {
156                    m: 5,
157                    k: 5,
158                    phi_f: 1.0,
159                };
160                let ps = setup_equal_parties(params, n);
161                let clerk = Clerk::new_clerk_from_signer(&ps[0]);
162
163                let all_ps: Vec<usize> = (0..n).collect();
164                let sigs = find_signatures(&msg, &ps, &all_ps);
165
166                let aggr_sig_type = AggregateSignatureType::Concatenation;
167                let msig = clerk.aggregate_signatures_with_type(&sigs, &msg, aggr_sig_type);
168                ProofTest { msig, clerk, msg }
169            })
170        })
171    }

173    fn with_proof_mod<F>(mut tc: ProofTest, f: F)
174    where
175        F: Fn(&mut Sig, &mut Clerk<D>, &mut [u8; 16]),
176    {
177        match tc.msig {
178            Ok(mut aggr) => {
179                f(&mut aggr, &mut tc.clerk, &mut tc.msg);
180                assert!(
181                    aggr.verify(
182                        &tc.msg,
183                        &tc.clerk.compute_aggregate_verification_key(),
184                        &tc.clerk.params
185                    )
186                    .is_err()
187                )
188            }
189            Err(e) => unreachable!("Reached an unexpected error: {:?}", e),
190        }
191    }

197        /// Test that when a quorum is found, the aggregate signature can be verified by anyone with
198        /// access to the avk and the parameters.
199        fn test_aggregate_sig(nparties in 2_usize..30,
200                              m in 10_u64..20,
201                              k in 1_u64..5,
202                              msg in any::<[u8;16]>()) {
203            let params = Parameters { m, k, phi_f: 0.2 };
204            let ps = setup_equal_parties(params, nparties);
205            let clerk = Clerk::new_clerk_from_signer(&ps[0]);
206            let aggr_sig_type = AggregateSignatureType::Concatenation;
207
208            let all_ps: Vec<usize> = (0..nparties).collect();
209            let sigs = find_signatures(&msg, &ps, &all_ps);
210            let msig = clerk.aggregate_signatures_with_type(&sigs, &msg, aggr_sig_type);
211
212            match msig {
213                Ok(aggr) => {
214                    println!("Aggregate ok");
215                    assert!(aggr.verify(&msg, &clerk.compute_aggregate_verification_key(), &params).is_ok());
216                }
217                Err(error) => match error.downcast_ref::<AggregationError>() {
218                    Some(AggregationError::NotEnoughSignatures(n, k)) => {
219                        println!("Not enough signatures");
220                        assert!(n < &params.k && k == &params.k)
221                    },
222                    Some(AggregationError::UnsupportedProofSystem(aggregate_signature_type)) => {
223                        panic!("Unsupported proof system: {:?}", aggregate_signature_type);
224                    },
225                    _ => {
226                        panic!("Unexpected error during aggregation: {:?}", error);
227                    }
228                },
229            }
230        }

233        /// Test that batch verification of certificates works
234        fn batch_verify(nparties in 2_usize..30,
235                              m in 10_u64..20,
236                              k in 1_u64..4,
237                              seed in any::<[u8;32]>(),
238                              batch_size in 2..10,
239        ) {
240            let aggr_sig_type = AggregateSignatureType::Concatenation;
241            let mut rng = ChaCha20Rng::from_seed(seed);
242            let mut aggr_avks = Vec::new();
243            let mut aggr_stms = Vec::new();
244            let mut batch_msgs = Vec::new();
245            let mut batch_params = Vec::new();
246            for _ in 0..batch_size {
247                let mut msg = [0u8; 32];
248                rng.fill_bytes(&mut msg);
249                let params = Parameters { m, k, phi_f: 0.95 };
250                let ps = setup_equal_parties(params, nparties);
251                let clerk = Clerk::new_clerk_from_signer(&ps[0]);
252
253                let all_ps: Vec<usize> = (0..nparties).collect();
254                let sigs = find_signatures(&msg, &ps, &all_ps);
255                let msig = clerk.aggregate_signatures_with_type(&sigs, &msg, aggr_sig_type);
256
257                match msig {
258                    Ok(aggr) => {
259                        aggr_avks.push(clerk.compute_aggregate_verification_key());
260                        aggr_stms.push(aggr);
261                        batch_msgs.push(msg.to_vec());
262                        batch_params.push(params);
263                    }
264                    Err(error) => { assert!(
265                        matches!(
266                            error.downcast_ref::<AggregationError>(),
267                            Some(AggregationError::NotEnoughSignatures{..})
268                        ),
269                        "Unexpected error: {error:?}");
270                    }
271                }
272            }
273
274            assert!(AggregateSignature::batch_verify(&aggr_stms, &batch_msgs, &aggr_avks, &batch_params).is_ok());
275
276            let mut msg = [0u8; 32];
277            rng.fill_bytes(&mut msg);
278            let params = Parameters { m, k, phi_f: 0.8 };
279            let ps = setup_equal_parties(params, nparties);
280            let clerk = Clerk::new_clerk_from_signer(&ps[0]);
281
282            let all_ps: Vec<usize> = (0..nparties).collect();
283            let sigs = find_signatures(&msg, &ps, &all_ps);
284            let fake_msig = clerk.aggregate_signatures_with_type(&sigs, &msg, aggr_sig_type);
285
286            aggr_stms[0] = fake_msig.unwrap();
287            assert!(AggregateSignature::batch_verify(&aggr_stms, &batch_msgs, &aggr_avks, &batch_params).is_err());
288        }
289    }

367        /// Test that when the adversaries do not hold sufficient stake, they can not form a quorum
368        fn test_adversary_quorum(
369            (adversaries, parties) in arb_parties_adversary_stake(8, 30, 16, 4),
370            msg in any::<[u8;16]>(),
371        ) {
372            // Test sanity check:
373            // Check that the adversarial party has less than 40% of the total stake.
374            let (good, bad) = parties.iter().enumerate().fold((0,0), |(acc1, acc2), (i, st)| {
375                if adversaries.contains(&i) {
376                    (acc1, acc2 + *st)
377                } else {
378                    (acc1 + *st, acc2)
379                }
380            });
381            assert!(bad as f64 / ((good + bad) as f64) < 0.4);
382
383            let params = Parameters { m: 2642, k: 357, phi_f: 0.2 }; // From Table 1
384            let ps = setup_parties(params, parties);
385
386            let sigs =  find_signatures(&msg, &ps, &adversaries.into_iter().collect::<Vec<_>>());
387
388            assert!(sigs.len() < params.k as usize);
389
390            let clerk = Clerk::new_clerk_from_signer(&ps[0]);
391            let aggr_sig_type = AggregateSignatureType::Concatenation;
392
393            let error = clerk.aggregate_signatures_with_type(&sigs, &msg, aggr_sig_type).expect_err("Not enough quorum should fail!");
394            assert!(
395                matches!(
396                    error.downcast_ref::<AggregationError>(),
397                    Some(AggregationError::NotEnoughSignatures{..})
398                ),
399                "Unexpected error: {error:?}");
400            }
