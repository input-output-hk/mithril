13/// Path of hashes from root to leaf in a Merkle Tree.
14/// Contains all hashes on the path, and the index of the leaf.
15/// Used to verify that signatures come from eligible signers.
16#[derive(Default, Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
17pub struct MerklePath<D: Digest> {
18    pub(crate) values: Vec<Vec<u8>>,
19    pub(crate) index: usize,
20    hasher: PhantomData<D>,
21}
22
23#[cfg(feature = "future_snark")]
24// TODO: remove this allow dead_code directive when function is called or future_snark is activated
25#[allow(dead_code)]
26impl<D: Digest + FixedOutput> MerklePath<D> {
27    pub(crate) fn new(values: Vec<Vec<u8>>, index: usize) -> Self {
28        Self {
29            values,
30            index,
31            hasher: Default::default(),
32        }
33    }
34
35    /// Convert to bytes
36    /// # Layout
37    /// * Index representing the position in the Merkle Tree
38    /// * Size of the Path
39    /// * Path of hashes
40    pub fn to_bytes(&self) -> Vec<u8> {
41        let mut output = Vec::new();
42        output.extend_from_slice(&u64::try_from(self.index).unwrap().to_be_bytes());
43        output.extend_from_slice(&u64::try_from(self.values.len()).unwrap().to_be_bytes());
44        for value in &self.values {
45            output.extend_from_slice(value)
46        }
47
48        output
49    }
50
51    /// Extract a `Path` from a byte slice.
52    /// # Error
53    /// This function fails if the bytes cannot retrieve path.
54    pub fn from_bytes(bytes: &[u8]) -> StmResult<MerklePath<D>> {
55        let mut u64_bytes = [0u8; 8];
56        u64_bytes.copy_from_slice(bytes.get(..8).ok_or(MerkleTreeError::SerializationError)?);
57        let index = usize::try_from(u64::from_be_bytes(u64_bytes))
58            .map_err(|_| MerkleTreeError::SerializationError)?;
59        u64_bytes.copy_from_slice(bytes.get(8..16).ok_or(MerkleTreeError::SerializationError)?);
60        let len = usize::try_from(u64::from_be_bytes(u64_bytes))
61            .map_err(|_| MerkleTreeError::SerializationError)?;
62        let mut values = Vec::with_capacity(len);
63        for i in 0..len {
64            values.push(
65                bytes
66                    .get(
67                        16 + i * <D as Digest>::output_size()
68                            ..16 + (i + 1) * <D as Digest>::output_size(),
69                    )
70                    .ok_or(MerkleTreeError::SerializationError)?
71                    .to_vec(),
72            );
73        }
74
75        Ok(MerklePath {
76            values,
77            index,
78            hasher: PhantomData,
79        })
80    }
81}

83/// Path of hashes for a batch of indices.
84/// Contains the hashes and the corresponding merkle tree indices of given batch.
85/// Used to verify the signatures are issued by the registered signers.
86#[derive(Default, Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
87pub struct MerkleBatchPath<D: Digest + FixedOutput> {
88    pub(crate) values: Vec<Vec<u8>>,
89    pub(crate) indices: Vec<usize>,
90    pub(crate) hasher: PhantomData<D>,
91}
92
93impl<D: Digest + FixedOutput> MerkleBatchPath<D> {
94    pub(crate) fn new(values: Vec<Vec<u8>>, indices: Vec<usize>) -> MerkleBatchPath<D> {
95        Self {
96            values,
97            indices,
98            hasher: Default::default(),
99        }
100    }
101
102    /// Convert the `BatchPath` into byte representation.
103    ///
104    /// # Layout
105    /// The layout of a `BatchPath` is
106    /// * Length of values
107    /// * Length of indices
108    /// * Values
109    /// * Indices
110    pub fn to_bytes(&self) -> Vec<u8> {
111        let mut output = Vec::new();
112        let len_v = self.values.len();
113        let len_i = self.indices.len();
114
115        output.extend_from_slice(&u64::try_from(len_v).unwrap().to_be_bytes());
116        output.extend_from_slice(&u64::try_from(len_i).unwrap().to_be_bytes());
117
118        for value in &self.values {
119            output.extend_from_slice(value.as_slice())
120        }
121
122        for &index in &self.indices {
123            output.extend_from_slice(&u64::try_from(index).unwrap().to_be_bytes());
124        }
125        output
126    }
127
128    /// Try to convert a byte string into a `BatchPath`.
129    pub fn from_bytes(bytes: &[u8]) -> StmResult<Self> {
130        let mut u64_bytes = [0u8; 8];
131        u64_bytes.copy_from_slice(&bytes[..8]);
132        let len_v = usize::try_from(u64::from_be_bytes(u64_bytes))
133            .map_err(|_| MerkleTreeError::SerializationError)?;
134
135        u64_bytes.copy_from_slice(&bytes[8..16]);
136        let len_i = usize::try_from(u64::from_be_bytes(u64_bytes))
137            .map_err(|_| MerkleTreeError::SerializationError)?;
138
139        let mut values = Vec::with_capacity(len_v);
140        for i in 0..len_v {
141            values.push(
142                bytes[16 + i * <D as Digest>::output_size()
143                    ..16 + (i + 1) * <D as Digest>::output_size()]
144                    .to_vec(),
145            );
146        }
147        let offset = 16 + len_v * <D as Digest>::output_size();
148
149        let mut indices = Vec::with_capacity(len_v);
150        for i in 0..len_i {
151            u64_bytes.copy_from_slice(&bytes[offset + i * 8..offset + (i + 1) * 8]);
152            indices.push(
153                usize::try_from(u64::from_be_bytes(u64_bytes))
154                    .map_err(|_| MerkleTreeError::SerializationError)?,
155            );
156        }
157
158        Ok(MerkleBatchPath {
159            values,
160            indices,
161            hasher: PhantomData,
162        })
163    }
164}
