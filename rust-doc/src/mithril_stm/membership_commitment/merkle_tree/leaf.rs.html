9/// Trait implemented to be used as a Merkle tree leaf.
10pub trait MerkleTreeLeaf: Clone + Send + Sync + Copy {
11    /// Converts the Merkle tree leaf to a bytes representation used internally by the Merkle tree
12    fn as_bytes_for_merkle_tree(&self) -> Vec<u8>;
13}
14
15/// The values that are committed in the Merkle Tree for `ConcatenationProof`.
16/// Namely, a verified `BlsVerificationKey` and its corresponding stake.
17#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Serialize, Deserialize, Hash)]
18pub struct MerkleTreeConcatenationLeaf(pub BlsVerificationKey, pub Stake);
19
20impl MerkleTreeLeaf for MerkleTreeConcatenationLeaf {
21    fn as_bytes_for_merkle_tree(&self) -> Vec<u8> {
22        self.to_bytes()
23    }
24}
25
26impl MerkleTreeConcatenationLeaf {
27    fn to_bytes(self) -> Vec<u8> {
28        let mut result = [0u8; 104];
29        result[..96].copy_from_slice(&self.0.to_bytes());
30        result[96..].copy_from_slice(&self.1.to_be_bytes());
31        result.to_vec()
32    }
33
34    pub(crate) fn from_bytes(bytes: &[u8]) -> StmResult<Self> {
35        let pk =
36            VerificationKey::from_bytes(bytes).map_err(|_| MerkleTreeError::SerializationError)?;
37        let mut u64_bytes = [0u8; 8];
38        u64_bytes.copy_from_slice(&bytes[96..]);
39        let stake = Stake::from_be_bytes(u64_bytes);
40        Ok(MerkleTreeConcatenationLeaf(pk, stake))
41    }
42}
43
44impl From<MerkleTreeConcatenationLeaf> for (VerificationKey, Stake) {
45    fn from(leaf: MerkleTreeConcatenationLeaf) -> (VerificationKey, Stake) {
46        (leaf.0, leaf.1)
47    }
48}
49
50impl PartialOrd for MerkleTreeConcatenationLeaf {
51    /// Ordering of MT Values.
52    ///
53    /// First we order by stake, then by key. By having this ordering,
54    /// we have the players with higher stake close together,
55    /// meaning that the probability of having several signatures in the same side of the tree, is higher.
56    /// This allows us to produce a more efficient batch opening of the merkle tree.
57    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
58        Some(std::cmp::Ord::cmp(self, other))
59    }
60}
61
62impl Ord for MerkleTreeConcatenationLeaf {
63    fn cmp(&self, other: &Self) -> Ordering {
64        self.1.cmp(&other.1).then(self.0.cmp(&other.0))
65    }
