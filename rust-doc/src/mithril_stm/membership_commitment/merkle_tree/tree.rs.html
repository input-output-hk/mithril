18/// Tree of hashes, providing a commitment of data and its ordering.
19#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
20pub struct MerkleTree<D: Digest, L: MerkleTreeLeaf> {
21    /// The nodes are stored in an array heap:
22    /// * `nodes[0]` is the root,
23    /// * the parent of `nodes[i]` is `nodes[(i-1)/2]`
24    /// * the children of `nodes[i]` are `{nodes[2i + 1], nodes[2i + 2]}`
25    /// * All nodes have size `Output<D>::output_size()`, even leafs (which are hashed before committing them).
26    nodes: Vec<Vec<u8>>,
27    /// The leaves begin at `nodes[leaf_off]`.
28    leaf_off: usize,
29    /// Number of leaves cached in the merkle tree.
30    n: usize,
31    /// Phantom type to link the tree with its hasher
32    hasher: PhantomData<D>,
33    /// Phantom type to link the tree with its leaves
34    leaves: PhantomData<L>,
35}
36
37impl<D: Digest + FixedOutput, L: MerkleTreeLeaf> MerkleTree<D, L> {
38    /// Provided a non-empty list of leaves, `create` generates its corresponding `MerkleTree`.
39    pub(crate) fn new(leaves: &[L]) -> MerkleTree<D, L> {
40        let n = leaves.len();
41        assert!(n > 0, "MerkleTree::create() called with no leaves");
42
43        let num_nodes = n + n.next_power_of_two() - 1;
44
45        let mut nodes = vec![vec![0u8]; num_nodes];
46
47        for i in 0..leaves.len() {
48            nodes[num_nodes - n + i] = D::digest(leaves[i].as_bytes_for_merkle_tree()).to_vec();
49        }
50
51        let z = D::digest([0u8]).to_vec();
52        for i in (0..num_nodes - n).rev() {
53            let left = if left_child(i) < num_nodes {
54                &nodes[left_child(i)]
55            } else {
56                &z
57            };
58            let right = if right_child(i) < num_nodes {
59                &nodes[right_child(i)]
60            } else {
61                &z
62            };
63            nodes[i] = D::new().chain_update(left).chain_update(right).finalize().to_vec();
64        }
65
66        Self {
67            nodes,
68            n,
69            leaf_off: num_nodes - n,
70            hasher: PhantomData,
71            leaves: PhantomData,
72        }
73    }
74
75    /// Return the index of the leaf.
76    fn get_leaf_index(&self, i: usize) -> usize {
77        self.leaf_off + i
78    }
79
80    /// Convert merkle tree to a batch compatible commitment.
81    /// This function simply returns the root and the number of leaves in the tree.
82    pub(crate) fn to_merkle_tree_batch_commitment(&self) -> MerkleTreeBatchCommitment<D, L> {
83        MerkleTreeBatchCommitment::new(self.nodes[0].clone(), self.n)
84    }

86    /// Get a path for a batch of leaves. The indices must be ordered. We use the Octopus algorithm to
87    /// avoid redundancy with nodes in the path. Let `x1, . . . , xk` be the indices of elements we
88    /// want to produce an opening for. The algorithm takes as input `x1, . . ., xk`, and  proceeds as follows:
89    /// 1. Initialise the proof vector, `proof = []`.
90    /// 2. Given an input vector `v = v1, . . .,vl`, if `v.len() == 1`, return `proof`, else, continue.
91    /// 3. Map each `vi` to the corresponding number of the leaf (by adding the offset).
92    /// 4. Initialise a new empty vector `p = []`. Next, iterate over each element `vi`
93    ///    a. Append the parent of `vi` to `p`
94    ///    b. Compute the sibling, `si` of `vi`
95    ///    c. If `si == v(i+1)` then do nothing, and skip step four for `v(i+1)`. Else append `si` to `proof`
96    /// 5. Iterate from step 2 with input vector `p`
97    ///
98    /// # Panics
99    /// If the indices provided are out of bounds (higher than the number of elements
100    /// committed in the `MerkleTree`) or are not ordered, the function fails.
101    // todo: Update doc.
102    pub(crate) fn compute_merkle_tree_batch_path(&self, indices: Vec<usize>) -> MerkleBatchPath<D>
103    where
104        D: FixedOutput,
105    {
106        assert!(
107            !indices.is_empty(),
108            "get_batched_path() called with no indices"
109        );
110        for i in &indices {
111            assert!(
112                i < &self.n,
113                "Proof index out of bounds: asked for {} out of {}",
114                i,
115                self.n
116            );
117        }
118
119        let mut ordered_indices: Vec<usize> = indices.clone();
120        ordered_indices.sort_unstable();
121
122        assert_eq!(ordered_indices, indices, "Indices should be ordered");
123
124        ordered_indices = ordered_indices.into_iter().map(|i| self.get_leaf_index(i)).collect();
125
126        let mut idx = ordered_indices[0];
127        let mut proof = Vec::new();
128
129        while idx > 0 {
130            let mut new_indices = Vec::with_capacity(ordered_indices.len());
131            let mut i = 0;
132            idx = parent(idx);
133            while i < ordered_indices.len() {
134                new_indices.push(parent(ordered_indices[i]));
135                let sibling = sibling(ordered_indices[i]);
136                if i < ordered_indices.len() - 1 && ordered_indices[i + 1] == sibling {
137                    i += 1;
138                } else if sibling < self.nodes.len() {
139                    proof.push(self.nodes[sibling].clone());
140                }
141                i += 1;
142            }
143            ordered_indices.clone_from(&new_indices);
144        }
145
146        MerkleBatchPath::new(proof, indices)
147    }

154    /// Try to convert a byte string into a `MerkleTree`.
155    /// # Error
156    /// It returns error if conversion fails.
157    pub fn from_bytes(bytes: &[u8]) -> StmResult<Self> {
158        codec::from_versioned_bytes(bytes, Self::from_bytes_legacy)
159    }

161    /// Try to convert a byte string into a `MerkleTree` using the legacy format.
162    /// # Layout
163    /// * Number of leaves committed in the Merkle Tree (as u64)
164    /// * All nodes of the merkle tree (starting with the root)
165    fn from_bytes_legacy(bytes: &[u8]) -> StmResult<Self> {
166        let mut u64_bytes = [0u8; 8];
167        u64_bytes.copy_from_slice(bytes.get(..8).ok_or(MerkleTreeError::SerializationError)?);
168        let n = usize::try_from(u64::from_be_bytes(u64_bytes))
169            .map_err(|_| MerkleTreeError::SerializationError)?;
170        let num_nodes = n + n.next_power_of_two() - 1;
171        let mut nodes = Vec::with_capacity(num_nodes);
172        for i in 0..num_nodes {
173            nodes.push(
174                bytes
175                    .get(
176                        8 + i * <D as Digest>::output_size()
177                            ..8 + (i + 1) * <D as Digest>::output_size(),
178                    )
179                    .ok_or(MerkleTreeError::SerializationError)?
180                    .to_vec(),
181            );
182        }
183        Ok(Self {
184            nodes,
185            leaf_off: num_nodes - n,
186            n,
187            hasher: PhantomData,
188            leaves: PhantomData,
189        })
190    }

203    /// Get a path (hashes of siblings of the path to the root node)
204    /// for the `i`th value stored in the tree.
205    /// Requires `i < self.n`
206    pub(crate) fn compute_merkle_tree_path(&self, i: usize) -> MerklePath<D> {
207        assert!(
208            i < self.n,
209            "Proof index out of bounds: asked for {} out of {}",
210            i,
211            self.n
212        );
213        let mut idx = self.get_leaf_index(i);
214        let mut proof = Vec::new();
215
216        while idx > 0 {
217            let h = if sibling(idx) < self.nodes.len() {
218                self.nodes[sibling(idx)].clone()
219            } else {
220                D::digest([0u8]).to_vec()
221            };
222            proof.push(h);
223            idx = parent(idx);
224        }
225
226        MerklePath::new(proof, i)
227    }
228
229    #[cfg(feature = "future_snark")]
230    /// Get a path (hashes of siblings of the path to the root node)
231    /// for the `i`th value stored in the tree and pad it to reach the
232    /// given length.
233    /// The padding is done by adding values of `0` to the path in the form
234    /// of `[0u8; hash_output_size]`. Since finding the preimage is hard, this allows us
235    /// to use this value to detect when the "real" path ends.
236    /// Requires `i < self.n`
237    pub(crate) fn compute_merkle_tree_path_fixed_length(
238        &self,
239        i: usize,
240        path_length: u32,
241    ) -> MerklePath<D> {
242        let mut path = self.compute_merkle_tree_path(i);
243
244        while path.values.len() < path_length as usize {
245            path.values.push(vec![0u8; <D as Digest>::output_size()]);
246        }
247
248        path
249    }

610            // Elements added by padding must be zero vectors of the hash output size
611            let leaves = make_leaves(4); // natural depth = 2
612            let tree = MerkleTree::<SnarkHash, MerkleTreeSnarkLeaf>::new(&leaves);
613            let natural_len = tree.compute_merkle_tree_path(0).values.len();
614            let padded = tree.compute_merkle_tree_path_fixed_length(0, 5);
615
616            let expected_zero = vec![0u8; <SnarkHash as Digest>::output_size()];
617            for pad_elem in &padded.values[natural_len..] {
