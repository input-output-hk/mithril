22/// `SnarkProof` contains a proof generated by the certificate circuit represented as a vector of
23/// bytes. It can be verified using verifier parameters derived from the srs, the circuit verification key
24/// and the public inputs which are all public values
25#[derive(Debug, Clone, Serialize, Deserialize)]
26pub struct SnarkProof<D: MembershipDigest> {
27    /// Raw proof bytes produced by the SNARK prover.
28    circuit_proof: Vec<u8>,
29    // The protocol parameters
30    // Note: this field is temporary and will be removed once the circuit setup is stable
31    params: Parameters,
32    /// The maximum Merkle tree depth
33    // Note: this field is temporary and will be removed once the circuit setup is stable
34    merkle_tree_depth: u32,
35    /// Circuit verification key
36    circuit_verification_key: CircuitVerificationKey,
37    /// Phantom data
38    #[serde(skip)]
39    phantom: PhantomData<D>,
40}
41
42impl<D: MembershipDigest> SnarkProof<D> {
43    /// Tries to generate a new SnarkProof given as input a vector of bytes representing a proof,
44    /// protocol parameters and a merkle tree depth
45    pub fn try_new(
46        circuit_proof: Vec<u8>,
47        params: Parameters,
48        merkle_tree_depth: u32,
49    ) -> StmResult<Self> {
50        let snark_setup = SnarkSetup::try_new(&params, merkle_tree_depth)?;
51        Ok(Self {
52            circuit_proof,
53            circuit_verification_key: CircuitVerificationKey::new(snark_setup.verification_key),
54            params,
55            merkle_tree_depth,
56            phantom: PhantomData,
57        })
58    }
59
60    /// Verify a SNARK proof given a message, an aggregate verification key for snark
61    /// and a circuit verification key
62    ///
63    /// For now, we generate a temporary SnarkSetup inside the function to get access
64    /// to the srs and generate the verifier parameters. This part will be removed once
65    /// the circuit is stable and the srs is stored and available.
66    ///
67    /// This proof verifies that there exists a witness such that:
68    /// * each signatures in the witness is valid for the given message
69    /// * each signatures has a winning lottery index
70    /// * each leaf in the witness has a valid merkle path
71    /// * each lottery index is unique and fits the parameters
72    /// * enough lottery index are valid to meet the parameters requirements
73    ///
74    pub fn verify(
75        &self,
76        message: &[u8],
77        aggregate_verification_key_for_snark: &AggregateVerificationKeyForSnark<D>,
78    ) -> StmResult<()> {
79        let snark_setup = SnarkSetup::try_new(&self.params, MERKLE_TREE_DEPTH_FOR_SNARK)?;
80
81        let merkle_root = &aggregate_verification_key_for_snark.get_merkle_tree_commitment().root;
82        let proof_message = build_snark_message(merkle_root, message)?;
83        let proof_instance = (proof_message[0].into(), proof_message[1].into());
84
85        let verify_result = zk::verify::<StmCircuit, PoseidonState<CircuitBase>>(
86            &snark_setup.srs.verifier_params(),
87            &self.circuit_verification_key.get_midnight_vk(),
88            &proof_instance,
89            None,
90            &self.circuit_proof,
91        );
92
93        verify_result.map_err(|_| SnarkError::VerifyProofFail.into())
94    }

96    /// Converts a SnarkProof to CBOR bytes with a version prefix.
97    pub fn to_bytes(&self) -> StmResult<Vec<u8>> {
98        codec::to_cbor_bytes(self)
99    }

101    /// Deserialise a proof from bytes.
102    ///
103    /// Supports both the new versioned CBOR format and the legacy fixed-offset format.
104    pub fn from_bytes(bytes: &[u8]) -> StmResult<Self> {
105        codec::from_versioned_bytes(bytes, Self::from_bytes_legacy)
106    }
107
108    /// Deserialise a proof from the legacy fixed-offset byte format.
109    ///
110    /// # Layout
111    /// * Parameters (24 bytes, big-endian m/k/phi_f)
112    /// * Merkle tree depth (u32, little-endian)
113    /// * Circuit proof length (u64, little-endian)
114    /// * Circuit proof bytes
115    /// * Circuit verification key bytes (remaining)
116    fn from_bytes_legacy(bytes: &[u8]) -> StmResult<Self> {
117        // Legacy Parameters format is exactly 24 bytes (m: u64 + k: u64 + phi_f: f64, big-endian).
118        let params =
119            Parameters::from_bytes(bytes.get(0..24).ok_or(SnarkError::SerializationError)?)?;
120
121        let merkle_tree_depth = u32::from_le_bytes(
122            bytes.get(24..28).ok_or(SnarkError::SerializationError)?.try_into()?,
123        );
124
125        let circuit_proof_length = u64::from_le_bytes(
126            bytes.get(28..36).ok_or(SnarkError::SerializationError)?.try_into()?,
127        ) as usize;
128        let circuit_proof = bytes
129            .get(36..(circuit_proof_length + 36))
130            .ok_or(SnarkError::SerializationError)?
131            .to_vec();
132
133        let circuit_vk_bytes = bytes
134            .get((circuit_proof_length + 36)..)
135            .ok_or(SnarkError::SerializationError)?;
136        let circuit_verification_key = CircuitVerificationKey::from_bytes(circuit_vk_bytes)?;
137
138        Ok(Self {
139            circuit_proof,
140            params,
141            merkle_tree_depth,
142            circuit_verification_key,
143            phantom: PhantomData,
144        })
145    }

199impl<R: RngCore + CryptoRng> SnarkProver<R> {
200    /// Aggregate a set of single signatures into a SNARK proof.
201    ///
202    /// Prepares the prover input (public instance and per-index witness) from the collected
203    /// single signatures, then runs the SNARK prover.
204    ///
205    /// Returns an error if fewer than `k` valid signatures are available or if proof generation fails.
206    pub fn aggregate_signatures<D: MembershipDigest>(
207        &mut self,
208        clerk: &SnarkClerk,
209        signatures: &[SingleSignature],
210        message: &[u8],
211    ) -> StmResult<SnarkProof<D>> {
212        let snark_prover_input =
213            SnarkProverInput::prepare_prover_input::<D>(clerk, signatures, message)
214                .with_context(|| "Failed to prepare SNARK prover input")?;
215        let instance = snark_prover_input.get_instance();
216        let witness = snark_prover_input.into_witness();
217
218        let circuit_proof = zk::prove::<StmCircuit, PoseidonState<CircuitBase>>(
219            &self.setup.srs,
220            &self.setup.proving_key,
221            &self.setup.circuit,
222            &instance,
223            witness,
224            &mut self.rng,
225        )
226        .with_context(|| "SNARK proof generation failed")?;
227
228        Ok(SnarkProof {
229            circuit_proof,
230            circuit_verification_key: CircuitVerificationKey::new(
231                self.setup.verification_key.clone(),
232            ),
233            params: clerk.parameters,
234            merkle_tree_depth: clerk
235                .closed_key_registration
236                .number_of_registered_parties()
237                .next_power_of_two()
238                .trailing_zeros(),
239            phantom: PhantomData,
240        })
241    }
242}

273    fn setup_signers_and_clerk(
274        params: Parameters,
275        nparties: usize,
276        rng: &mut ChaCha20Rng,
277    ) -> (Vec<Signer<D>>, SnarkClerk) {
278        let mut key_reg = KeyRegistration::initialize();
279        let mut initializers = Vec::with_capacity(nparties);
280
281        for _ in 0..nparties {
282            let init = Initializer::new(params, 1, rng);
283            key_reg.register_by_entry(&init.clone().try_into().unwrap()).unwrap();
284            initializers.push(init);
285        }
286
287        let closed_reg = key_reg.close_registration(&params).unwrap();
288        let signers: Vec<Signer<D>> = initializers
289            .into_iter()
290            .map(|init| init.try_create_signer::<D>(&closed_reg).unwrap())
291            .collect();
292
293        let clerk = SnarkClerk::new_clerk_from_signer(&signers[0]);
294        (signers, clerk)
295    }
