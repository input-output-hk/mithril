22    /// Computes the lottery target value for SNARK proof system as a base field element.
23    ///
24    /// The target value determines the probability of winning the lottery based on the
25    /// participant's stake relative to the total stake. A higher stake results in a higher
26    /// target value, increasing the probability of eligibility.
27    ///
28    /// The target value is computed using the following formula, we need to check that:
29    /// signer_lottery_hash < p * (1 - (1 - phi_f)^w)
30    /// where p is the modulus of the field, phi_f is the protocol security parameter constant comprised in ]0,1],
31    /// w = stake/total_stake and signer_lottery_hash is the hash computed using the
32    /// signer's signature and the index tested for the lottery.
33    /// Since the modulus is a 255 bits number, we need to compute (1 - (1 - phi_f)^w) with enough precision
34    /// to maintain the lottery functional, i.e. have different targets for different stakes
35    /// and maintain the same order, however close they are.
36    /// In addition, we need this computation to be deterministic as it will be computed by all signers to create
37    /// the merkle_tree. The output should be the same no matter where it is computed (i.e. different config, OS, ...).
38    ///
39    /// In order to do that we change the expression:
40    /// 1 - (1 - phi_f)^w = 1 - exp(w * ln(1 - phi_f))
41    /// and we use Taylor expansion to approximate the exponential and natural logarithm functions to a given precision.
42    ///
43    /// Once the precise expression obtained, we can compute the target value as:
44    /// target = floor(p * (1 - exp(w * ln(1 - phi_f))))
45    ///
46    /// Input: (ln(1 - phi_f) approximation, stake of the signer, total_stake)
47    ///
48    /// Output: the lottery target value used for the lottery eligibility
49    pub fn compute_target_value(ln_one_minus_phi_f: &Ratio<BigInt>, stake: Stake, total_stake: Stake) -> LotteryTargetValue {
50        // It is safe to use .expect() as the value used is a constant so the creation of
51        // the BigInt will never fail
52        let modulus = BigInt::from_str_radix(
53            JUBJUB_BASE_FIELD_MODULUS,
54            16,
55        )
56        .expect("Hardcoded modulus of the Jubjub base field hex string is valid");
57
58        let stake_ratio = Ratio::new_raw(BigInt::from(stake), BigInt::from(total_stake));
59
60        let exp_ln_one_minus_phi_f_stake_ratio = compute_exponential_taylor_expansion(ln_one_minus_phi_f, &stake_ratio, TAYLOR_EXPANSION_ITERATIONS);
61
62        let modulus_ratio = Ratio::from(modulus);
63        let target_as_ratio = modulus_ratio.clone() - modulus_ratio * exp_ln_one_minus_phi_f_stake_ratio;
64
65        // Euclidean division
66        let (target_as_int, _) = target_as_ratio.numer().div_rem(target_as_ratio.denom());
67
68        let (_, mut bytes) = target_as_int.to_bytes_le();
69        bytes.resize(32, 0);
70        // It is safe to use .expect() as the value resulting from the computation is always lower than
71        // the Jubjub modulus
72        BaseFieldElement::from_bytes(&bytes).expect("Input bytes are always lower than the Jubjub modulus hence canonical.")
73    }

75    /// Computes a Taylor expansion of the exponential exp(c*w) up to the (N-1)th term
76    /// where N corresponds to the number of iterations
77    /// exp(c*w) = 1 + c*w + (c*w)^2/2! + (c*w)^3/3! + ... + (c*w)^{N-1}/(N-1)! + O((c*w)^N)
78    /// We want to stop when the next term is less than our precision target epsilon
79    /// Hence we stop when |(c*w)^N / N!| < epsilon
80    /// We can check instead (c * w)^N < epsilon which gives us the bound N < log(epsilon) / log(|c*w|)
81    /// Setting epsilon to a specific precision gives use the number of iterations we need to do
82    ///
83    /// Since the value c * w is a float between 0 and 1, it is expressed as a Ratio of BigInt and the exponential
84    /// approximation is adapted to that form
85    pub fn compute_exponential_taylor_expansion(c: &Ratio<BigInt>, w: &Ratio<BigInt>, iterations: usize) -> Ratio<BigInt> {
86        let cw = c * w;
87        let (num, denom, _) = exponential_approximation(0, iterations, cw.numer(), cw.denom());
88
89        Ratio::new_raw(num, denom)
90    }

93    /// Function that computes an approximation of exp(a/b) using a binomial splitting
94    /// to compute the taylor expansion terms between first_term and last_term,
95    /// i.e. between (a/b)^first_term * (1/first_term!) and (a/b)^last_term * (1/last_term!)
96    ///
97    /// We have that exp(a/b) = 1 + a/b + (a/b)^2/2! + (a/b)^3/3! + ... + (a/b)^{N-1}/(N-1)! + O((a/b)^N)
98    /// This function splits this computation in the middle and recursively compute each side by finding the quotient
99    /// X/Y that equals the sum of the terms in the split. Those terms are recursively combined to obtain the final
100    /// approximation of exp(a/b)
101    pub fn exponential_approximation(first_term: usize, last_term: usize, a: &BigInt, b: &BigInt) -> (BigInt, BigInt, BigInt) {
102        if last_term - first_term == 1 {
103            if first_term == 0 {
104                return (BigInt::one(), BigInt::one(), BigInt::one());
105            }
106            return (a.clone(), b * BigInt::from(first_term), a.clone());
107        }
108
109        let middle = (first_term + last_term) / 2;
110        // Computes the terms to the left of the middle
111        let (numerator_left, denominator_left, auxiliary_value_left) = exponential_approximation(first_term, middle, a, b);
112        // Computes the terms to the right of the middle
113        let (numerator_right, denominator_right, auxiliary_value_right) = exponential_approximation(middle, last_term, a, b);
114
115        let numerator = &numerator_left * &denominator_right + &auxiliary_value_left * &numerator_right;
116        let denominator = &denominator_left * &denominator_right;
117        let auxiliary_value = auxiliary_value_left * auxiliary_value_right;
118
119        // returns the numerator and denominator of the computed terms
120        // as well as an additional value to help with the rest of the computation
121        (numerator, denominator, auxiliary_value)
122    }

124    /// Function that computes an approximation of ln(1 - a/b) using a taylor expansion
125    /// for a given number N of iterations
126    /// ln(1 - a/b) = -a/b - ((a/b)^2)/2) - ((a/b)^3)/3) - ... - ((a/b)^(N))/N) + o((a/b)^(N+1))
127    ///
128    /// It performs a straighforward for loop in a naive way updating the numerator and denominator
129    /// every loop and the accumulator using the new values. The numerator stores the power of a and
130    /// the denominator the power of b.
131    #[allow(dead_code)]
132    fn ln_1p_taylor_expansion(iterations: usize, a: &BigInt, b: &BigInt) -> Ratio<BigInt> {
133        let mut numerator = a.clone();
134        let mut denominator = b.clone();
135        let mut accumulator = Ratio::new_raw(a.clone(),b.clone());
136        for i in 2..(iterations + 1) {
137            numerator *= a;
138            denominator *= b;
139            accumulator += Ratio::new_raw(numerator.clone(), denominator.clone() * i);
140        }
141
142        -accumulator
143    }

148    /// Verifies if a lottery index is eligible based on the signature and target value.
149    ///
150    /// This function checks whether a given index wins the lottery by computing an
151    /// evaluation value from the signature's commitment point and the index, then
152    /// comparing it against the target value. An index is eligible if its
153    /// evaluation value is less than or equal to the target.
154    ///
155    /// The evaluation is computed as: `ev = Poseidon(prefix, commitment_point_x, commitment_point_y, index)`
156    /// where `(commitment_point_x, commitment_point_y)` are the coordinates of the signature's commitment point.
157    pub fn verify_lottery_eligibility(
158        signature: &UniqueSchnorrSignature,
159        lottery_index: LotteryIndex,
160        m: u64,
161        prefix: BaseFieldElement,
162        target: LotteryTargetValue,
163    ) -> StmResult<()> {
164        if lottery_index > m {
165            return Err(SignatureError::IndexBoundFailed(lottery_index, m).into());
166        }
167
168        let lottery_index_as_base_field_element = BaseFieldElement::from(lottery_index);
169        let (commitment_point_x, commitment_point_y) = signature.commitment_point.get_coordinates();
170        let lottery_evaluation = compute_poseidon_digest(&[prefix, commitment_point_x, commitment_point_y, lottery_index_as_base_field_element]);
171
172        // check if ev <= target
173        if lottery_evaluation > target {
174            return Err(SignatureError::LotteryLost.into());
175        }
176
177        Ok(())
178    }

226        let split = 10;

229        let phi_full = compute_exponential_taylor_expansion(
