9/// Aggregate verification key of the snark proof system.
10#[derive(Debug, Clone, Serialize, Deserialize, Eq, PartialEq)]
11pub struct AggregateVerificationKeyForSnark<D: MembershipDigest> {
12    merkle_tree_commitment: MerkleTreeCommitment<D::SnarkHash, MerkleTreeSnarkLeaf>,
13}
14
15impl<D: MembershipDigest> AggregateVerificationKeyForSnark<D> {
16    /// Get the Merkle tree commitment.
17    pub(crate) fn get_merkle_tree_commitment(
18        &self,
19    ) -> &MerkleTreeCommitment<D::SnarkHash, MerkleTreeSnarkLeaf> {
20        &self.merkle_tree_commitment
21    }
22}
23
24impl<D: MembershipDigest> From<&ClosedKeyRegistration> for AggregateVerificationKeyForSnark<D> {
25    fn from(reg: &ClosedKeyRegistration) -> Self {
26        let key_registration_commitment_for_snark =
27            reg.to_merkle_tree::<D::SnarkHash, RegistrationEntryForSnark>();
28        Self {
29            merkle_tree_commitment: key_registration_commitment_for_snark
30                .to_merkle_tree_commitment(),
31        }
32    }
33}
