14/// `ConcatenationProof` uses the "concatenation" proving system (as described in Section 4.3 of the original paper.)
15/// This means that the aggregated signature contains a vector with all individual signatures.
16/// BatchPath is also a part of the aggregate signature which covers path for all signatures.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18#[serde(bound(
19    serialize = "MerkleBatchPath<D::ConcatenationHash>: Serialize",
20    deserialize = "MerkleBatchPath<D::ConcatenationHash>: Deserialize<'de>"
21))]
22pub struct ConcatenationProof<D: MembershipDigest> {
23    pub(crate) signatures: Vec<SingleSignatureWithRegisteredParty>,
24    /// The list of unique merkle tree nodes that covers path for all signatures.
25    pub batch_proof: MerkleBatchPath<D::ConcatenationHash>,
26}
27
28impl<D: MembershipDigest> ConcatenationProof<D> {
29    /// Aggregate a set of signatures for their corresponding indices.
30    ///
31    /// This function first deduplicates the repeated signatures, and if there are enough signatures, it collects the merkle tree indexes of unique signatures.
32    /// The list of merkle tree indexes is used to create a batch proof, to prove that all signatures are from eligible signers.
33    ///
34    /// It returns an instance of `ConcatenationProof`.
35    pub fn aggregate_signatures(
36        clerk: &ConcatenationClerk,
37        sigs: &[SingleSignature],
38        msg: &[u8],
39    ) -> StmResult<ConcatenationProof<D>> {
40        let sig_reg_list = sigs
41            .iter()
42            .map(|sig| {
43                clerk
44                    .closed_key_registration
45                    .get_registration_entry_for_index(&sig.signer_index)
46                    .map(|reg_party| SingleSignatureWithRegisteredParty {
47                        sig: sig.clone(),
48                        reg_party,
49                    })
50            })
51            .collect::<Result<Vec<_>, _>>()?;
52
53        let avk: AggregateVerificationKeyForConcatenation<D> =
54            clerk.compute_aggregate_verification_key_for_concatenation();
55        let mut unique_sigs = ConcatenationClerk::select_valid_signatures_for_k_indices(
56            &clerk.parameters,
57            msg,
58            &sig_reg_list,
59            &avk,
60        )
61        .with_context(
62            || "Failed to aggregate unique signatures during selection for the k indices.",
63        )?;
64
65        unique_sigs.sort_unstable();
66
67        let mt_index_list = unique_sigs
68            .iter()
69            .map(|sig_reg| sig_reg.sig.signer_index as usize)
70            .collect::<Vec<usize>>();
71
72        let batch_proof = clerk
73            .closed_key_registration
74            .to_merkle_tree::<D::ConcatenationHash, RegistrationEntryForConcatenation>()
75            .compute_merkle_tree_batch_path(mt_index_list);
76
77        Ok(Self {
78            signatures: unique_sigs,
79            batch_proof,
80        })
81    }
82
83    /// Verify all checks from signatures, except for the signature verification itself.
84    ///
85    /// Checks that each signature contains only valid indices, the lottery is indeed won by each
86    /// one of them. Validates whether all collected indices are unique and at least `k` are present.
87    /// It collects leaves from signatures and checks the batch proof.
88    /// After batch proof is checked, it collects and returns the signatures and
89    /// verification keys to be used by aggregate verification.
90    fn preliminary_verify(
91        &self,
92        msg: &[u8],
93        avk: &AggregateVerificationKeyForConcatenation<D>,
94        parameters: &Parameters,
95    ) -> StmResult<(Vec<BlsSignature>, Vec<VerificationKeyForConcatenation>)> {
96        let msgp = avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg);
97
98        let mut nr_indices = 0;
99        let mut unique_indices = HashSet::new();
100
101        for sig_reg in self.signatures.clone() {
102            sig_reg
103                .sig
104                .check_indices(
105                    parameters,
106                    &sig_reg.reg_party.get_stake(),
107                    &msgp,
108                    &avk.get_total_stake(),
109                )
110                .with_context(|| "Preliminary verification for basic verifier failed.")?;
111            for &index in &sig_reg.sig.get_concatenation_signature_indices() {
112                unique_indices.insert(index);
113                nr_indices += 1;
114            }
115        }
116
117        if nr_indices != unique_indices.len() {
118            return Err(anyhow!(AggregationError::IndexNotUnique));
119        }
120        if (nr_indices as u64) < parameters.k {
121            return Err(anyhow!(AggregationError::NotEnoughSignatures(
122                nr_indices as u64,
123                parameters.k
124            )));
125        }
126
127        let leaves = self
128            .signatures
129            .iter()
130            .filter_map(|r| r.reg_party.into())
131            .collect::<Vec<RegistrationEntryForConcatenation>>();
132
133        avk.get_merkle_tree_batch_commitment()
134            .verify_leaves_membership_from_batch_path(&leaves, &self.batch_proof)
135            .with_context(|| "Batch proof is invalid in preliminary verification.")?;
136
137        Ok(self.collect_signatures_verification_keys())
138    }
139
140    /// Verify concatenation proof, by checking that
141    /// * each signature contains only valid indices,
142    /// * the lottery is indeed won by each one of them,
143    /// * the merkle tree path is valid,
144    /// * the aggregate signature validates with respect to the aggregate verification key
145    ///   (aggregation is computed using functions `MSP.BKey` and `MSP.BSig` as described in Section 2.4 of the paper).
146    pub fn verify(
147        &self,
148        msg: &[u8],
149        avk: &AggregateVerificationKeyForConcatenation<D>,
150        parameters: &Parameters,
151    ) -> StmResult<()> {
152        let msgp = avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg);
153        let (sigs, vks) = self
154            .preliminary_verify(msg, avk, parameters)
155            .with_context(|| "Aggregate signature verification failed")?;
156
157        BlsSignature::verify_aggregate(msgp.as_slice(), &vks, &sigs)
158            .with_context(|| "Aggregate signature verification failed")?;
159        Ok(())
160    }
161
162    /// Batch verify a set of signatures, with different messages and avks.
163    pub fn batch_verify(
164        stm_signatures: &[Self],
165        msgs: &[Vec<u8>],
166        avks: &[AggregateVerificationKeyForConcatenation<D>],
167        parameters: &[Parameters],
168    ) -> StmResult<()> {
169        let batch_size = stm_signatures.len();
170        assert_eq!(
171            batch_size,
172            msgs.len(),
173            "Number of messages should correspond to size of the batch"
174        );
175        assert_eq!(
176            batch_size,
177            avks.len(),
178            "Number of avks should correspond to size of the batch"
179        );
180        assert_eq!(
181            batch_size,
182            parameters.len(),
183            "Number of parameters should correspond to size of the batch"
184        );
185
186        let mut aggr_sigs = Vec::with_capacity(batch_size);
187        let mut aggr_vks = Vec::with_capacity(batch_size);
188        for (idx, sig_group) in stm_signatures.iter().enumerate() {
189            sig_group.preliminary_verify(&msgs[idx], &avks[idx], &parameters[idx])?;
190            let grouped_sigs: Vec<BlsSignature> = sig_group
191                .signatures
192                .iter()
193                .map(|sig_reg| sig_reg.sig.get_concatenation_signature_sigma())
194                .collect();
195            let grouped_vks: Vec<VerificationKeyForConcatenation> = sig_group
196                .signatures
197                .iter()
198                .map(|sig_reg| sig_reg.reg_party.get_verification_key_for_concatenation())
199                .collect();
200
201            let (aggr_vk, aggr_sig) = BlsSignature::aggregate(&grouped_vks, &grouped_sigs).unwrap();
202            aggr_sigs.push(aggr_sig);
203            aggr_vks.push(aggr_vk);
204        }
205
206        let concat_msgs: Vec<Vec<u8>> = msgs
207            .iter()
208            .zip(avks.iter())
209            .map(|(msg, avk)| avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg))
210            .collect();
211
212        BlsSignature::batch_verify_aggregates(&concat_msgs, &aggr_vks, &aggr_sigs)?;
213        Ok(())
214    }

216    /// Convert concatenation proof to bytes
217    /// # Layout
218    /// * Number of the pairs of Signatures and Registered Parties (SigRegParty) (as u64)
219    /// * Pairs of Signatures and Registered Parties (prefixed with their size as u64)
220    /// * Batch proof
221    pub fn to_bytes(&self) -> Vec<u8> {
222        let mut out = Vec::new();
223        out.extend_from_slice(&u64::try_from(self.signatures.len()).unwrap().to_be_bytes());
224        for sig_reg in &self.signatures {
225            out.extend_from_slice(&u64::try_from(sig_reg.to_bytes().len()).unwrap().to_be_bytes());
226            out.extend_from_slice(&sig_reg.to_bytes());
227        }
228        let proof = &self.batch_proof;
229        out.extend_from_slice(&proof.to_bytes());
230
231        out
232    }
233
234    ///Extract a concatenation proof from a byte slice.
235    pub fn from_bytes(bytes: &[u8]) -> StmResult<ConcatenationProof<D>> {
236        let mut bytes_index = 0;
237
238        let mut u64_bytes = [0u8; 8];
239        u64_bytes.copy_from_slice(
240            bytes
241                .get(bytes_index..bytes_index + 8)
242                .ok_or(AggregateSignatureError::SerializationError)?,
243        );
244        let total_sigs = usize::try_from(u64::from_be_bytes(u64_bytes))
245            .map_err(|_| AggregateSignatureError::SerializationError)?;
246        bytes_index += 8;
247
248        let mut sig_reg_list = Vec::with_capacity(total_sigs);
249        for _ in 0..total_sigs {
250            u64_bytes.copy_from_slice(
251                bytes
252                    .get(bytes_index..bytes_index + 8)
253                    .ok_or(AggregateSignatureError::SerializationError)?,
254            );
255            let sig_reg_size = usize::try_from(u64::from_be_bytes(u64_bytes))
256                .map_err(|_| AggregateSignatureError::SerializationError)?;
257            let sig_reg = SingleSignatureWithRegisteredParty::from_bytes::<D>(
258                bytes
259                    .get(bytes_index + 8..bytes_index + 8 + sig_reg_size)
260                    .ok_or(AggregateSignatureError::SerializationError)?,
261            )?;
262            bytes_index += 8 + sig_reg_size;
263            sig_reg_list.push(sig_reg);
264        }
265
266        let batch_proof = MerkleBatchPath::from_bytes(
267            bytes
268                .get(bytes_index..)
269                .ok_or(AggregateSignatureError::SerializationError)?,
270        )?;
271
272        Ok(ConcatenationProof {
273            signatures: sig_reg_list,
274            batch_proof,
275        })
276    }

278    /// Collect and return `Vec<BlsSignature>, Vec<BlsVerificationKey>` which will be used
279    /// by the aggregate verification.
280    pub(crate) fn collect_signatures_verification_keys(
281        &self,
282    ) -> (Vec<BlsSignature>, Vec<VerificationKeyForConcatenation>) {
283        let sigs = self
284            .signatures
285            .iter()
286            .map(|sig_reg| sig_reg.sig.get_concatenation_signature_sigma())
287            .collect::<Vec<BlsSignature>>();
288        let vks = self
289            .signatures
290            .iter()
291            .map(|sig_reg| sig_reg.reg_party.get_verification_key_for_concatenation())
292            .collect::<Vec<VerificationKeyForConcatenation>>();
293
294        (sigs, vks)
295    }
