10/// The `ConcatenationClerk` is responsible for managing the proof system related to
11/// concatenation signatures.
12#[derive(Debug, Clone)]
13pub struct ConcatenationClerk {
14    /// The closed key registration associated with this clerk.
15    pub(crate) closed_key_registration: ClosedKeyRegistration,
16    /// Protocol parameters
17    pub(crate) parameters: Parameters,
18}

20impl ConcatenationClerk {
21    /// Create a new `ConcatenationClerk` from a closed registration instance.
22    pub fn new_clerk_from_closed_key_registration(
23        parameters: &Parameters,
24        closed_key_registration: &ClosedKeyRegistration,
25    ) -> Self {
26        Self {
27            parameters: *parameters,
28            closed_key_registration: closed_key_registration.clone(),
29        }
30    }
31
32    /// Create a `ConcatenationClerk` from a signer.
33    pub fn new_clerk_from_signer<D: MembershipDigest>(signer: &Signer<D>) -> Self {
34        Self {
35            parameters: signer.parameters,
36            closed_key_registration: signer.closed_key_registration.clone(),
37        }
38    }
39
40    /// Compute the `ConcatenationProofKey` related to the used registration.
41    pub fn compute_aggregate_verification_key_for_concatenation<D: MembershipDigest>(
42        &self,
43    ) -> AggregateVerificationKeyForConcatenation<D> {
44        AggregateVerificationKeyForConcatenation::from(&self.closed_key_registration)
45    }

57    /// Given a slice of `sig_reg_list`, this function returns a new list of `sig_reg_list` with only valid indices.
58    /// In case of conflict (having several signatures for the same index)
59    /// it selects the smallest signature (i.e. takes the signature with the smallest scalar).
60    /// The function selects at least `self.k` indexes.
61    ///  # Error
62    /// If there is no sufficient signatures, then the function fails.
63    // todo: We need to agree on a criteria to dedup (by default we use a BTreeMap that guarantees keys order)
64    // todo: not good, because it only removes index if there is a conflict (see benches)
65    pub fn select_valid_signatures_for_k_indices<D: MembershipDigest>(
66        params: &Parameters,
67        msg: &[u8],
68        sigs: &[SingleSignatureWithRegisteredParty],
69        avk: &AggregateVerificationKeyForConcatenation<D>,
70    ) -> StmResult<Vec<SingleSignatureWithRegisteredParty>> {
71        let mut sig_by_index: BTreeMap<LotteryIndex, &SingleSignatureWithRegisteredParty> =
72            BTreeMap::new();
73        let mut removal_idx_by_vk: HashMap<&SingleSignatureWithRegisteredParty, Vec<LotteryIndex>> =
74            HashMap::new();
75
76        for sig_reg in sigs.iter() {
77            if sig_reg
78                .sig
79                .concatenation_signature
80                .verify(
81                    params,
82                    &sig_reg.reg_party.get_verification_key_for_concatenation(),
83                    &sig_reg.reg_party.get_stake(),
84                    avk,
85                    msg,
86                )
87                .is_err()
88            {
89                continue;
90            }
91            for index in sig_reg.sig.get_concatenation_signature_indices().iter() {
92                let mut insert_this_sig = false;
93                if let Some(&previous_sig) = sig_by_index.get(index) {
94                    let sig_to_remove_index = if sig_reg.sig.get_concatenation_signature_sigma()
95                        < previous_sig.sig.get_concatenation_signature_sigma()
96                    {
97                        insert_this_sig = true;
98                        previous_sig
99                    } else {
100                        sig_reg
101                    };
102
103                    if let Some(indexes) = removal_idx_by_vk.get_mut(sig_to_remove_index) {
104                        indexes.push(*index);
105                    } else {
106                        removal_idx_by_vk.insert(sig_to_remove_index, vec![*index]);
107                    }
108                } else {
109                    insert_this_sig = true;
110                }
111
112                if insert_this_sig {
113                    sig_by_index.insert(*index, sig_reg);
114                }
115            }
116        }
117
118        let mut dedup_sigs: HashSet<SingleSignatureWithRegisteredParty> = HashSet::new();
119        let mut count: u64 = 0;
120
121        for (_, &sig_reg) in sig_by_index.iter() {
122            if dedup_sigs.contains(sig_reg) {
123                continue;
124            }
125            let mut deduped_sig = sig_reg.clone();
126            if let Some(indexes) = removal_idx_by_vk.get(sig_reg) {
127                let indices = deduped_sig
128                    .sig
129                    .get_concatenation_signature_indices()
130                    .into_iter()
131                    .filter(|i| !indexes.contains(i))
132                    .collect::<Vec<LotteryIndex>>();
133                deduped_sig.sig.set_concatenation_signature_indices(&indices);
134            }
135
136            let size: Result<u64, _> =
137                deduped_sig.sig.get_concatenation_signature_indices().len().try_into();
138            if let Ok(size) = size {
139                if dedup_sigs.contains(&deduped_sig) {
140                    panic!(
141                        "Invariant violation: duplicate signature encountered in deduplication set, which should not be possible."
142                    );
143                }
144                dedup_sigs.insert(deduped_sig);
145                count += size;
146
147                if count >= params.k {
148                    return Ok(dedup_sigs.into_iter().collect());
149                }
150            }
151        }
152        Err(anyhow!(AggregationError::NotEnoughSignatures(
153            count, params.k
154        )))
155    }

159mod tests {
160    use std::collections::HashSet;
161
162    use proptest::prelude::*;
163    use rand_chacha::ChaCha20Rng;
164    use rand_core::{RngCore, SeedableRng};
165
166    use crate::{
167        ClosedKeyRegistration, Initializer, KeyRegistration, MithrilMembershipDigest, Parameters,
168        SingleSignatureWithRegisteredParty, proof_system::ConcatenationClerk,
169    };
170
171    use super::AggregationError;
172
173    type D = MithrilMembershipDigest;
174
175    proptest! {
176        #![proptest_config(ProptestConfig::with_cases(50))]
177
178        #[test]
179        fn test_dedup(
180            seed in any::<[u8; 32]>(),
181            msg in any::<[u8;16]>(),
182            nparties in 1_usize..10,
183            m in 1_u64..20,
184            k in 1_u64..10,
185            phi_f in 0.1_f64..1.0,
186            num_invalid_sigs_per_party in 0_usize..3,
187        ) {
188            let params = Parameters { m, k, phi_f };
189            let mut rng = ChaCha20Rng::from_seed(seed);
190
191            // False messages
192            let mut false_msgs = Vec::new();
193            for _ in 0..num_invalid_sigs_per_party {
194                let mut false_msg = vec![0u8; 32];
195                rng.fill_bytes(&mut false_msg);
196                if false_msg == msg {
197                    false_msg[0] = msg[0].wrapping_add(1);
198                }
199                false_msgs.push(false_msg);
200            }
201
202            let mut key_registration = KeyRegistration::initialize();
203            let mut initializers = Vec::new();
204
205            for i in 0..nparties {
206                let stake = (i as u64 + 1) * 10;
207                let initializer = Initializer::new(params, stake, &mut rng);
208                key_registration.register_by_entry(&initializer.clone().into()).unwrap();
209                initializers.push(initializer);
210            }
211
212            let closed_registration: ClosedKeyRegistration = key_registration.close_registration();
213
214            let signers: Vec<_> = initializers
215                .into_iter()
216                .map(|init| init.try_create_signer::<MithrilMembershipDigest>(&closed_registration).unwrap())
217                .collect();
218
219            let clerk = ConcatenationClerk::new_clerk_from_signer(&signers[0]);
220            let avk = clerk.compute_aggregate_verification_key_for_concatenation::<D>();
221
222            let mut all_sigs = Vec::new();
223            for signer in &signers {
224                // Add invalid signatures
225                for false_msg in &false_msgs {
226                    if let Ok(sig) = signer.create_single_signature(false_msg) {
227                        all_sigs.push(sig);
228                    }
229                }
230
231                // Add valid signatures
232                if let Ok(sig) = signer.create_single_signature(&msg) {
233                    all_sigs.push(sig);
234                }
235            }
236
237            let sig_reg_list = all_sigs
238                .iter()
239                .map(|sig| SingleSignatureWithRegisteredParty {
240                    sig: sig.clone(),
241                    reg_party: clerk.closed_key_registration.get_registration_entry_for_index(&sig.signer_index).unwrap(),
242                })
243                .collect::<Vec<SingleSignatureWithRegisteredParty>>();
244
245            let dedup_result =
246                ConcatenationClerk::select_valid_signatures_for_k_indices(&params, &msg, &sig_reg_list, &avk);
247
248            match dedup_result {
249                Ok(valid_sigs) => {
250                    assert!(!valid_sigs.is_empty(), "Should have at least one valid signature");
251                    for passed_sigs in &valid_sigs {
252                        let signer = &signers[passed_sigs.sig.signer_index as usize];
253                        let verify_result = passed_sigs.sig.concatenation_signature.verify(
254                            &params,
255                            &signer.get_bls_verification_key(),
256                            &passed_sigs.reg_party.get_stake(),
257                            &avk,
258                            &msg,
259                        );
260                        assert!(verify_result.is_ok(), "All returned signatures should verify: {:?}", verify_result);
261                    }
262                    // If all signatures are valid, check that indices are unique and at least k
263                    // unique indices are collected across all signatures
264                    let mut unique_indices = HashSet::new();
265                    let mut count: u64 = 0;
266                    for sig in valid_sigs.iter() {
267                        for index in sig.sig.get_concatenation_signature_indices().iter() {
268                            count += 1;
269                            unique_indices.insert(*index);
270                        }
271                    }
272                    assert_eq!(unique_indices.len() as u64, count, "Indices should be unique");
273                    assert!(count >= k, "Should have at least k unique indices");
274                }
275                Err(error) => {
276                    assert!(
277                        matches!(
278                            error.downcast_ref::<AggregationError>(),
279                            Some(AggregationError::NotEnoughSignatures(..))
280                        ),
281                        "Expected NotEnoughSignatures, got: {:?}", error
282                    );
283                }
284            }
285        }
286    }
