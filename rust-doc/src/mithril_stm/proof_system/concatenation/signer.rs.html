12/// Concatenation Proof single signature generator.
13/// Contains the signer's stake and signing key besides the total stake of the closed registration,
14/// protocol parameters, and the Merkle Tree commitment of the key registration for concatenation
15/// proof.
16#[derive(Debug, Clone)]
17pub(crate) struct ConcatenationProofSigner<D: MembershipDigest> {
18    pub(crate) stake: Stake,
19    total_stake: Stake,
20    parameters: Parameters,
21    signing_key: BlsSigningKey,
22    verification_key: VerificationKeyForConcatenation,
23    key_registration_commitment:
24        MerkleTree<D::ConcatenationHash, RegistrationEntryForConcatenation>,
25}
26
27impl<D: MembershipDigest> ConcatenationProofSigner<D> {
28    /// Generates a new instance of `ConcatenationProofSigner`
29    pub fn new(
30        stake: Stake,
31        total_stake: Stake,
32        parameters: Parameters,
33        signing_key: BlsSigningKey,
34        verification_key: VerificationKeyForConcatenation,
35        key_registration_commitment: MerkleTree<
36            D::ConcatenationHash,
37            RegistrationEntryForConcatenation,
38        >,
39    ) -> Self {
40        Self {
41            stake,
42            total_stake,
43            parameters,
44            signing_key,
45            verification_key,
46            key_registration_commitment,
47        }
48    }
49
50    /// Generates a single signature for concatenation proof system.
51    /// First, concatenates the given message with the Merkle tree commitment. Then, generates a bls
52    /// signature for the `message_with_commitment`. Checks the lottery and collects the winning
53    /// indices. If there is no winning indices, returns an error. Otherwise, returns
54    /// `SingleSignatureForConcatenation`.
55    pub fn create_single_signature(
56        &self,
57        message: &[u8],
58    ) -> StmResult<SingleSignatureForConcatenation> {
59        let message_with_commitment = self
60            .key_registration_commitment
61            .to_merkle_tree_batch_commitment()
62            .concatenate_with_message(message);
63
64        let sigma = self.signing_key.sign(&message_with_commitment);
65        let indices = self.check_lottery(&message_with_commitment, &sigma);
66
67        if indices.is_empty() {
68            Err(anyhow!(SignatureError::LotteryLost))
69        } else {
70            Ok(SingleSignatureForConcatenation::new(sigma, indices))
71        }
72    }
73
74    /// Checks the lottery for given `message_with_commitment` and the `sigma`.
75    /// Returns a vector of winning indices.
76    pub fn check_lottery(&self, message_with_commitment: &[u8], sigma: &BlsSignature) -> Vec<u64> {
77        let mut indices = Vec::new();
78        for index in 0..self.parameters.m {
79            if is_lottery_won(
80                self.parameters.phi_f,
81                sigma.evaluate_dense_mapping(message_with_commitment, index),
82                self.stake,
83                self.total_stake,
84            ) {
85                indices.push(index);
86            }
87        }
88        indices
89    }
90
91    pub fn get_verification_key(&self) -> VerificationKeyForConcatenation {
92        self.verification_key
93    }
94}
