14/// `ConcatenationProof` uses the "concatenation" proving system (as described in Section 4.3 of the original paper.)
15/// This means that the aggregated signature contains a vector with all individual signatures.
16/// BatchPath is also a part of the aggregate signature which covers path for all signatures.
17#[derive(Debug, Clone, Serialize, Deserialize)]
18#[serde(bound(
19    serialize = "MerkleBatchPath<D>: Serialize",
20    deserialize = "MerkleBatchPath<D>: Deserialize<'de>"
21))]
22pub struct ConcatenationProof<D: Clone + Digest + FixedOutput> {
23    pub(crate) signatures: Vec<SingleSignatureWithRegisteredParty>,
24    /// The list of unique merkle tree nodes that covers path for all signatures.
25    pub batch_proof: MerkleBatchPath<D>,
26}
27
28impl<D: Clone + Digest + FixedOutput + Send + Sync> ConcatenationProof<D> {
29    /// Aggregate a set of signatures for their corresponding indices.
30    ///
31    /// This function first deduplicates the repeated signatures, and if there are enough signatures, it collects the merkle tree indexes of unique signatures.
32    /// The list of merkle tree indexes is used to create a batch proof, to prove that all signatures are from eligible signers.
33    ///
34    /// It returns an instance of `ConcatenationProof`.
35    pub fn aggregate_signatures(
36        clerk: &Clerk<D>,
37        sigs: &[SingleSignature],
38        msg: &[u8],
39    ) -> Result<ConcatenationProof<D>, AggregationError> {
40        let sig_reg_list = sigs
41            .iter()
42            .map(|sig| SingleSignatureWithRegisteredParty {
43                sig: sig.clone(),
44                reg_party: clerk.closed_reg.reg_parties[sig.signer_index as usize],
45            })
46            .collect::<Vec<SingleSignatureWithRegisteredParty>>();
47
48        let avk = AggregateVerificationKey::from(&clerk.closed_reg);
49        let msgp = avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg);
50        let mut unique_sigs = BasicVerifier::select_valid_signatures_for_k_indices(
51            &clerk.closed_reg.total_stake,
52            &clerk.params,
53            &msgp,
54            &sig_reg_list,
55        )?;
56
57        unique_sigs.sort_unstable();
58
59        let mt_index_list = unique_sigs
60            .iter()
61            .map(|sig_reg| sig_reg.sig.signer_index as usize)
62            .collect::<Vec<usize>>();
63
64        let batch_proof = clerk
65            .closed_reg
66            .merkle_tree
67            .compute_merkle_tree_batch_path(mt_index_list);
68
69        Ok(Self {
70            signatures: unique_sigs,
71            batch_proof,
72        })
73    }
74
75    /// Verify all checks from signatures, except for the signature verification itself.
76    ///
77    /// Indices and quorum are checked by `BasicVerifier::preliminary_verify` with `msgp`.
78    /// It collects leaves from signatures and checks the batch proof.
79    /// After batch proof is checked, it collects and returns the signatures and
80    /// verification keys to be used by aggregate verification.
81    fn preliminary_verify(
82        &self,
83        msg: &[u8],
84        avk: &AggregateVerificationKey<D>,
85        parameters: &Parameters,
86    ) -> Result<(Vec<BlsSignature>, Vec<BlsVerificationKey>), StmAggregateSignatureError<D>> {
87        let msgp = avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg);
88        BasicVerifier::preliminary_verify(
89            &avk.get_total_stake(),
90            &self.signatures,
91            parameters,
92            &msgp,
93        )?;
94
95        let leaves = self
96            .signatures
97            .iter()
98            .map(|r| r.reg_party)
99            .collect::<Vec<RegisteredParty>>();
100
101        avk.get_merkle_tree_batch_commitment()
102            .verify_leaves_membership_from_batch_path(&leaves, &self.batch_proof)?;
103
104        Ok(BasicVerifier::collect_signatures_verification_keys(
105            &self.signatures,
106        ))
107    }
108
109    /// Verify concatenation proof, by checking that
110    /// * each signature contains only valid indices,
111    /// * the lottery is indeed won by each one of them,
112    /// * the merkle tree path is valid,
113    /// * the aggregate signature validates with respect to the aggregate verification key
114    ///   (aggregation is computed using functions `MSP.BKey` and `MSP.BSig` as described in Section 2.4 of the paper).
115    pub fn verify(
116        &self,
117        msg: &[u8],
118        avk: &AggregateVerificationKey<D>,
119        parameters: &Parameters,
120    ) -> Result<(), StmAggregateSignatureError<D>> {
121        let msgp = avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg);
122        let (sigs, vks) = self.preliminary_verify(msg, avk, parameters)?;
123
124        BlsSignature::verify_aggregate(msgp.as_slice(), &vks, &sigs)?;
125        Ok(())
126    }
127
128    /// Batch verify a set of signatures, with different messages and avks.
129    pub fn batch_verify(
130        stm_signatures: &[Self],
131        msgs: &[Vec<u8>],
132        avks: &[AggregateVerificationKey<D>],
133        parameters: &[Parameters],
134    ) -> Result<(), StmAggregateSignatureError<D>> {
135        let batch_size = stm_signatures.len();
136        assert_eq!(
137            batch_size,
138            msgs.len(),
139            "Number of messages should correspond to size of the batch"
140        );
141        assert_eq!(
142            batch_size,
143            avks.len(),
144            "Number of avks should correspond to size of the batch"
145        );
146        assert_eq!(
147            batch_size,
148            parameters.len(),
149            "Number of parameters should correspond to size of the batch"
150        );
151
152        let mut aggr_sigs = Vec::with_capacity(batch_size);
153        let mut aggr_vks = Vec::with_capacity(batch_size);
154        for (idx, sig_group) in stm_signatures.iter().enumerate() {
155            sig_group.preliminary_verify(&msgs[idx], &avks[idx], &parameters[idx])?;
156            let grouped_sigs: Vec<BlsSignature> =
157                sig_group.signatures.iter().map(|sig_reg| sig_reg.sig.sigma).collect();
158            let grouped_vks: Vec<BlsVerificationKey> = sig_group
159                .signatures
160                .iter()
161                .map(|sig_reg| sig_reg.reg_party.0)
162                .collect();
163
164            let (aggr_vk, aggr_sig) = BlsSignature::aggregate(&grouped_vks, &grouped_sigs).unwrap();
165            aggr_sigs.push(aggr_sig);
166            aggr_vks.push(aggr_vk);
167        }
168
169        let concat_msgs: Vec<Vec<u8>> = msgs
170            .iter()
171            .zip(avks.iter())
172            .map(|(msg, avk)| avk.get_merkle_tree_batch_commitment().concatenate_with_message(msg))
173            .collect();
174
175        BlsSignature::batch_verify_aggregates(&concat_msgs, &aggr_vks, &aggr_sigs)?;
176        Ok(())
177    }

179    /// Convert concatenation proof to bytes
180    /// # Layout
181    /// * Number of the pairs of Signatures and Registered Parties (SigRegParty) (as u64)
182    /// * Pairs of Signatures and Registered Parties (prefixed with their size as u64)
183    /// * Batch proof
184    pub fn to_bytes(&self) -> Vec<u8> {
185        let mut out = Vec::new();
186        out.extend_from_slice(&u64::try_from(self.signatures.len()).unwrap().to_be_bytes());
187        for sig_reg in &self.signatures {
188            out.extend_from_slice(&u64::try_from(sig_reg.to_bytes().len()).unwrap().to_be_bytes());
189            out.extend_from_slice(&sig_reg.to_bytes());
190        }
191        let proof = &self.batch_proof;
192        out.extend_from_slice(&proof.to_bytes());
193
194        out
195    }
196
197    ///Extract a concatenation proof from a byte slice.
198    pub fn from_bytes(
199        bytes: &[u8],
200    ) -> Result<ConcatenationProof<D>, StmAggregateSignatureError<D>> {
201        let mut bytes_index = 0;
202
203        let mut u64_bytes = [0u8; 8];
204        u64_bytes.copy_from_slice(
205            bytes
206                .get(bytes_index..bytes_index + 8)
207                .ok_or(StmAggregateSignatureError::SerializationError)?,
208        );
209        let total_sigs = usize::try_from(u64::from_be_bytes(u64_bytes))
210            .map_err(|_| StmAggregateSignatureError::SerializationError)?;
211        bytes_index += 8;
212
213        let mut sig_reg_list = Vec::with_capacity(total_sigs);
214        for _ in 0..total_sigs {
215            u64_bytes.copy_from_slice(
216                bytes
217                    .get(bytes_index..bytes_index + 8)
218                    .ok_or(StmAggregateSignatureError::SerializationError)?,
219            );
220            let sig_reg_size = usize::try_from(u64::from_be_bytes(u64_bytes))
221                .map_err(|_| StmAggregateSignatureError::SerializationError)?;
222            let sig_reg = SingleSignatureWithRegisteredParty::from_bytes::<D>(
223                bytes
224                    .get(bytes_index + 8..bytes_index + 8 + sig_reg_size)
225                    .ok_or(StmAggregateSignatureError::SerializationError)?,
226            )?;
227            bytes_index += 8 + sig_reg_size;
228            sig_reg_list.push(sig_reg);
229        }
230
231        let batch_proof = MerkleBatchPath::from_bytes(
232            bytes
233                .get(bytes_index..)
234                .ok_or(StmAggregateSignatureError::SerializationError)?,
235        )?;
236
237        Ok(ConcatenationProof {
238            signatures: sig_reg_list,
239            batch_proof,
240        })
241    }
