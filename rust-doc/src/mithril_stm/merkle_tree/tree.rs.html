12/// Tree of hashes, providing a commitment of data and its ordering.
13#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
14pub struct MerkleTree<D: Digest> {
15    /// The nodes are stored in an array heap:
16    /// * `nodes[0]` is the root,
17    /// * the parent of `nodes[i]` is `nodes[(i-1)/2]`
18    /// * the children of `nodes[i]` are `{nodes[2i + 1], nodes[2i + 2]}`
19    /// * All nodes have size `Output<D>::output_size()`, even leafs (which are hashed before committing them).
20    nodes: Vec<Vec<u8>>,
21    /// The leaves begin at `nodes[leaf_off]`.
22    leaf_off: usize,
23    /// Number of leaves cached in the merkle tree.
24    n: usize,
25    /// Phantom type to link the tree with its hasher
26    hasher: PhantomData<D>,
27}
28
29impl<D: Digest + FixedOutput> MerkleTree<D> {
30    /// Provided a non-empty list of leaves, `create` generates its corresponding `MerkleTree`.
31    pub(crate) fn new(leaves: &[MerkleTreeLeaf]) -> MerkleTree<D> {
32        let n = leaves.len();
33        assert!(n > 0, "MerkleTree::create() called with no leaves");
34
35        let num_nodes = n + n.next_power_of_two() - 1;
36
37        let mut nodes = vec![vec![0u8]; num_nodes];
38
39        for i in 0..leaves.len() {
40            nodes[num_nodes - n + i] = D::digest(leaves[i].to_bytes()).to_vec();
41        }
42
43        for i in (0..num_nodes - n).rev() {
44            let z = D::digest([0u8]).to_vec();
45            let left = if left_child(i) < num_nodes {
46                &nodes[left_child(i)]
47            } else {
48                &z
49            };
50            let right = if right_child(i) < num_nodes {
51                &nodes[right_child(i)]
52            } else {
53                &z
54            };
55            nodes[i] = D::new().chain_update(left).chain_update(right).finalize().to_vec();
56        }
57
58        Self {
59            nodes,
60            n,
61            leaf_off: num_nodes - n,
62            hasher: PhantomData,
63        }
64    }
65

93    /// Get a path for a batch of leaves. The indices must be ordered. We use the Octopus algorithm to
94    /// avoid redundancy with nodes in the path. Let `x1, . . . , xk` be the indices of elements we
95    /// want to produce an opening for. The algorithm takes as input `x1, . . ., xk`, and  proceeds as follows:
96    /// 1. Initialise the proof vector, `proof = []`.
97    /// 2. Given an input vector `v = v1, . . .,vl`, if `v.len() == 1`, return `proof`, else, continue.
98    /// 3. Map each `vi` to the corresponding number of the leaf (by adding the offset).
99    /// 4. Initialise a new empty vector `p = []`. Next, iterate over each element `vi`
100    ///    a. Append the parent of `vi` to `p`
101    ///    b. Compute the sibling, `si` of `vi`
102    ///    c. If `si == v(i+1)` then do nothing, and skip step four for `v(i+1)`. Else append `si` to `proof`
103    /// 5. Iterate from step 2 with input vector `p`
104    ///
105    /// # Panics
106    /// If the indices provided are out of bounds (higher than the number of elements
107    /// committed in the `MerkleTree`) or are not ordered, the function fails.
108    // todo: Update doc.
109    pub(crate) fn compute_merkle_tree_batch_path(&self, indices: Vec<usize>) -> MerkleBatchPath<D>
110    where
111        D: FixedOutput,
112    {
113        assert!(
114            !indices.is_empty(),
115            "get_batched_path() called with no indices"
116        );
117        for i in &indices {
118            assert!(
119                i < &self.n,
120                "Proof index out of bounds: asked for {} out of {}",
121                i,
122                self.n
123            );
124        }
125
126        let mut ordered_indices: Vec<usize> = indices.clone();
127        ordered_indices.sort_unstable();
128
129        assert_eq!(ordered_indices, indices, "Indices should be ordered");
130
131        ordered_indices = ordered_indices.into_iter().map(|i| self.get_leaf_index(i)).collect();
132
133        let mut idx = ordered_indices[0];
134        let mut proof = Vec::new();
135
136        while idx > 0 {
137            let mut new_indices = Vec::with_capacity(ordered_indices.len());
138            let mut i = 0;
139            idx = parent(idx);
140            while i < ordered_indices.len() {
141                new_indices.push(parent(ordered_indices[i]));
142                let sibling = sibling(ordered_indices[i]);
143                if i < ordered_indices.len() - 1 && ordered_indices[i + 1] == sibling {
144                    i += 1;
145                } else if sibling < self.nodes.len() {
146                    proof.push(self.nodes[sibling].clone());
147                }
148                i += 1;
149            }
150            ordered_indices.clone_from(&new_indices);
151        }
152
153        MerkleBatchPath::new(proof, indices)
154    }

156    /// Get a path for a batch of leaves. The indices must be ordered. We use the Octopus algorithm to
157    /// avoid redundancy with nodes in the path. Let `x1, . . . , xk` be the indices of elements we
158    /// want to produce an opening for. The algorithm takes as input `x1, . . ., xk`, and  proceeds as follows:
159    /// 1. Initialise the proof vector, `proof = []`.
160    /// 2. Given an input vector `v = v1, . . .,vl`, if `v.len() == 1`, return `proof`, else, continue.
161    /// 3. Map each `vi` to the corresponding number of the leaf (by adding the offset).
162    /// 4. Initialise a new empty vector `p = []`. Next, iterate over each element `vi`
163    ///    a. Append the parent of `vi` to `p`
164    ///    b. Compute the sibling, `si` of `vi`
165    ///    c. If `si == v(i+1)` then do nothing, and skip step four for `v(i+1)`. Else append `si` to `proof`
166    /// 5. Iterate from step 2 with input vector `p`

170    /// committed in the `MerkleTree`) or are not ordered, the function fails.

180    /// Convert a `MerkleTree` into a byte string, containing $4 + n * S$ bytes where $n$ is the
181    /// number of nodes and $S$ the output size of the hash function.
182    /// # Layout
183    /// * Number of leaves committed in the Merkle Tree (as u64)
184    /// * All nodes of the merkle tree (starting with the root)
185    pub fn to_bytes(&self) -> Vec<u8> {
186        let mut result = Vec::with_capacity(8 + self.nodes.len() * <D as Digest>::output_size());
187        result.extend_from_slice(&u64::try_from(self.n).unwrap().to_be_bytes());
188        for node in self.nodes.iter() {
189            result.extend_from_slice(node);
190        }
191        result
192    }
193
194    /// Try to convert a byte string into a `MerkleTree`.
195    /// # Error
196    /// It returns error if conversion fails.
197    pub fn from_bytes(bytes: &[u8]) -> Result<Self, MerkleTreeError<D>> {
198        let mut u64_bytes = [0u8; 8];
199        u64_bytes.copy_from_slice(bytes.get(..8).ok_or(MerkleTreeError::SerializationError)?);
200        let n = usize::try_from(u64::from_be_bytes(u64_bytes))
201            .map_err(|_| MerkleTreeError::SerializationError)?;
202        let num_nodes = n + n.next_power_of_two() - 1;
203        let mut nodes = Vec::with_capacity(num_nodes);
204        for i in 0..num_nodes {
205            nodes.push(
206                bytes
207                    .get(
208                        8 + i * <D as Digest>::output_size()
209                            ..8 + (i + 1) * <D as Digest>::output_size(),
210                    )
211                    .ok_or(MerkleTreeError::SerializationError)?
212                    .to_vec(),
213            );
214        }
215        Ok(Self {
216            nodes,
217            leaf_off: num_nodes - n,
218            n,
219            hasher: PhantomData,
220        })
221    }

234    /// Get a path (hashes of siblings of the path to the root node)
235    /// for the `i`th value stored in the tree.
236    /// Requires `i < self.n`
237    pub(crate) fn compute_merkle_tree_path(&self, i: usize) -> MerklePath<D> {
238        assert!(
239            i < self.n,
240            "Proof index out of bounds: asked for {} out of {}",
241            i,
242            self.n
243        );
244        let mut idx = self.get_leaf_index(i);
245        let mut proof = Vec::new();
246
247        while idx > 0 {
248            let h = if sibling(idx) < self.nodes.len() {
249                self.nodes[sibling(idx)].clone()
250            } else {
251                D::digest([0u8]).to_vec()
252            };
253            proof.push(h.clone());
254            idx = parent(idx);
255        }
256
257        MerklePath::new(proof, i)
258    }

264    pub fn get_path(&self, i: usize) -> MerklePath<D> {
265        Self::compute_merkle_tree_path(self, i)
266    }

387        fn arb_tree_arb_batch(max_size: u32)
388                   (v in vec(any::<u64>(), 2..max_size as usize)) -> (MerkleTree<Blake2b<U32>>, Vec<MerkleTreeLeaf>, Vec<usize>) {
389            let mut rng = rng();
390            let size = v.len();
391            let pks = vec![BlsVerificationKey::default(); size];
392            let leaves = pks.into_iter().zip(v.into_iter()).map(|(key, stake)| MerkleTreeLeaf(key, stake)).collect::<Vec<MerkleTreeLeaf>>();
393
394            let indices: Vec<usize> = (0..size).collect();
395            let mut mt_list: Vec<usize> = indices.into_iter().choose_multiple(&mut rng, size * 2 / 10 + 1);
396            mt_list.sort_unstable();
397
398            let mut batch_values: Vec<MerkleTreeLeaf> = Vec::with_capacity(mt_list.len());
399            for i in mt_list.iter() {
400                batch_values.push(leaves[*i]);
401            }
402
403            (MerkleTree::<Blake2b<U32>>::new(&leaves), batch_values, mt_list)
404        }
405    }
