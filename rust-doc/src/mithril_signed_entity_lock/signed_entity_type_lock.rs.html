1//! # Signed Entity Type Lock
2//!
3//! This module provides a non-blocking lock mechanism for signed entity types to prevent multiple
4//! modifications on a same entity type at the same time.
5
6use std::collections::BTreeSet;
7
8use tokio::sync::RwLock;
9
10use mithril_common::entities::SignedEntityTypeDiscriminants;
11
12/// Non-blocking lock mechanism for signed entity types to prevent multiple
13/// modifications on a same entity type at the same time.
14pub struct SignedEntityTypeLock {
15    locked_entities: RwLock<BTreeSet<SignedEntityTypeDiscriminants>>,
16}
17
18impl SignedEntityTypeLock {
19    /// Create a new instance of `SignedEntityLock` without any signed entity locked.
20    pub fn new() -> Self {
21        Self {
22            locked_entities: RwLock::new(BTreeSet::new()),
23        }
24    }
25
26    /// Check if a signed entity is locked.
27    pub async fn is_locked<T: Into<SignedEntityTypeDiscriminants>>(&self, entity_type: T) -> bool {
28        let locked_entities = self.locked_entities.read().await;
29        locked_entities.contains(&entity_type.into())
30    }
31
32    /// Lock a signed entity.
33    ///
34    /// If the entity is already locked, this function does nothing.
35    pub async fn lock<T: Into<SignedEntityTypeDiscriminants>>(&self, entity_type: T) {
36        let mut locked_entities = self.locked_entities.write().await;
37        locked_entities.insert(entity_type.into());
38    }
39
40    /// Release a locked signed entity.
41    ///
42    /// If the entity is not locked, this function does nothing.
43    pub async fn release<T: Into<SignedEntityTypeDiscriminants>>(&self, entity_type: T) {
44        let mut locked_entities = self.locked_entities.write().await;
45        locked_entities.remove(&entity_type.into());
46    }
47
48    /// Check if there are any locked signed entities.
49    pub async fn has_locked_entities(&self) -> bool {
50        let locked_entities = self.locked_entities.read().await;
51        !locked_entities.is_empty()
52    }
53
54    /// List only the unlocked signed entities in the given list.
55    pub async fn filter_unlocked_entries<T: Into<SignedEntityTypeDiscriminants> + Clone>(
56        &self,
57        entries: Vec<T>,
58    ) -> Vec<T> {
59        let locked_entities = self.locked_entities.read().await;
60        entries
61            .into_iter()
62            .filter(|entry| !locked_entities.contains(&entry.clone().into()))
63            .collect()
64    }
65}
66
67impl Default for SignedEntityTypeLock {
68    /// Create a new instance of `SignedEntityLock` without any signed entity locked.
69    fn default() -> Self {
70        Self::new()
71    }
72}

74#[cfg(test)]
75mod tests {
76    use super::*;
77
78    #[tokio::test]
79    async fn by_default_entity_is_not_locked() {
80        let signed_entity_type_lock = SignedEntityTypeLock::new();
81        for entity_type in SignedEntityTypeDiscriminants::all() {
82            assert!(!signed_entity_type_lock.is_locked(entity_type).await);
83        }
84    }
85
86    #[tokio::test]
87    async fn lock_a_signed_entity() {
88        for entity_type in SignedEntityTypeDiscriminants::all() {
89            let signed_entity_type_lock = SignedEntityTypeLock::new();
90            signed_entity_type_lock.lock(entity_type).await;
91            assert!(signed_entity_type_lock.is_locked(entity_type).await);
92        }
93    }
94
95    #[tokio::test]
96    async fn locking_an_entity_more_than_once_in_a_row_dont_crash() {
97        let entity = SignedEntityTypeDiscriminants::MithrilStakeDistribution;
98        let signed_entity_type_lock = SignedEntityTypeLock::new();
99
100        signed_entity_type_lock.lock(entity).await;
101        signed_entity_type_lock.lock(entity).await;
102        signed_entity_type_lock.lock(entity).await;
103
104        assert!(signed_entity_type_lock.is_locked(entity).await);
105    }
106
107    #[tokio::test]
108    async fn locking_a_signed_entity_does_not_lock_other_entity() {
109        let signed_entity_type_lock = SignedEntityTypeLock::new();
110        signed_entity_type_lock
111            .lock(SignedEntityTypeDiscriminants::CardanoImmutableFilesFull)
112            .await;
113        assert!(
114            signed_entity_type_lock
115                .is_locked(SignedEntityTypeDiscriminants::CardanoImmutableFilesFull)
116                .await
117        );
118        assert!(
119            !signed_entity_type_lock
120                .is_locked(SignedEntityTypeDiscriminants::MithrilStakeDistribution)
121                .await
122        );
123    }
124
125    #[tokio::test]
126    async fn releasing_an_locked_entity() {
127        let entity = SignedEntityTypeDiscriminants::MithrilStakeDistribution;
128        let signed_entity_type_lock = SignedEntityTypeLock::new();
129
130        signed_entity_type_lock.lock(entity).await;
131        signed_entity_type_lock.release(entity).await;
132
133        assert!(!signed_entity_type_lock.is_locked(entity).await);
134    }
135
136    #[tokio::test]
137    async fn releasing_an_already_released_entity() {
138        let entity = SignedEntityTypeDiscriminants::MithrilStakeDistribution;
139        let signed_entity_type_lock = SignedEntityTypeLock::new();
140
141        assert!(!signed_entity_type_lock.is_locked(entity).await);
142
143        signed_entity_type_lock.release(entity).await;
144
145        assert!(!signed_entity_type_lock.is_locked(entity).await);
146    }
147
148    #[tokio::test]
149    async fn releasing_a_signed_entity_does_not_release_other_entity() {
150        let released_entity = SignedEntityTypeDiscriminants::MithrilStakeDistribution;
151        let signed_entity_type_lock = SignedEntityTypeLock::new();
152
153        for entity in SignedEntityTypeDiscriminants::all() {
154            signed_entity_type_lock.lock(entity).await;
155        }
156
157        signed_entity_type_lock.release(released_entity).await;
158
159        assert!(!signed_entity_type_lock.is_locked(released_entity).await);
160        assert!(
161            signed_entity_type_lock
162                .is_locked(SignedEntityTypeDiscriminants::CardanoImmutableFilesFull)
163                .await
164        );
165        assert!(
166            signed_entity_type_lock
167                .is_locked(SignedEntityTypeDiscriminants::CardanoTransactions)
168                .await
169        );
170    }
171
172    #[tokio::test]
173    async fn filters_out_locked_entities() {
174        let signed_entity_type_lock = SignedEntityTypeLock::new();
175        let signed_entities: Vec<_> = vec![
176            SignedEntityTypeDiscriminants::MithrilStakeDistribution,
177            SignedEntityTypeDiscriminants::CardanoTransactions,
178        ];
179
180        assert_eq!(
181            signed_entity_type_lock
182                .filter_unlocked_entries(signed_entities.clone())
183                .await,
184            signed_entities.clone()
185        );
186
187        signed_entity_type_lock
188            .lock(SignedEntityTypeDiscriminants::MithrilStakeDistribution)
189            .await;
190
191        assert_eq!(
192            signed_entity_type_lock
193                .filter_unlocked_entries(signed_entities.clone())
194                .await,
195            vec![SignedEntityTypeDiscriminants::CardanoTransactions]
196        );
197    }
198
199    #[tokio::test]
200    async fn has_locked_entities() {
201        let signed_entity_type_lock = SignedEntityTypeLock::new();
202
203        assert!(!signed_entity_type_lock.has_locked_entities().await);
204
205        signed_entity_type_lock
206            .lock(SignedEntityTypeDiscriminants::MithrilStakeDistribution)
207            .await;
208        assert!(signed_entity_type_lock.has_locked_entities().await);
209    }
