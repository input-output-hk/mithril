17/// A client to send HTTP requests to a Mithril Aggregator
18pub struct AggregatorClient {
19    pub(super) aggregator_endpoint: Url,
20    pub(super) api_version_provider: APIVersionProvider,
21    pub(super) additional_headers: HeaderMap,
22    pub(super) timeout_duration: Option<Duration>,
23    pub(super) client: reqwest::Client,
24    pub(super) logger: Logger,
25}
26
27impl AggregatorClient {
28    /// Creates a [AggregatorClientBuilder] to configure a `AggregatorClient`.
29    //
30    // This is the same as `AggregatorClient::builder()`.
31    pub fn builder<U: IntoUrl>(aggregator_url: U) -> AggregatorClientBuilder {
32        AggregatorClientBuilder::new(aggregator_url)
33    }
34
35    /// Send the given query to the Mithril Aggregator
36    pub async fn send<Q: AggregatorQuery>(&self, query: Q) -> AggregatorClientResult<Q::Response> {
37        // Todo: error handling ? Reuse the version in `warn_if_api_version_mismatch` ?
38        let current_api_version = self.api_version_provider.compute_current_version().unwrap();
39        let mut request_builder = match Q::method() {
40            QueryMethod::Get => self.client.get(self.join_aggregator_endpoint(&query.route())?),
41            QueryMethod::Post => self.client.post(self.join_aggregator_endpoint(&query.route())?),
42        }
43        .headers(self.additional_headers.clone())
44        .header(MITHRIL_API_VERSION_HEADER, current_api_version.to_string());
45
46        if let Some(body) = query.body() {
47            request_builder = request_builder.json(&body);
48        }
49
50        if let Some(timeout) = self.timeout_duration {
51            request_builder = request_builder.timeout(timeout);
52        }
53
54        match request_builder.send().await {
55            Ok(response) => {
56                self.warn_if_api_version_mismatch(&response);
57
58                let context = QueryContext {
59                    response,
60                    logger: self.logger.clone(),
61                };
62                query.handle_response(context).await
63            }
64            Err(err) => Err(AggregatorClientError::RemoteServerUnreachable(anyhow!(err))),
65        }
66    }
67
68    fn join_aggregator_endpoint(&self, endpoint: &str) -> AggregatorClientResult<Url> {
69        self.aggregator_endpoint
70            .join(endpoint)
71            .with_context(|| {
72                format!(
73                    "Invalid url when joining given endpoint, '{endpoint}', to aggregator url '{}'",
74                    self.aggregator_endpoint
75                )
76            })
77            .map_err(AggregatorClientError::InvalidEndpoint)
78    }
79
80    /// Check API version mismatch and log a warning if the aggregator's version is more recent.
81    fn warn_if_api_version_mismatch(&self, response: &Response) {
82        let remote_aggregator_version = response
83            .headers()
84            .get(MITHRIL_API_VERSION_HEADER)
85            .and_then(|v| v.to_str().ok())
86            .and_then(|s| Version::parse(s).ok());
87
88        let client_version = self.api_version_provider.compute_current_version();
89
90        match (remote_aggregator_version, client_version) {
91            (Some(aggregator), Ok(client)) if client < aggregator => {
92                warn!(self.logger, "{}", API_VERSION_MISMATCH_WARNING_MESSAGE;
93                    "remote_aggregator_version" => %aggregator,
94                    "caller_version" => %client,
95                );
96            }
97            (Some(_), Err(error)) => {
98                error!(
99                    self.logger,
100                    "Failed to compute the current API version";
101                    "error" => error.to_string()
102                );
103            }
104            _ => {}
105        }
106    }

306        fn build_fake_response_with_header<K: Into<String>, V: Into<String>>(
307            key: K,
308            value: V,

317        fn assert_api_version_warning_logged<A: Into<String>, S: Into<String>>(
318            log_inspector: &MemoryDrainForTestInspector,
319            aggregator_version: A,
320            client_version: S,
321        ) {
322            assert!(log_inspector.contains_log(API_VERSION_MISMATCH_WARNING_MESSAGE));
323            assert!(log_inspector.contains_log(&format!(
324                "remote_aggregator_version={}",
325                aggregator_version.into()
326            )));
327            assert!(
328                log_inspector.contains_log(&format!("caller_version={}", client_version.into()))
329            );
330        }
