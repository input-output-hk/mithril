21/// This is responsible for creating new instances of ProtocolInitializer.
22pub struct MithrilProtocolInitializerBuilder {}
23
24impl MithrilProtocolInitializerBuilder {
25    /// Create a ProtocolInitializer instance.
26    pub fn build(
27        stake: &Stake,
28        protocol_parameters: &ProtocolParameters,
29        kes_signer: Option<Arc<dyn KesSigner>>,
30        kes_period: Option<KesPeriod>,
31    ) -> StdResult<ProtocolInitializer> {
32        let mut rng = rand_core::OsRng;
33        let protocol_initializer = ProtocolInitializer::setup(
34            protocol_parameters.to_owned().into(),
35            kes_signer,
36            kes_period,
37            stake.to_owned(),
38            &mut rng,
39        )?;
40
41        Ok(protocol_initializer)
42    }
43}

45/// The SingleSigner is the structure responsible for issuing [SingleSignature].

48pub trait SingleSigner: Sync + Send {

50    async fn compute_single_signature(
51        &self,
52        protocol_message: &ProtocolMessage,

56    fn get_party_id(&self) -> PartyId;
57}

59/// SingleSigner error structure.
60#[derive(Error, Debug)]
61pub enum SingleSignerError {
62    /// Cryptographic Signer creation error.
63    #[error("the protocol signer creation failed")]
64    ProtocolSignerCreationFailure(#[source] StdError),
65
66    /// Signature Error
67    #[error("Signature Error")]
68    SignatureFailed(#[source] StdError),
69
70    /// Avk computation Error
71    #[error("Aggregate verification key computation Error")]
72    AggregateVerificationKeyComputationFailed(#[source] StdError),
73}

75/// Implementation of the SingleSigner.
76pub struct MithrilSingleSigner {
77    party_id: PartyId,
78    epoch_service: EpochServiceWrapper,
79    logger: Logger,
80}

82impl MithrilSingleSigner {
83    /// Create a new instance of the MithrilSingleSigner.
84    pub fn new(party_id: PartyId, epoch_service: EpochServiceWrapper, logger: Logger) -> Self {
85        Self {
86            party_id,
87            epoch_service,
88            logger: logger.new_with_component_name::<Self>(),
89        }
90    }
91
92    async fn build_protocol_single_signer(&self) -> StdResult<ProtocolSingleSigner> {
93        let epoch_service = self.epoch_service.read().await;
94        let protocol_initializer =
95            epoch_service.protocol_initializer()?.as_ref().with_context(|| {
96                format!(
97                    "Can not Sign or Compute AVK, No protocol initializer found for party_id: '{}'",
98                    self.party_id.clone()
99                )
100            })?;
101
102        #[cfg(not(feature = "future_snark"))]
103        let protocol_initializer = protocol_initializer.clone();
104        #[cfg(feature = "future_snark")]
105        let mut protocol_initializer = protocol_initializer.clone();
106
107        let current_signers_with_stake = epoch_service.current_signers_with_stake().await?;
108
109        #[cfg(feature = "future_snark")]
110        let current_signers_with_stake = {
111            if epoch_service.mithril_era()? == SupportedEra::Pythagoras {
112                protocol_initializer.strip_snark_keys();
113                SignerWithStake::strip_snark_fields(current_signers_with_stake)
114            } else {
115                current_signers_with_stake
116            }
117        };
118
119        let builder = SignerBuilder::new(
120            &current_signers_with_stake,
121            &protocol_initializer.get_protocol_parameters().into(),
122        )
123        .with_context(|| "Mithril Single Signer can not build signer")
124        .map_err(SingleSignerError::ProtocolSignerCreationFailure)?;
125
126        let single_signer = builder
127            .restore_signer_from_initializer(self.party_id.clone(), protocol_initializer)
128            .with_context(|| {
129                format!(
130                    "Mithril Single Signer can not restore signer with party_id: '{}'",
131                    self.party_id.clone()
132                )
133            })
134            .map_err(SingleSignerError::ProtocolSignerCreationFailure)?;
135
136        Ok(single_signer)
137    }
138}

140#[async_trait]
141impl SingleSigner for MithrilSingleSigner {
142    async fn compute_single_signature(
143        &self,
144        protocol_message: &ProtocolMessage,
145    ) -> StdResult<Option<SingleSignature>> {
146        let protocol_single_signer = self.build_protocol_single_signer().await?;
147
148        info!(
149            self.logger, "Signing protocol message";
150            "protocol_message" =>  #?protocol_message,
151            "signed message" => protocol_message.compute_hash().encode_hex::<String>()
152        );
153        let signature = protocol_single_signer
154            .sign(protocol_message)
155            .with_context(|| {
156                format!(
157                    "Mithril Single Signer can not sign protocol_message: '{protocol_message:?}'"
158                )
159            })
160            .map_err(SingleSignerError::SignatureFailed)?;
161
162        match &signature {
163            Some(signature) => {
164                trace!(
165                    self.logger,
166                    "Party #{}: lottery #{:?} won", signature.party_id, &signature.won_indexes
167                );
168            }
169            None => {
170                warn!(
171                    self.logger,
172                    "No signature computed, all lotteries were lost"
173                );
174            }
175        };
176
177        Ok(signature)
178    }
179
180    /// Get party id
181    fn get_party_id(&self) -> PartyId {
182        self.party_id.clone()
183    }
184}

252    async fn sign_and_verify(
253        single_signer: &MithrilSingleSigner,
254        fixture: &mithril_common::test::builder::MithrilFixture,
255    ) {
256        let current_signer = &fixture.signers_fixture()[0];
257        let clerk = ProtocolClerk::new_clerk_from_signer(&current_signer.protocol_signer);
258        let avk = clerk.compute_aggregate_verification_key();
259
260        let mut protocol_message = ProtocolMessage::new();
261        protocol_message
262            .set_message_part(ProtocolMessagePartKey::SnapshotDigest, "digest".to_string());
263        let sign_result = single_signer
264            .compute_single_signature(&protocol_message)
265            .await
266            .expect("single signer should not fail")
267            .expect("single signer should produce a signature here");
268
269        let expected_message = protocol_message.compute_hash().as_bytes().to_vec();
270        let decoded_sig = sign_result.to_protocol_signature();
271        assert!(
272            decoded_sig
273                .verify(
274                    &fixture.protocol_parameters().into(),
275                    &current_signer.protocol_signer.get_bls_verification_key(),
276                    &current_signer.protocol_signer.get_stake(),
277                    &avk,
278                    &expected_message,
279                    #[cfg(feature = "future_snark")]
280                    None,
281                )
282                .is_ok(),
283            "produced single signature should be valid"
284        );
285    }
